\begin{definition}
    Let $\Omega$ be the sample space (...) to an experiment and $\mathbb{F}$ a $\sigma$-algebra of subsets of $\Omega$. (...)
    \begin{enumerate}
        \item $P(A) \geq 0$ for all $A \in \mathbb{F}$.
        \item $P(\Omega) = 1$.
        \item If $A_1, A_2, \dots$ is a collection of pairwise disjoint subsets of $\Omega$ that belong to F then are sets in $\mathbb{F}$ such that $A_i \subseteq A_{i+1}$ for all $i$, then $P(\cup_{i=1}^\infty A_i) = \sum_{i=1}^\infty P(A_i)$. (Pairwise disjoint $A_k \cap A_l = \phi$ if $k \neq l$)
    \end{enumerate}
    Properties:
    \begin{enumerate}
        \item If $A \subseteq B$ then $P(A) \leq P(B)$. Proof: $B = A \cup (B\setminus A)$ and $P(B) = P(A) + P(B\setminus A)$ since $A$ and $B\setminus A$ are disjoint. $P(B) = P(A) + P(B\setminus A) \geq P(A)$ since $P(B\setminus A) \geq 0$.
    \end{enumerate}
\end{definition}
\begin{corollary}
    Let $A \in \mathbb{F}$. (A is an event). Then $P(A) \leq 1$. Proof: $A \subseteq \Omega$, (...) by the previous proof, $P(A) \leq P(\Omega) = 1$.
\end{corollary}
\begin{proposition}
    $P(A^c) = 1 - P(A)$. Proof: $\Omega = A \cup A^c$ and $A \cap A^c = \phi$. $P(\Omega) = P(A \cup A^c) = P(A) + P(A^c) = 1$, so $P(A^c) = 1 - P(A)$.
\end{proposition}
Let $A, B$ be events $(A, B \in \mathbb{F})$ . Then $A \cap B \in \mathbb{F}$ and $P(A\cup B) = P(A) + P(B) - P(A\cap B)$. $A \cup B = A \cup (B \cap A^c)$, $A \cap (B\cap A^c) = \phi$, $P(A\cup B) = P(A) + P(B\cap A^c)$.
\begin{lemma}
    $P(B) = P(A^c\cap B) + P(A\cap B)$ and $P(A \cap B) = P(B) - P(A^c\cap B)$, $B = (A\cap B)\cup (A^c\cap B)$, $A\cap B$, $A^c \cap B$ are disjoint $P(B) = P(A\cap B)+P(A^c \cap B)$ and hence $P(A\cap B) = P(B) - P(A\cap B)$ and hence $P(A\cup B) = P(A) + P(B) - P(A\cap B)$.
\end{lemma}
