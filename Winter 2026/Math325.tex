\section{Prerequisite Knowledge}
\subsection{Real Analysis}
\subsubsection{Norms}
\begin{definition}[Norm]
    A norm $\|\cdot\|$ on $\R^n$ is a way of measuring the 
    ``size'' or ``length'' of a vector $x \in \R^n$. It must satisfy:
    \begin{itemize}
        \item $\|x\| \geq 0$, and $\|x\| = 0$ only if $x = 0$,
        \item $\|\alpha x\| = |\alpha|\,\|x\|$ for any scalar $\alpha$,
        \item $\|x + y\| \leq \|x\| + \|y\|$ \quad (triangle inequality).
    \end{itemize}
\end{definition}
\begin{example}
    The three most common norms on $\R^n$ are:
    \begin{itemize}
        \item \textbf{Euclidean norm:} $\|x\|_2 = \sqrt{x_1^2 + \cdots + x_n^2}$ 
        (standard geometric length),
        \item \textbf{Supremum norm:} $\|x\|_\infty = \max\{|x_1|,\ldots,|x_n|\}$ 
        (size of the largest component),
        \item \textbf{$\ell^1$ norm:} $\|x\|_1 = |x_1| + \cdots + |x_n|$ 
        (sum of all components, also called the taxicab norm).
    \end{itemize}
\end{example}
\begin{remark}
    In this course, any norm can be used unless stated otherwise. The choice 
    of norm does not affect whether a solution exists or is unique.
\end{remark}

\subsubsection{Sequences and Convergence}
\begin{definition}[Cauchy Sequence]
    A sequence of vectors $\{x_k\} \subset \R^n$ is called a Cauchy 
    sequence if its terms eventually get arbitrarily close to one another:
    for every $\varepsilon > 0$, there exists $N \in \mathbb{N}$ such that
    \[\|x_m - x_n\| < \varepsilon \quad \text{for all } m, n \geq N.\]
\end{definition}
\begin{theorem}
    Every Cauchy sequence in $\R^n$ converges to a point in $\R^n$.
\end{theorem}
\begin{remark}
    This is the key fact used in Picard iteration: we show that the sequence 
    of approximate solutions $\{y_k\}$ is Cauchy, and therefore it must 
    converge to an actual solution.
\end{remark}

\subsubsection{Uniform Convergence}
\begin{definition}[Uniform Convergence]
    A sequence of functions $f_k : J \to \R^n$ converges 
    \emph{uniformly} to $f : J \to \R^n$ if
    \[\lim_{k \to \infty} \sup_{t \in J} \|f_k(t) - f(t)\| = 0.\]
    In plain terms: the approximation error $\|f_k(t) - f(t)\|$ becomes 
    small \emph{simultaneously} for all $t \in J$, not just pointwise.
\end{definition}
\begin{theorem}
    If $f_k \to f$ uniformly on $J$ and each $f_k$ is continuous, then $f$ 
    is also continuous. Moreover,
    \[\lim_{k\to\infty}\int_J f_k(t)\,dt = \int_J f(t)\,dt.\]
\end{theorem}
\begin{remark}
    The second statement says that for uniformly convergent sequences, 
    you can swap the limit and the integral. This is used in the Picard 
    iteration proof to pass the limit inside the integral and confirm that 
    the limit function actually solves the IVP.
\end{remark}

\subsubsection{The Mean Value Inequality}
\label{MVI}
\begin{theorem}[Mean Value Inequality]
    Let $U \subseteq \R^n$ be open and $f : U \to \R^n$ 
    be $C^1$. For any compact set $K \subset U$ and all $x, y \in K$,
    \[\|f(y) - f(x)\| \leq \left(\sup_{z \in K}\|Df(z)\|\right)\|y - x\|,\]
    where $Df$ denotes the Jacobian matrix of $f$.
\end{theorem}
\begin{remark}
    This generalizes the familiar scalar fact $|f(b)-f(a)| \leq 
    \sup|f'|\cdot|b-a|$ to higher dimensions. Its main consequence for 
    this course is that any $C^1$ function is locally Lipschitz: on a 
    compact set $K$, the sup of $\|Df\|$ is finite, providing a Lipschitz 
    constant. This is why the hypothesis ``$f$ is $C^1$'' is often used 
    as a sufficient condition for the Existence and Uniqueness Theorem.
\end{remark}

\subsection{Intro}
\marginpar{January 06, 2026.}
\begin{definition}[Differential Equation]
    A differential equation (DE) is a relation that involves an unknown function and some of its derivatives.
\end{definition}
\begin{definition}[Ordinary Differential Equation]
    An ordinary differential equation (ODE) is a differential equation whose unknown function depends on one independent variable only.
\end{definition}
\begin{definition}[Partial Differential Equation]
    A partial differential equation (PDE) is a differential equation whose unknown function depends on more than one independent variable. \textbf{Will not be taught in this course.}
\end{definition}

\begin{example}[Motion under gravity and linear drag]
    A ball of mass $m$ falls subject to gravity and linear air resistance. Let $v(t)$ be its velocity (positive downward). The net force gives
    \[m\frac{dv}{dt} = mg - \lambda v,\]
    where $\lambda \geq 0$ is the drag coefficient. Writing $v = y'$, this becomes the second-order ODE
    \[my'' + \lambda y' = mg.\]
\end{example}
\begin{example}[The Malthusian Growth Model]
    Let $N(t)$ be a population size. Assuming the rate of change is 
    proportional to the current population,
    \[\frac{dN}{dt} = rN,\]
    where $r$ is the growth rate ($r>0$) or decay rate ($r<0$). Separating variables and applying the initial condition $N(0)=N_0$ gives the unique solution
    \[N(t) = N_0 e^{rt}.\]
\end{example}
\begin{example}[The Logistic Equation]
    If the growth rate depends on the population itself, with carrying capacity $K$, the simplest choice $R(N) = r(1-N/K)$ gives
    \[\frac{dN}{dt} = rN\!\left(1 - \frac{N}{K}\right),\]
    called the logistic equation. It models growth when $N < K$ and decay when $N > K$.
\end{example}
\begin{example}[Heat Equation]
    The Heat Equation is an example of a PDE. Let \(u = u(x,t)\), \(\frac{\partial u}{\partial t} = \frac{\partial^2 u}{\partial x^2}\). This PDE denotes the temperature of a body at time t and at position x.
\end{example}

\subsection{Classification}
\marginpar{January 08, 2026.}
\subsubsection{The Order}
\begin{definition}
    The order of an ODE is the order of the highest derivative that appears in the equation.
\end{definition}
\begin{example}
    $N' = rN$ is first order. \quad $y'' + 2y' = e^t$ is second order.
\end{example}

\subsubsection{Scalar vs.\ System}
\begin{definition}[Scalar ODE]
    A scalar ODE has one unknown function.
\end{definition}
\begin{definition}[System of ODEs]
    A system of ODEs has several unknown functions.
\end{definition}
\begin{example}[Scalar ODE vs System of ODEs]

    Scalar: $y'' + y = 0$.\\
    System: $\begin{cases}y_1' = y_2\\y_2' = -y_1\end{cases}$
\end{example}
\begin{definition}[First Order Sytem]
    A first order system is an equation of the form
    \[y'(t) = f(y(t),\, t),\]
    where the unknown $y(t) = (y_1(t),\dots,y_n(t)) \in \R^n$ is a vector-valued
    function and $f : \R^n \times \R \to \R^n$. Written out by
    component, this is just $n$ coupled equations:
    \[y_i'(t) = f_i(y_1,\dots,y_n,\, t), \qquad i = 1,\dots,n.\]
\end{definition}
\begin{proposition}[Reduction of Higher-Order ODEs]
    Any single $n$th-order ODE
    \[y^{(n)} = G\!\left(t,\, y,\, y',\dots, y^{(n-1)}\right)\]
    can be recast as a first order system by introducing new variables for each derivative:
    \[y_1 = y, \quad y_2 = y', \quad \dots \quad y_n = y^{(n-1)}.\]
    Differentiating each substitution gives the equivalent system
    \[\begin{cases}y_1' = y_2, \\ y_2' = y_3, \\ \;\vdots \\ y_{n-1}' = y_n, \\ y_n' = G(t,\, y_1,\dots, y_n).\end{cases}\]
    The first $n-1$ equations just say each variable is the derivative of the previous
    one; the last equation encodes the original ODE.
\end{proposition}
\begin{example}[System of First Order ODEs]
    Consider the second-order ODE
    \[y'' + 2y' + y = e^t.\]
    Define new variables
    \[y_1 = y, \qquad y_2 = y'.\]
    Then the system becomes
    \[\begin{pmatrix}y_1' \\ y_2'\end{pmatrix} = \begin{pmatrix}0 & 1 \\ -1 & -2\end{pmatrix} \begin{pmatrix}y_1 \\ y_2\end{pmatrix} + \begin{pmatrix} 0 \\ e^t \end{pmatrix}.\]
    Let
    \[A = \begin{pmatrix} 0 & 1 \\ -1 & -2 \end{pmatrix},\quad r(t) = \begin{pmatrix} 0 \\ e^t \end{pmatrix}.\]
\end{example}
\begin{problem}[System of First Order ODEs]
    Rewrite this third order ODE as a first order system:
    \[y''' + 4y' - y = 0\]
\end{problem}

\subsubsection{Linearity}
\begin{definition}[Linearity]
    An ODE is linear if the unknown function and its derivatives appear only to the first power, are not multiplied together, and are not composed with nonlinear functions. They may be multiplied by functions of the independent variable.
\end{definition}
\begin{example}[Linearity]
    Consider the following ODEs:
    \begin{itemize}
        \item Linear: $y' +3y = 0$
        \item Linear: $y'' - 2xy' + y = cos(x)$
        \item Non-linear: $y' + y^2 = 0$
        \item Non-linear: $y'' + sin(y) = 0$
    \end{itemize}
\end{example}

\subsubsection{Autonomy}
\begin{definition}[Autonomy]
    The \(n\)\textsuperscript{th} order ODE \(F(t, y, y', \ldots, y^{(n)}) = 0\) is autonomous if \(F\) does not depend explicitly on t, that is, it can be written as \(F(y, y', \ldots, y^{(n)}) = 0\). Otherwise, it is non-autonomous.
\end{definition}
\begin{example}[Autonomy]
    Consider the following ODEs:
    \begin{itemize}
        \item Non-autonomous: \(y'' + 2y' + y - e^t = 0\)
        \item Autonomous: \(N'(t) = rN(t)\)
        \item Non-autonomous: \(y'(t) = ty(t)\)
    \end{itemize}
\end{example}

\subsection{Solutions and Initial Value Problems}
\begin{definition}[Solution]
    A solution of $y'(t) = f(y(t), t)$ on an interval $J$ is a 
    differentiable function $y : J \to D$ such that
    \[y'(t) = f(y(t), t) \quad \forall\, t \in J.\]
\end{definition}
\begin{example}[Explicit Solution]
    Consider the following ODE $y' + y = 1$. We can verify that \(y(t) = e^{-t} + 1\), and therefore \(y'(t) = -e^{-t}\), is a solution on \(\R\). Indeed,
    \[y' + y = -e^{-t} + (e^{-t} + 1) = 1.\]
\end{example}
\begin{example}[Implicit Solution]
    Consider the ODE $y \frac{dy}{dx} = x$. This is a nonautonomous, nonlinear, first-order scalar ODE. Separating $y\,dy = x\,dx$ and integrating gives the implicit solution $x^2 - y^2 = C, \quad C \in \R$.
\end{example}
\marginpar{January 13, 2026}
\begin{definition}[Initial Value Problem (IVP)]
    An initial value problem consists of an ODE together with a condition $y(t_0)=y_0$, where the value of the solution is prescribed at an initial time \(t_0\).
\end{definition}

\subsection{Existence and Uniqueness Theorem}
\subsubsection{Lipschitz continuity}
\begin{definition}[Lipschitz Continuity]
    A function $f : D \to \R^n$ is Lipschitz continuous on $D \subseteq \R^n$ if there exists $L \geq 0$ such that
    \[||f(y_1) - f(y_2)|| \le L||y_1 - y_2|| \quad \forall y_1,y_2 \in D.\]
    The smallest such $L$ is called the Lipschitz constant, denoted $\mathrm{Lip}(f)$.
\end{definition}
\begin{remark}[How to find a Lipschitz constant]
    If $f$ is differentiable on $D \subset \R$ and 
    $|f'(y)| \leq M$ for all $y \in D$, then $f$ is Lipschitz with 
    $\mathrm{Lip}(f) = M$.
\end{remark}
\begin{example}
    Let \(f(y) = 4y - 5\), \(D = \R\), and \(||\cdot|| = |\cdot|\).
    Then
    \[|f(y_1) - f(y_2)| = |4y_1 - 4y_2| = 4|y_1 - y_2|.\]
    So \(f\) is Lipschitz on \(\R\) with \(Lip(f) = 4\).
\end{example}
\begin{example}
    Let
    \[f(y) = \frac{1}{y - 1}, \quad f'(y) = -\frac{1}{(y - 1)^2}, \quad D = (1, +\infty).\]
    This function is not Lipschitz continuous on \(D\) since as \(y \to 1^+\), the derivative \(f'(y)\) approaches \(-\infty\), and so the ratio \(\frac{|f(y_2) - f(y_1)|}{|y_2 - y_1|}\) approaches \(\infty\) for any \(y_1,y_2 \in D\). Therefore, there is no Lipschitz constant for \(f\) on \(D\).\\
    To make this function Lipschitz continuous, we fix \(\delta > 1\) and define \(D_\delta = (\delta, +\infty)\). For \(y_1,y_2 \in D_\delta\), by the Mean Value Theorem, there exists \(z \in (y_1,y_2)\) such that
    \[f(y_2) - f(y_1) = f'(z)(y_2 - y_1).\]
    So
    \[|f(y_2) - f(y_1)| \le \frac{1}{(z-1)^2}|y_2 - y_1|\le \frac{1}{(\delta-1)^2}|y_2 - y_1|.\]
    Thus \(f\) is Lipschitz on \(D_\delta\) with
    \[Lip(f) = \frac{1}{(\delta - 1)^2}.\]
\end{example}

\subsubsection{Local Lipschitz continuity}
\begin{definition}[Locally Lipschitz]
    Let $D \subseteq \R^n$ be open. A function $f : D \to \R^n$ is \emph{locally Lipschitz} if for every compact set $K \subset D$ there exists $L > 0$ such that
    \[\|f(y_1) - f(y_2)\| \leq L\|y_1 - y_2\| \quad \forall\, y_1, y_2 \in K.\]
\end{definition}
\begin{remark}
    Lipschitz $\Rightarrow$ locally Lipschitz, but not conversely.  
    Any $C^1$ function is locally Lipschitz (by the Mean Value 
    Inequality from Section~\ref{MVI}).
\end{remark}
\begin{example}[Locally Lipschitz but not Lipschitz]
    \[f(y) = y^2 \in \R.\]
    Since \(f'(y)=2y\) is unbounded on \(\R\), no single constant works on the whole domain, so \(f\) is not Lipschitz.  
    However, on any bounded set \(K=[-M,M]\), \(|f'(y)| \le 2M\), so \(f\) is locally Lipschitz.
\end{example}
\begin{example}[Lipschitz $\implies$ locally Lipschitz]
    \[f(y) = \sin y.\]
    Since \(|f'(y)| = |\cos y| \leq 1\), $\forall y \in \R$, $f$ is Lipschitz on \(\R\), with \(Lip(f) = 1\).
\end{example}
\begin{example}[Continuous but not locally Lipschitz]
    \[f(y) = \sqrt{|y|}.\]
    The derivative $f'(y) = \frac{\sgn(y)}{2\sqrt{|y|}} = \begin{cases} \dfrac{1}{2\sqrt{y}},\,y > 0 \\ -\dfrac{1}{2\sqrt{-y}},\, y < 0 \end{cases}$ is unbounded near \(y = 0\), so no Lipschitz constant exists even locally. Hence \(f\) is not locally Lipschitz.
\end{example}

\subsubsection{Integral Form of the IVP}
\marginpar{January 13, 2026}
\begin{lemma}
    A function \(y\) solves the IVP if and only if
    \[y(t) = y_0 + \int_{t_0}^{t} f(y(s),s)\,ds.\]
\end{lemma}
\begin{proof}
    If \(y' = f(y,t)\) and \(y(t_0) = y_0\), then by the Fundamental Theorem of Calculus,
    \[y(t) = y_0 + \int_{t_0}^{t} f(y(s),s)ds.\]
    Conversely, differentiating the right-hand side gives
    \[y'(t) = f(y(t),t), \quad y(t_0) = y_0.\]
\end{proof}

\subsubsection{Picard operator}
The integral form of the IVP motivates defining the following operator.
\begin{algorithm}[Finding an Interval of Existence]
    Since $(y_0, t_0) \in D \times (a,b)$ and the set is open, choose $\alpha, \delta > 0$ such that
    \[D_{\alpha,\delta} = \{(y,t) : \|y-y_0\| \leq \alpha,\; |t-t_0| \leq \delta\} \subset D \times (a,b),\]
    and set
    \[M_{\alpha,\delta} = \sup_{(y,t)\in D_{\alpha,\delta}}\|f(y,t)\| < +\infty, \qquad \varepsilon = \min\!\left(\delta,\frac{\alpha}{M_{\alpha,\delta}}\right), \qquad J = (t_0-\varepsilon, t_0+\varepsilon).\footnote{You will have to memorize this algorithm for the exams when finding an interval of existence.}\]
\end{algorithm}
\begin{definition}[Picard Operator]
    For any $y : J \to \R^n$ with $y(t_0) = y_0$ and 
    $(y(t),t) \in D_{\alpha,\delta}$ for all $t \in J$, define
    \[T(y)(t) = y_0 + \int_{t_0}^{t} f(y(s),s)\, ds.\]
    A function $y$ solves the IVP if and only if $T(y) = y$, i.e.\ 
    $y$ is a \emph{fixed point} of $T$.
\end{definition}
\begin{lemma}[Invariance]
    If \(y(t_0)=y_0\) and \((y(t),t) \in D_{\alpha,\delta}\) \(\forall t \in J\), then \(T(y)(t_0)=y_0\) and \((T(y)(t),t) \in D_{\alpha,\delta}\) for all \(t \in J\).\footnote{Basically, If a function stays inside the box, then applying $T$ keeps it inside the box.}
\end{lemma}
\begin{proof}
    Clearly \(T(y)(t_0)=y_0\). For \(t \in J\),
    \[\|T(y)(t) - y_0\| \leq \int_{t_0}^{t} \|f(y(s),s)\|ds \leq M_{\alpha,\delta}|t - t_0| \leq M_{\alpha,\delta}\epsilon \leq \alpha.\]
    Hence \((T(y)(t),t) \in D_{\alpha,\delta}\).
\end{proof}
\begin{example}[Picard operator]
    Consider the IVP $y' = y$, $y(0) = 1$. It's integral form is $y(t) = 1 + \int_0^t y(s)ds$. Start with the simplest function that satisfies the initial condition: $y_0(t) = 1$. Now iterate:
    \[y_1 = T(y_0) = 1 + \int_0^t 1ds = 1 + t,\]
    \[y_2 = T(y_1) = 1 + \int_0^t (1+s)ds = 1+t+\frac{t^2}{2},\]
    \[y_3 = T(y_2) = 1 + t + \frac{t^2}{2} + \frac{t^3}{6}.\]
    Each step adds to the next power of t. We can express this relation as
    \[y(t) = \sum_{k=0}^\infty \frac{t^k}{k!} = e^t.\]
    Indeed, $T(y)(t) = 1 + \int_0^t e^s ds = e^t$, so $T(y)=y$. Therefore, $T$ is an operator that maps a function to a new one, and only the true solution is a fixed point of $T$.
\end{example}

\subsubsection{Existence and Uniqueness}
\label{EUT}
\begin{theorem}[Existence and Uniqueness]
    Let $D \subseteq \R^n$ be open and $(a,b)$ an open interval containing $t_0$. Suppose $f : D \times (a,b) \to \R^n$\,\footnote{\(D \times (a,b)\) denotes the set of all pairs \((y,t)\) with \(y \in D \subset \R^n\) and \(t \in (a,b)\), i.e., all allowed state-time inputs of \(f\).} is continuous and locally Lipschitz in $y$. If $y_0 \in D$, then the IVP
    \[\begin{cases}y' = f(y,t),\\y(t_0) = y_0\end{cases}\]
    has a unique solution on some open interval $J$ containing $t_0$.
\end{theorem}
\marginpar{January 20, 2026}
To apply the Existence and Uniqueness Theorem, we choose \(\varepsilon\) so that the solution remains inside the domain \(D \times (a,b)\). Consider the IVP
\[\begin{cases}y' = f(y,t), \\ y(t_0) = y_0.\end{cases}\]
The theorem guarantees that there exists a unique solution 
\[y : (t_0-\varepsilon, t_0+\varepsilon) \to \R^n.\]
\begin{algorithm}[Choosing $\varepsilon$]
    Choose \(\alpha > 0\) and \(\delta > 0\) such that
    \[D_{\alpha,\delta} := \{(y,t) : \|y-y_0\| \le \alpha,\ |t-t_0| \le \delta\} \subset D \times (a,b).\]
    Since \(f(y,t)\) is continuous, define
    \[M_{\alpha,\delta} = \sup_{(y,t)\in D_{\alpha,\delta}} \|f(y,t)\| < \infty.\]
    Then one may take
    \[\varepsilon = \min\!\left(\delta, \frac{\alpha}{M_{\alpha,\delta}}\right).\]
\end{algorithm}
\begin{remark}[Maximizing $\varepsilon$]
    To determine the largest value of $\varepsilon$  for which the Theorem of Existence and Uniqueness ensures the existence of a solution, then one may try to maximize $\min\!\left(\delta, \frac{\alpha}{M_{\alpha,\delta}}\right)$ by equating both terms together $\delta = \frac{\alpha}{M_{\alpha,\delta}}$.
\end{remark}
\begin{example}
    Consider the IVP
    \[y' = t^2+y^2, \quad y(0) = 0.\]
    Determine the largest value of $\varepsilon$ for which the Theorem of Existence and Uniqueness ensures the existence of a solution y : ($-\varepsilon, \varepsilon) \to \R$.\\
    First and foremost, define
    \[D_{\alpha,\delta} = \left\{(y,t): |y-0| \leq \alpha,\; |t-0| \leq \delta\right\}\subset D \times (a,b).\]
    Set $M_{\alpha,\delta} = \sup_{(y,t) \in D_{\alpha,\delta}} ||f(y,t)||$ and $\varepsilon = \min\!\left(\delta, \frac{\alpha}{M_{\alpha,\delta}}\right)$. Since $f(y,t) = t^2 + y^2$ is $C^1$ on $\R^2$, it is locally Lipschitz, so we take $D = \R$ and $(a,b) = \R$. Now we need to find $M_{\alpha,\delta}$.
    \[M_{\alpha, \delta} = \sup_{(y,t) \in D_{\alpha,\delta}} |t^2+y^2|,\]
    since $t^2$ is maximized when $t=\delta$ and $y^2$ is maximized when $y=\alpha$, then
    \[M_{\alpha, \delta} = \delta^2 + \alpha^2.\]
    Now we want to maximize $\varepsilon = \min\!\left(\delta, \frac{\alpha}{\delta^2 + \alpha^2}\right)$ over $\alpha, \delta > 0$.\\
    \textbf{Step 1: Optimize over $\alpha$.} For fixed $\delta$, let $g(\alpha) = \frac{\alpha}{\delta^2 + \alpha^2}$. Then
    \[g'(\alpha) = \frac{\delta^2 - \alpha^2}{(\delta^2 + \alpha^2)^2} = 0 \implies \alpha = \delta,\]
    so the best choice is $\alpha = \delta$, giving $\varepsilon = \min\!\left(\delta, \frac{1}{2\delta}\right)$.\\
    \textbf{Step 2: Optimize over $\delta$.} The minimum $\min\!\left(\delta, \frac{1}{2\delta}\right)$ is maximized when both terms are equal, since below $\delta = \frac{1}{\sqrt{2}}$ the first term is the bottleneck, and above it the second term is. Setting $\delta = \frac{1}{2\delta}$ gives $\delta = \frac{1}{\sqrt{2}}$. Hence the largest guaranteed $\varepsilon$ is
    \[\varepsilon = \frac{1}{\sqrt{2}}.\]
\end{example}
\begin{problem}[Existence and Uniqueness Theorem]
    Consider the initial value problem
    \[\begin{cases}y' = t^2+y^2, \\ y(0) = 0,\end{cases}\]
    Determine the largest value of $\epsilon$ for which the Theorem of Existence and Uniqueness ensures the existence of a solution $y: (-\epsilon,\epsilon) \to \R$.
\end{problem}
\begin{problem}[LLC]
    Consider the initial value problem
    \[\begin{cases}y' = 3y^{2/3}, \\ y(0) = 0.\end{cases}\]
    Investigate existence and uniqueness.
\end{problem}

\marginpar{January 15, 2026}
\begin{corollary}[Picard Iteration \& Existence]
    Define $y_0(t) \equiv y_0$ (constant). For $k \geq 1$ set
    \[y_k(t) = T(y_{k-1})(t) = y_0 + \int_{t_0}^{t} f(y_{k-1}(s), s)\, ds.\]
    By the invariance lemma, each $y_k$ stays inside $D_{\alpha,\delta}$. 
\end{corollary}
\begin{remark}
    The sequence $\{y_k\}$ is Cauchy in $\R^n$ (see Appendix~\ref{pf:picard-convergence}) and therefore converges uniformly to a function $y: J \to \R^n$ which satisfies the integral equation, hence solves the IVP.
\end{remark}

\subsubsection{Uniqueness}
\begin{proof}[Proof of Uniqueness]
    Assume $y(t)$ and $z(t)$ are two solutions of the IVP with the same initial condition. By the Lipschitz property of $f$,
    \[\|y(t)-z(t)\| = \|\int_{t_0}^t f(y(s),s) - f(z(s),s) ds\| \leq \int_{t_0}^t L\|y(s) - z(s)\|ds,\]
    for some Lipschitz constant $L>0$. Define 
    \[g(t) = \int_{t_0}^{t} \|y(s)-z(s)\|ds,\]
    so $g'(t) \leq L g(t)$ and $g(t_0)=0$. Grönwall's idea\footnote{Multiplying by $e^{-L(t-t_0)}$ gives $\frac{d}{dt}(e^{-L(t-t_0)}g(t)) \le 0$, which shows $e^{-L(t-t_0)}g(t)$ is decreasing and thus $g(t) = 0$} tells us this forces $g(t) = 0$ for all $t$, so $\|y(t) - z(t)\| = 0$. Hence, $y(t) = z(t)$, meaning two solutions with the same initial condition cannot diverge, this proves the solution is unique.
\end{proof}

\section{First-Order Scalar Equation}
\marginpar{January 22, 2026}
A first-order scalar ODE involves a single unknown function $y(t)$ 
and its first derivative:
\[y' = f(y,t).\]
We study three main classes: linear, separable, and exact equations.
\subsection{First Order Linear Equations}
\begin{definition}[First Order Linear Equation]
    A first-order linear equation has the form
    \[a_0(t)y' + a_1(t)y = g(t), \quad a_0(t) \neq 0.\]
    Dividing by $a_0(t)$ gives the standard form
    \[y' + P(t)y = Q(t).\]
\end{definition}

\subsubsection{Integrating Factor Method}
To solve $y' + P(t)y = Q(t)$, multiply both sides by a function $\mu(t)$ chosen so that the left-hand side becomes an exact derivative:
\[\mu y' + \mu P(t)y = \frac{d}{dt}(\mu y).\]
This requires $\mu'(t) = \mu(t)P(t)$, which gives
\[\mu(t) = e^{\int P(t)\,dt}.\]
The equation then becomes
\[\frac{d}{dt}(\mu y) = \mu Q(t),\]
and integrating both sides yields
\[y(t) = \frac{1}{\mu(t)}\left(\int \mu(t)Q(t)\,dt + C\right), \quad C \in \R.\]
\begin{example}
    Solve $y' - 2y = 3e^t$ with $y(0) = 1$.\\
    Here $P(t) = -2$, $Q(t) = 3e^t$, so $\mu(t) = e^{-2t}$. 
    Multiplying through:
    \[\frac{d}{dt}(e^{-2t}y) = 3e^{-t}.\]
    Integrating:
    \[e^{-2t}y = -3e^{-t} + C \implies y(t) = -3e^t + Ce^{2t}.\]
    Applying $y(0) = 1$ gives $C = 4$, so
    \[y(t) = -3e^t + 4e^{2t}.\]
\end{example}
\begin{remark}[Quick Solution Method]
    A quick method to solve a first-order linear equation (not presented by the professor) is
    \[y(t) = \frac{\int \mu(t) Q(t)}{\mu(t)}.\]
\end{remark}

\subsubsection{Existence and Uniqueness for First Order Linear Equations}
\begin{theorem}[Existence \& Uniqueness]
    If $P$ and $Q$ are continuous on an open interval containing $t_0$, then the IVP
    \[y' + P(t)y = Q(t), \quad y(t_0) = y_0\]
    has a unique solution on that entire interval.
\end{theorem}
\begin{remark}
    Unlike the general E\&U theorem from Chapter~\ref{EUT}, this result gives a global solution on the full interval where $P$ and $Q$ are continuous, not just locally.
\end{remark}
\begin{example}
    For the IVP
    \[\begin{cases}y' + \dfrac{1}{t-1}y = \dfrac{1}{\cos t},\\y(0) = 1,\end{cases}\]
    $P(t)$ is discontinuous at $t = 1$ and $Q(t)$ is discontinuous 
    at $t = \pm\frac{\pi}{2}$. The largest interval containing 
    $t_0 = 0$ on which both are continuous is
    \[J_{\max} = \left(-\frac{\pi}{2},\, 1\right).\]
\end{example}

\subsection{Separable Equations}
\marginpar{January 27, 2026}
\begin{definition}[Separable Equation]
    The ODE $\frac{dy}{dx} = f(x,y)$ is \emph{separable} if $f(x,y) = f_1(x)f_2(y)$ for some functions $f_1$ and $f_2$.
\end{definition}
\begin{example}
    $\dfrac{dy}{dx} = -\dfrac{x}{y}$ is separable with $f_1(x) = x$, $f_2(y) = -\dfrac{1}{y}$.\\
    $\dfrac{dy}{dx} = x^2 + y^2$ is not separable.
\end{example}
One can rewrite the separable ODE as
\begin{equation}
    \label{SeparableODE}
    M(x) + N(y)\frac{dy}{dx} = 0,
\end{equation}
where $M(x) = -f_1(x)$ and $N(y) = \frac{1}{f_2(y)}$, and letting $H_1$, $H_2$ be antiderivatives of $M$, $N$ respectively, the chain rule gives
\[\frac{d}{dx}(H_1(x) + H_2(y(x))) = 0,\]
so the implicit general solution is
\[H_1(x) + H_2(y) = C, \quad C \in \R,\]
or equivalently in integral form,
\[\int \frac{dy}{f_2(y)} = \int f_1(x)\,dx + C.\]
\begin{example}[Logistic Equation]
    The logistic equation
    \[\frac{dN}{dt} = rN\!\left(1 - \frac{N}{K}\right)\]
    is separable. Using partial fractions,
    \[\int\frac{dN}{N} + \int\frac{dN}{N-K} = \int r\,dt \implies \ln\left|\frac{N}{N-K}\right| = rt + C.\]
    Solving with $N(0) = N_0$ gives
    \[N(t) = \frac{KN_0}{N_0 + (K-N_0)e^{-rt}}.\]
    As $t\to\infty$, $N(t)\to K$ (the carrying capacity), provided $N_0 \neq 0$.
\end{example}

\subsection{Exact Equations}
\marginpar{January 29, 2026}
\subsubsection{Potential Functions and Integral Curves}
\begin{definition}[Potential Function]
    A function $\Psi(x,y)$ is a potential function for the equation \eqref{SeparableODE} if
    \[\frac{\partial\Psi}{\partial x} = M \quad \text{and} \quad \frac{\partial\Psi}{\partial y} = N.\]
    The general solution is then given implicitly by the level curves
    \[\Psi(x,y) = C.\]
\end{definition}
\begin{definition}[Integral Curve]
    An integral curve is the level curve of $\Psi$ passing through a specific initial condition $(x_0,y_0)$:
    \[\Gamma_{(x_0,y_0)} = \{(x,y)\mid \Psi(x,y) = \Psi(x_0,y_0)\}.\]
\end{definition}

\begin{remark}
    For a separable equation, the general solution $\Psi(x,y) = C$ describes all possible solution curves, and picking a particular $C$ gives the integral curve passing through a given point.
\end{remark}
\begin{example}
    For $\frac{dy}{dx} = -\frac{x}{y}$, $y \neq 0$, separating gives $y\,dy + x\,dx = 0$, so the potential function is
    \[\Psi(x,y) = \frac{x^2}{2} + \frac{y^2}{2}.\]
    The general solution is the family of circles $x^2+y^2 = C$. For $y(0)=2$, $\Psi(0,2)=2$, so the unique integral curve is
    \[y(x) = \sqrt{4-x^2}, \quad J_{\max} = (-2,2).\]
\end{example}

\subsubsection{Definition and Exactness Criterion}
\begin{definition}[Exact Equation]
    The equation \eqref{SeparableODE} is exact on a domain $D$ if there exists a potential function $\Psi$ such that $\Psi_x = M$ and $\Psi_y = N$. Along any solution $y(x)$,
    \[\frac{d}{dx}\Psi(x,y(x)) = 0,\]
    so $\Psi$ is constant along solutions.
\end{definition}
\begin{theorem}
    Let $D\subset\R^2$ be open and simply connected\footnote{Meaning there are no holes in the domain.}. Suppose $M$, $N$, $M_y$, $N_x$ are continuous on $D$. Then \eqref{SeparableODE} is exact on $D$ if and only if
    \[M_y(x,y) = N_x(x,y) \quad \forall\,(x,y)\in D.\]
\end{theorem}
\begin{proof}
    ($\Rightarrow$) Suppose it is exact, then $M = \Psi_x$ and $N = \Psi_y$, so $M_y = \Psi_{xy} = \Psi_{yx} = N_x$ by equality of mixed partials.

    ($\Leftarrow$) If $M_y = N_x$ on a simply connected domain and 
    partials are continuous, one can construct $\Psi$ with 
    $\Psi_x = M$ and $\Psi_y = N$ explicitly by integration.\footnote{The construction is shown for rectangles in the notes; the general case follows similarly.}
\end{proof}
\begin{example}
    Solve
    \[y\cos x + 2xe^y + (\sin x + x^2e^y - 1)\,\frac{dy}{dx} = 0.\]

    Identify $M = y\cos x + 2xe^y$ and $N = \sin x + x^2e^y - 1$. Check:
    \[M_y = \cos x + 2xe^y = N_x. \quad \checkmark\]
    So the equation is exact. Integrate $\Psi_x = M$ in $x$:
    \[\Psi = y\sin x + x^2e^y + c(y).\]
    Differentiate in $y$ and match with $N$:
    \[\Psi_y = \sin x + x^2e^y + c'(y) = \sin x + x^2e^y - 1 
    \implies c'(y) = -1 \implies c(y) = -y.\]
    The general solution is
    \[\Psi(x,y) = y\sin x + x^2e^y - y = C.\]
\end{example}
\begin{remark}
    Every separable equation is exact on a simply connected domain. Writing $M(x)\,dx + N(y)\,dy = 0$ gives $M_y = 0 = N_x$ automatically, so exactness always holds. The potential function is simply $\Psi(x,y) = H_1(x) + H_2(y)$.
\end{remark}

\subsubsection{Integrating Factors for Non-Exact Equations}
\marginpar{February 3, 2026}
If $M_y \neq N_x$, we seek a function $\mu(x,y)$ such that $\mu M\,dx + \mu N\,dy = 0$ is exact. The condition $(\mu M)_y = (\mu N)_x$ expands to
\[\mu_y M + \mu M_y = \mu_x N + \mu N_x.\]

\textbf{Case: $\mu = \mu(x)$ only.} Then $\mu_y = 0$ and the condition reduces to
\[\frac{d\mu}{dx} = \mu\,\frac{M_y - N_x}{N}.\]
If $\frac{M_y - N_x}{N}$ depends only on $x$, this is a separable ODE for $\mu(x)$ that can be solved directly.
\begin{example}
    Solve
    \[(3xy + y^2)\,dx + (x^2 + xy)\,dy = 0.\]

    Here $M_y = 3x+2y$ and $N_x = 2x+y$, so $M_y \neq N_x$. Compute:
    \[\frac{M_y - N_x}{N} = \frac{x+y}{x(x+y)} = \frac{1}{x},\]
    which depends only on $x$. Solving $\frac{d\mu}{dx} = \frac{\mu}{x}$ gives $\mu(x) = x$. Multiplying through by $x$:
    \[(3x^2y+xy^2)\,dx + (x^3+x^2y)\,dy = 0.\]
    Now $M_y = 3x^2+2xy = N_x$. \checkmark \; Integrating:
    \[\Psi = \int(3x^2y+xy^2)\,dx = x^3y + \frac{x^2y^2}{2} + c(y).\]
    Matching $\Psi_y = N$ gives $c'(y) = 0$, so the general solution is
    \[x^3y + \frac{x^2y^2}{2} = C.\]
\end{example}

\subsection{Applications}
\begin{example}[Falling Object with Air Resistance]
    An object of mass $m$ falls under gravity with damping 
    $\gamma > 0$. Setting $v(0) = 0$, the equation
    \[v' + \frac{\gamma}{m}v = g\]
    has integrating factor $\mu = e^{\frac{\gamma}{m}t}$, giving
    \[v(t) = \frac{gm}{\gamma}\left(1 - e^{-\frac{\gamma}{m}t}
    \right).\]
    As $t\to\infty$, the velocity approaches the terminal value 
    $v_\infty = \frac{gm}{\gamma}$.
\end{example}
\begin{example}[Mixing Problem]
    A tank holds 100 L of brine with $y_0$ g of salt. Pure salt 
    solution enters at 50 g/L at rate $R$ L/s, and the well-mixed 
    solution exits at the same rate. The salt amount $y(t)$ satisfies
    \[y' + \frac{R}{100}y = 50R, \quad y(0) = y_0.\]
    Using integrating factor $\mu = e^{\frac{R}{100}t}$,
    \[y(t) = 5000 + (y_0-5000)e^{-\frac{R}{100}t}.\]
    Regardless of $y_0$, the salt amount converges to 5000 g as 
    $t\to\infty$.
\end{example}


\section{Systems of Linear Equations}
\marginpar{February 3, 2026}

Many problems involve several unknown functions evolving together. 
Instead of a single ODE, we study systems of first-order linear ODEs, 
treating the unknowns as a vector and applying matrix algebra.

We consider systems of the form
\begin{equation}
    \label{eq:ode-system}
    y'(t) = A(t)y(t) + r(t), \quad t \in Q,
\end{equation}
where
\[y(t) = \begin{pmatrix}y_1(t)\\\vdots\\y_n(t)\end{pmatrix} \in 
\R^n, \quad A(t) \in M_n(\R), \quad
r(t) = \begin{pmatrix}r_1(t)\\\vdots\\r_n(t)\end{pmatrix}.\]

%------- 3.1 -------
\subsection{Reduction to First-Order Systems}

Any $n$th-order scalar ODE can be written as a first-order system 
by introducing new variables for each derivative.

\begin{example}
    Convert $y''' - 2y'' + 4y' - 5y = \sin t + t^3$ into a 
    first-order system. Set $y_1 = y$, $y_2 = y'$, $y_3 = y''$:
    \[\begin{pmatrix}y_1'\\y_2'\\y_3'\end{pmatrix} = 
    \begin{pmatrix}0&1&0\\0&0&1\\5&-4&2\end{pmatrix}
    \begin{pmatrix}y_1\\y_2\\y_3\end{pmatrix} + 
    \begin{pmatrix}0\\0\\\sin t + t^3\end{pmatrix},\]
    which is in the form $y' = Ay + r$ with $J_{\max} = \R$.
\end{example}

%------- 3.2 -------
\subsection{Existence, Uniqueness, and Solution Structure}

\begin{theorem}[Existence and Uniqueness]
    If every entry of $A(t)$ and $r(t)$ is continuous on $Q$, then 
    for any $t_0 \in Q$ and $y_0 \in \R^n$, the IVP
    \[\begin{cases}y'(t) = A(t)y(t) + r(t),\\ y(t_0) = y_0\end{cases}\]
    has a unique solution on all of $Q$. Hence $J_{\max} = Q$.
\end{theorem}

\begin{definition}[Homogeneity]
    If $r(t) = 0$, the system is homogeneous; otherwise it 
    is non-homogeneous.
\end{definition}

\begin{definition}[Control System]
    A \emph{linear control system} is a non-homogeneous system of 
    the form
    \[y' = Ay + Bu,\]
    where $A \in M_n(\R)$ governs the dynamics, 
    $u \in \R^m$ is the control input, and 
    $B \in M_{n \times m}(\R)$ determines how the control 
    enters the system. The term $Bu$ can be thought of as a 
    structured forcing term that one can choose to influence the 
    behavior of the solution.
\end{definition}

We now focus on the homogeneous system
\begin{equation}
    \label{eq:ode-system-matrix}
    y'(t) = A(t)y(t).
\end{equation}

\begin{definition}[Principle of Superposition]
    \label{POS}
    If $y_1(t), \dots, y_n(t)$ are solutions of 
    \eqref{eq:ode-system-matrix}, then any linear combination 
    is also a solution.
\end{definition}

\begin{definition}[Linear Independence]
    Solutions $y_1, \dots, y_n$ of \eqref{eq:ode-system-matrix} 
    are linearly independent if
    \[c_1 y_1(t) + \cdots + c_n y_n(t) = 0 \;\; \forall\, t \in Q 
    \implies c_1 = \cdots = c_n = 0.\]
\end{definition}

\begin{theorem}[Solution Space]
    The solution set $S$ of \eqref{eq:ode-system-matrix} is an 
    $n$-dimensional vector subspace of $C^1(Q)$.
\end{theorem}

\begin{proof}
    \textbf{Step 1: Vector space.} If $y_1, y_2 \in S$ and 
    $\alpha_1, \alpha_2 \in \R$,
    \[(\alpha_1 y_1 + \alpha_2 y_2)' = \alpha_1 y_1' + 
    \alpha_2 y_2' = A(t)(\alpha_1 y_1 + \alpha_2 y_2),\]
    so any linear combination is in $S$, and the zero function 
    is in $S$.

    \textbf{Step 2: Basis of dimension $n$.} Let $e_1,\dots,e_n$ 
    be the standard basis of $\R^n$, and let $y_i$ be the 
    unique solution of
    \[\begin{cases}y' = A(t)y,\\ y(t_0) = e_i\end{cases}\]
    for $i = 1,\dots,n$. By uniqueness, every $z \in S$ is a linear 
    combination of the $y_i$, so $\{y_1,\dots,y_n\}$ spans $S$. 
    Evaluating at $t_0$ shows they are independent. Hence 
    $\dim S = n$.
\end{proof}

\subsubsection{Matrix Solutions and the Wronskian}

\begin{definition}[Matrix Solution]
    A matrix solution of \eqref{eq:ode-system-matrix} is an 
    $n \times n$ matrix whose columns are solutions of 
    \eqref{eq:ode-system-matrix}.
\end{definition}

\begin{proposition}
    Let $y(t)$ be a matrix solution of \eqref{eq:ode-system-matrix}. 
    Either $\det(y(t)) \neq 0$ for all $t \in Q$, or 
    $\det(y(t)) = 0$ for all $t \in Q$.
\end{proposition}

\begin{proof}
    Suppose $\det(y(t_0)) = 0$ for some $t_0$. Then there exists a 
    nonzero $v \in \R^n$ such that $y(t_0)v = 0$. Define 
    $y_1(t) = y(t)v$; since each column of $y(t)$ is a solution, 
    $y_1 \in S$. Also define $y_2(t) = 0 \in S$. Since 
    $y_1(t_0) = 0 = y_2(t_0)$, uniqueness gives $y_1(t) = y_2(t)$ 
    for all $t$, so $y(t)v = 0$ for all $t$. Hence the columns of 
    $y(t)$ are linearly dependent and $\det(y(t)) = 0$ for all 
    $t \in Q$.
\end{proof}

\begin{definition}[Wronskian]
    The Wronskian of $y_1,\ldots,y_n$ is
    \[W(t) = W(y_1,\ldots,y_n)(t) = \det(y(t)).\]
\end{definition}

\begin{example}
    Let $y_1(t) = \begin{pmatrix}\cos 2t\\-\sin 2t\end{pmatrix}$, 
    $y_2(t) = \begin{pmatrix}-2\sin 2t\\-2\cos 2t\end{pmatrix}$.
    \[W(t) = \det\begin{pmatrix}\cos 2t & -2\sin 2t\\
    -\sin 2t & -2\cos 2t\end{pmatrix} = -2 \neq 0,\]
    so $y_1$ and $y_2$ are linearly independent.
\end{example}

\begin{definition}[Fundamental Matrix Solution]
    A matrix solution with nonvanishing Wronskian is called a 
    \emph{fundamental matrix solution}.
\end{definition}

\begin{theorem}[Liouville's Formula]
    Let $y(t)$ be a matrix solution of \eqref{eq:ode-system-matrix} 
    with $y(t_0) = y_0$. Then
    \[\det(y(t)) = \det(y_0)\,e^{\int_{t_0}^t \operatorname{tr}
    (A(s))\,ds},\]
    where $\operatorname{tr}(M)$ denotes the sum of the diagonal 
    entries of $M$.
\end{theorem}

\subsubsection{Recovering the Coefficient Matrix and Solving IVPs}

\begin{proposition}[Recovering $A$]
    \label{prop:recover-A}
    Let $\Phi(t)$ be a fundamental matrix solution of 
    \eqref{eq:ode-system-matrix}. Then $\Phi(t)$ is invertible for 
    all $t \in Q$ and
    \[A(t) = \Phi'(t)\,\Phi(t)^{-1}.\]
\end{proposition}

\begin{proof}
    Since $\Phi'(t) = A(t)\Phi(t)$ and $\det\Phi(t) \neq 0$, 
    multiplying on the right by $\Phi(t)^{-1}$ gives the result.
\end{proof}

\begin{example}
    Given $\Phi(t) = \begin{pmatrix}e^{2t}&0\\0&e^{-t}\end{pmatrix}$,
    \[\Phi'(t) = \begin{pmatrix}2e^{2t}&0\\0&-e^{-t}\end{pmatrix}, 
    \quad \Phi(t)^{-1} = \begin{pmatrix}e^{-2t}&0\\0&e^t
    \end{pmatrix},\]
    so $A = \Phi'(t)\Phi(t)^{-1} = \begin{pmatrix}2&0\\0&-1
    \end{pmatrix}$.
\end{example}

Since $y_1,\ldots,y_n$ are linearly independent, the general 
solution is
\[y(t) = c_1 y_1(t) + \cdots + c_n y_n(t).\]
Given $y(t_0) = y_0$, the constants $c = (c_1,\ldots,c_n)^T$ 
are uniquely determined by
\[\begin{pmatrix}|&&|\\ y_1(t_0)&\cdots&y_n(t_0)\\|&&|
\end{pmatrix}c = y_0 \implies c = \Phi(t_0)^{-1} y_0.\]

%------- 3.3 -------
\subsection{Homogeneous Systems with Constant Coefficients}
\label{sec:const-coeff}

Consider \eqref{eq:ode-system-matrix} with $A \in M_n(\R)$ 
constant. We look for solutions of the form
\begin{equation}
    \label{eq:exp-ansatz}
    y(t) = e^{\lambda t}u,
\end{equation}
where $u$ is a constant vector. Substituting into 
\eqref{eq:ode-system-matrix},
\[e^{\lambda t}(\lambda u) = e^{\lambda t}(Au) \implies Au = 
\lambda u.\]
So $\lambda$ must be an eigenvalue of $A$ and $u$ a corresponding 
eigenvector.

\begin{definition}[Kernel]
    $\ker A = \{u \mid Au = 0\}$ is the kernel (null space) of $A$.
\end{definition}

\begin{definition}[Characteristic Polynomial, Eigenvalue, Spectrum]
    The \emph{characteristic polynomial} of $A \in M_n(\R)$ 
    is $p(\lambda) = \det(A - \lambda I)$. An \emph{eigenvalue} is a 
    root of $p(\lambda)$, and the \emph{spectrum} is
    \[\sigma(A) = \{\lambda \mid p(\lambda) = 0\}.\]
\end{definition}

\subsubsection{Diagonalizable Case: Real Eigenvalues}

\begin{definition}[Diagonalizable Matrix]
    $A \in M_n(\R)$ is \emph{diagonalizable} if there exists 
    an invertible $P$ such that $P^{-1}AP$ is diagonal.
\end{definition}

\begin{theorem}[Diagonalization]
    $A$ is diagonalizable if and only if it has a basis of 
    eigenvectors. In particular, if all eigenvalues of $A$ are real 
    and distinct, then $A$ is diagonalizable.
\end{theorem}

\begin{proof}
    Let $\lambda_1,\ldots,\lambda_n$ be real distinct eigenvalues 
    with eigenvectors $u_1,\ldots,u_n$. Distinct eigenvalues imply 
    the eigenvectors are linearly independent, so they form a basis 
    of $\R^n$. Let $P = [u_1 \cdots u_n]$. Then
    \[AP = PD, \quad D = \operatorname{diag}(\lambda_1,\ldots,
    \lambda_n),\]
    so $P^{-1}AP = D$.
\end{proof}

\begin{theorem}[General Solution via Diagonalization]
    If $A$ is diagonalizable with eigenpairs 
    $(\lambda_i, u_i)_{i=1}^n$, the general solution of 
    \eqref{eq:ode-system-matrix} is
    \[y(t) = \sum_{i=1}^n c_i e^{\lambda_i t} u_i.\]
\end{theorem}

\begin{example}[$2\times 2$ system]
    Let $y' = Ay$ with $A = \begin{pmatrix}1&3\\3&1\end{pmatrix}$.

    \textbf{Eigenvalues:} $p(\lambda) = (1-\lambda)^2 - 9 = 0$ gives 
    $\lambda_1 = -2$, $\lambda_2 = 4$, so $\sigma(A) = \{-2, 4\}$.

    \textbf{Eigenvectors:} For $\lambda_1 = -2$: $(A+2I)u = 0$ gives 
    $u_1 = \begin{pmatrix}1\\-1\end{pmatrix}$. For $\lambda_2 = 4$: 
    $u_2 = \begin{pmatrix}1\\1\end{pmatrix}$.

    \textbf{Wronskian:} $W(0) = \begin{vmatrix}1&1\\-1&1
    \end{vmatrix} = 2 \neq 0$, so the solutions are independent.

    \textbf{General solution:}
    \[y(t) = c_1 e^{-2t}\begin{pmatrix}1\\-1\end{pmatrix} + 
    c_2 e^{4t}\begin{pmatrix}1\\1\end{pmatrix}.\]

    \textbf{With} $y(0) = \begin{pmatrix}1\\2\end{pmatrix}$:
    \[\begin{pmatrix}c_1\\c_2\end{pmatrix} = 
    \begin{pmatrix}1&1\\-1&1\end{pmatrix}^{-1}
    \begin{pmatrix}1\\2\end{pmatrix} = 
    \begin{pmatrix}1/2 & -1/2\\1/2&1/2\end{pmatrix}
    \begin{pmatrix}1\\2\end{pmatrix} = 
    \begin{pmatrix}-1/2\\3/2\end{pmatrix}.\]
\end{example}

\begin{example}[$3\times 3$ system]
    \marginpar{February 10, 2026}
    Let $y' = Ay$ with 
    $A = \begin{pmatrix}3&1&-2\\-1&5&-2\\-3&3&-2\end{pmatrix}$.

    \textbf{Eigenvalues:} $p(\lambda) = -(\lambda+2)(\lambda-4)^2$, 
    so $\lambda_1 = -2$ and $\lambda_2 = \lambda_3 = 4$.

    \textbf{Eigenvectors:}
    For $\lambda_1 = -2$: $(A+2I)u = 0$ gives 
    $u_1 = \begin{pmatrix}1\\1\\3\end{pmatrix}$.
    For $\lambda = 4$: row reduction of $(A-4I)u = 0$ yields two 
    free variables ($b$ and $c$ free, $a = b-2c$), giving
    \[u_2 = \begin{pmatrix}1\\1\\0\end{pmatrix}, \quad 
    u_3 = \begin{pmatrix}-2\\0\\1\end{pmatrix}.\]
    Since $\dim\ker(A-4I) = 2 =$ algebraic multiplicity, $A$ is 
    diagonalizable.

    \textbf{Wronskian:} $W(0) = \det\begin{pmatrix}1&1&-2\\1&1&0\\
    3&0&1\end{pmatrix} = 6 \neq 0$.

    \textbf{General solution:}
    \[y(t) = c_1 e^{-2t}\begin{pmatrix}1\\1\\3\end{pmatrix} + 
    c_2 e^{4t}\begin{pmatrix}1\\1\\0\end{pmatrix} + 
    c_3 e^{4t}\begin{pmatrix}-2\\0\\1\end{pmatrix}.\]
\end{example}

\begin{example}[Non-diagonalizable]
    Let $A = \begin{pmatrix}1&1\\0&1\end{pmatrix}$. Since $A$ is 
    upper triangular, $\lambda_1 = \lambda_2 = 1$ (multiplicity 2). 
    Solving $(A-I)u = 0$ gives only one independent eigenvector 
    $u = \begin{pmatrix}1\\0\end{pmatrix}$, so $A$ is 
    \emph{not} diagonalizable.
\end{example}

\subsubsection{Diagonalizable Case: Complex Eigenvalues}

If $A \in M_n(\R)$ has a complex eigenpair 
$(\lambda, u)$ with $\lambda = \alpha + i\beta$ and 
$u = p + iq$, then a complex solution is
\begin{equation}
    \label{eq:exp-ansatz-complex}
    y(t) = e^{(\alpha+i\beta)t}(p+iq).
\end{equation}

\begin{definition}[Euler's Formula]
    \begin{equation}
        \label{eq:EulerFormula}
        e^{ix} = \cos x + i\sin x, \quad x \in \R.
    \end{equation}
\end{definition}

Expanding \eqref{eq:exp-ansatz-complex} using Euler's formula and 
separating real and imaginary parts yields two real solutions:
\begin{equation}
    \label{eq:exp-ansatz-complex-sol}
    y_1(t) = e^{\alpha t}(p\cos\beta t - q\sin\beta t), \qquad 
    y_2(t) = e^{\alpha t}(p\sin\beta t + q\cos\beta t).
\end{equation}

\begin{proposition}[Complex Conjugate Eigenpairs]
    \label{prop:CCE}
    Let $A \in M_n(\R)$. If $(\lambda, u)$ is an eigenpair 
    with $\lambda \in \mathbb{C}$, then $(\bar\lambda, \bar u)$ is 
    also an eigenpair.
\end{proposition}

\begin{proof}
    From $Au = \lambda u$, taking conjugates and using 
    $\bar A = A$ gives $A\bar u = \bar\lambda\,\bar u$.
\end{proof}

\begin{proposition}
    \marginpar{February 12, 2026}
    If $\lambda = \alpha + i\beta \in \sigma(A)$ with $\beta \neq 0$ 
    and eigenvector $u = p + iq$, then the functions 
    \eqref{eq:exp-ansatz-complex-sol} are real-valued linearly 
    independent solutions of \eqref{eq:ode-system-matrix}.
\end{proposition}

\begin{example}
    Let $A = \begin{pmatrix}2&-1\\1&2\end{pmatrix}$.

    \textbf{Eigenvalues:} $p(\lambda) = (\lambda-2)^2 + 1 = 0$ 
    gives $\lambda_1 = 2+i$, $\lambda_2 = 2-i$.

    \textbf{Eigenvector for $\lambda_1 = 2+i$:} 
    $(A-(2+i)I)u = 0$ gives $b = ia$, so taking $a=1$:
    \[u_1 = \begin{pmatrix}1\\i\end{pmatrix} = 
    \underbrace{\begin{pmatrix}1\\0\end{pmatrix}}_{p} + 
    i\underbrace{\begin{pmatrix}0\\1\end{pmatrix}}_{q}.\]
    By Proposition~\ref{prop:CCE}, $u_2 = \bar u_1 = 
    \begin{pmatrix}1\\-i\end{pmatrix}$.
\end{example}

\subsubsection{Nondiagonalizable Matrices and Generalized Eigenvectors}

\begin{remark}
    If $A$ does not admit a basis of eigenvectors, there are not 
    enough solutions of the form $e^{\lambda t}u$ to span the 
    solution space.
\end{remark}

\begin{definition}[Algebraic and Geometric Multiplicity]
    Let $\lambda_0 \in \sigma(A)$.
    \begin{enumerate}
        \item The \emph{algebraic multiplicity} $m$ is the largest 
        integer such that $(\lambda - \lambda_0)^m$ divides 
        $p(\lambda)$.
        \item The \emph{geometric multiplicity} is 
        $\dim\ker(A - \lambda_0 I)$ (the number of independent 
        eigenvectors for $\lambda_0$).
    \end{enumerate}
    The \emph{defect} is $d_{\lambda_0} = m - \dim\ker(A-\lambda_0 I) 
    \geq 0$. We call $\lambda_0$ \emph{defective} if 
    $d_{\lambda_0} > 0$.
\end{definition}

\begin{example}
    Let $A = \begin{pmatrix}1&1\\0&1\end{pmatrix}$, 
    $p(\lambda) = (\lambda-1)^2$. Algebraic multiplicity $m=2$; 
    $\ker(A-I) = \operatorname{span}\!\left\{
    \begin{pmatrix}1\\0\end{pmatrix}\right\}$ has dimension 1. 
    Defect $= 1 > 0$, so $\lambda_0 = 1$ is defective.
\end{example}

\begin{example}
    Let $A = \begin{pmatrix}0&1&0&1\\0&-2&-1&-1\\1&1&0&0\\
    0&1&0&1\end{pmatrix}$, $p(\lambda) = \lambda^3(\lambda+1)$. 
    For $\lambda_0 = 0$: algebraic multiplicity $m=3$, 
    $\ker A = \operatorname{span}\!\left\{
    \begin{pmatrix}1\\-1\\1\\1\end{pmatrix}\right\}$ has dimension 1. 
    Defect $= 2$.
\end{example}

\begin{example}
    For $A = \begin{pmatrix}3&1&-2\\-1&5&-2\\-3&3&-2\end{pmatrix}$, 
    $p(\lambda) = (\lambda+2)(\lambda-4)^2$. For $\lambda_0 = 4$: 
    $m = 2$, $\dim\ker(A-4I) = 2$. Defect $= 0$, so 
    $\lambda_0 = 4$ is \emph{not} defective.
\end{example}

\begin{definition}[Generalized Eigenvector]
    Let $\lambda \in \sigma(A)$ with algebraic multiplicity $m$. 
    A \emph{generalized eigenvector} satisfies 
    $(A - \lambda I)^m u = 0$.
\end{definition}

\begin{theorem}[Generalized Eigenvector Basis]
    If $\lambda \in \sigma(A)$ has algebraic multiplicity $m$, then
    $\dim\ker(A-\lambda I)^m = m$. Consequently, every 
    $A \in M_n(\R)$ has a basis of generalized eigenvectors.
\end{theorem}

Denote $k_j = \ker(A-\lambda I)^j$. Then
\[k_0 = \{0\} \subsetneq k_1 \subsetneq k_2 \subsetneq 
\cdots \subsetneq k_m.\]

\begin{lemma}
    $k_j \subseteq k_{j+1}$ for all $j$. Moreover, if 
    $k_j = k_{j+1}$ for some $j$, then 
    $k_j = k_{j+1} = \cdots = k_m$.
\end{lemma}

\begin{proof}
    The inclusion is immediate. For the second part: assume 
    $k_j = k_{j+1}$ and let $w \in k_{j+2}$. Setting 
    $u = (A-\lambda I)w \in k_{j+1} = k_j$ gives 
    $(A-\lambda I)^{j+1}u = (A-\lambda I)^{j+2}w = 0$, hence 
    $(A-\lambda I)^j u = 0$, so $(A-\lambda I)^{j+1}w = 0$, meaning 
    $w \in k_{j+1}$. Iterate to get the result.
\end{proof}

\begin{algorithm}[Computing Generalized Eigenvectors]
    Given a basis of $k_i$ with $\dim(k_i) < m$, find 
    $u \in k_{i+1} \setminus k_i$ by solving
    $(A - \lambda I)u = w$ for each basis vector $w \in k_i$.
\end{algorithm}

\begin{example}
    For $A = \begin{pmatrix}1&1\\0&1\end{pmatrix}$: 
    $k_1 = \operatorname{span}\!\left\{
    \begin{pmatrix}1\\0\end{pmatrix}\right\}$. To find 
    $u \in k_2 \setminus k_1$, solve
    \[\begin{pmatrix}0&1\\0&0\end{pmatrix}
    \begin{pmatrix}a\\b\end{pmatrix} = 
    \begin{pmatrix}1\\0\end{pmatrix}.\]
    This gives $b=1$, $a$ free. Pick $a=0$: 
    $u_2 = \begin{pmatrix}0\\1\end{pmatrix}$.
    Check: $(A-I)^2 u_2 = (A-I)\begin{pmatrix}1\\0\end{pmatrix} = 0$. 
    \checkmark
\end{example}

\subsubsection{Practical Solution Guide for Repeated Eigenvalues}
Let $\lambda$ be an eigenvalue with algebraic multiplicity $2$.

\begin{itemize}
    \item If $\dim\ker(A - \lambda I) = 2$ (two independent eigenvectors), then
    \[y(t) = c_1 e^{\lambda t} v_1 + c_2 e^{\lambda t} v_2.\]

    \item If $\dim\ker(A - \lambda I) = 1$ (only one eigenvector $v$), then the system is defective and we must introduce a generalized eigenvector $w$ satisfying
    \[(A - \lambda I)w = v.\]

    The general solution is
    \[y(t) = e^{\lambda t}\left( c_1 v + c_2 (t v + w) \right).\]
\end{itemize}
\subsubsection*{higher multiplicity}
If an eigenvalue $\lambda$ has multiplicity $m > 2$ and the system is fully defective (only one eigenvector $v$), then we build a chain of generalized eigenvectors:
\[(A - \lambda I)w_1 = v, \quad (A - \lambda I)w_2 = w_1, \quad \dots\]
The solution follows the pattern:
\[y(t) = e^{\lambda t} \left[c_1 v+ c_2 (t v + w_1)+ c_3 \left(\frac{t^2}{2} v + t w_1 + w_2\right)+ \cdots\right].\]
\begin{remark}
    If you have an eigenvalue with multiplicity $m$ $\implies$ highest power is $t^{m-1}$.
\end{remark}

\subsection{The Matrix Exponential}
\marginpar{February 17, 2026}

\begin{definition}[Matrix Exponential]
    For $A \in M_n(\R)$,
    \[e^A = \sum_{k=0}^{\infty} \frac{A^k}{k!} = I + A + 
    \frac{A^2}{2!} + \cdots\]
\end{definition}

\begin{definition}[Induced Matrix Norm]
    \[\|A\| = \sup_{x \neq 0}\frac{\|Ax\|}{\|x\|} = 
    \sup_{\|x\|=1}\|Ax\|.\]
\end{definition}

\begin{proposition}[Norm Properties]
    For all $A, B \in M_n(\R)$ and $x \in \R^n$:
    \[\|Ax\| \leq \|A\|\cdot\|x\|, \qquad 
    \|AB\| \leq \|A\|\cdot\|B\|.\]
\end{proposition}

\begin{proof}
    For any nonzero $x$: $\|Ax\| = \|A(x/\|x\|)\|\cdot\|x\| 
    \leq \|A\|\cdot\|x\|$. For submultiplicativity, let $\|x\|=1$:
    $\|ABx\| \leq \|A\|\cdot\|Bx\| \leq \|A\|\cdot\|B\|$. Take the 
    supremum over unit vectors.
\end{proof}

\begin{proposition}[Convergence]
    The series defining $e^A$ converges absolutely for every 
    $A \in M_n(\R)$.
\end{proposition}

\begin{proof}
    By the triangle inequality, 
    $\|\sum_{k=0}^N A^k/k!\| \leq \sum_{k=0}^N \|A\|^k/k!$, 
    which converges to $e^{\|A\|}$.
\end{proof}

\begin{proposition}[Basic Properties]
    For all $A \in M_n(\R)$ and $t, s \in \R$:
    \[e^{A\cdot 0} = I, \quad e^{A(t+s)} = e^{At}e^{As}, \quad 
    e^{-At} = (e^{At})^{-1}, \quad \frac{d}{dt}e^{At} = Ae^{At} 
    = e^{At}A.\]
    If $AB = BA$, then $e^{A+B} = e^A e^B$.
\end{proposition}

\begin{proof}[Proof of the derivative formula]
    \[\frac{d}{dt}e^{At} = \frac{d}{dt}\sum_{k=0}^\infty 
    \frac{A^k t^k}{k!} = \sum_{k=1}^\infty \frac{A^k t^{k-1}}
    {(k-1)!} = A\sum_{k=0}^\infty \frac{A^k t^k}{k!} = Ae^{At}.\]
\end{proof}

\begin{theorem}[Fundamental Matrix Solution]
    $e^{At}$ is a fundamental matrix solution of 
    \eqref{eq:ode-system-matrix}. The general solution is 
    $y(t) = e^{At}C$, and the unique solution to the IVP 
    $y(0) = y_0$ is $y(t) = e^{At}y_0$.
\end{theorem}

\begin{proof}
    $\frac{d}{dt}e^{At} = Ae^{At}$ shows each column satisfies 
    the system. $e^{A\cdot 0} = I$ implies $\det(e^{A\cdot 0}) = 
    1 \neq 0$, so $e^{At}$ is invertible.
\end{proof}

\subsubsection{Computing $e^{At}$ from a Fundamental Matrix}
\marginpar{February 19, 2026}

Let $Y(t)$ be any fundamental matrix solution of $y' = Ay$. Since 
$e^{At}$ is also a fundamental matrix, there exists invertible 
$C$ such that $e^{At} = Y(t)C$. Evaluating at $t=0$:
\[I = Y(0)C \implies C = Y(0)^{-1}, \quad \therefore \quad 
e^{At} = Y(t)\,Y(0)^{-1}.\]

\begin{example}
    For $A = \begin{pmatrix}1&3\\3&1\end{pmatrix}$ with 
    $\lambda_1=-2$, $u_1=\begin{pmatrix}-1\\1\end{pmatrix}$ and 
    $\lambda_2=4$, $u_2=\begin{pmatrix}1\\1\end{pmatrix}$:
    \[Y(t) = \begin{pmatrix}-e^{-2t}&e^{4t}\\e^{-2t}&e^{4t}
    \end{pmatrix}, \quad Y(0)^{-1} = \frac{1}{2}
    \begin{pmatrix}-1&1\\1&1\end{pmatrix},\]
    \[e^{At} = Y(t)Y(0)^{-1} = \begin{pmatrix}
    \frac{1}{2}(e^{-2t}+e^{4t}) & \frac{1}{2}(-e^{-2t}+e^{4t})\\
    \frac{1}{2}(-e^{-2t}+e^{4t}) & \frac{1}{2}(e^{-2t}+e^{4t})
    \end{pmatrix}.\]
    Verify: $e^{A\cdot 0} = I$. \checkmark
\end{example}

\subsubsection{Action of $e^{At}$ on Eigenvectors and Generalized 
Eigenvectors}

\textbf{Eigenvectors.} If $Au = \lambda u$, then
\[e^{At}u = \sum_{k=0}^\infty \frac{A^k t^k}{k!}u = 
\sum_{k=0}^\infty \frac{\lambda^k t^k}{k!}u = e^{\lambda t}u.\]

\textbf{Generalized eigenvectors.} If $(A-\lambda I)^{j+1}u=0$, 
write $A = (A-\lambda I) + \lambda I$. Since these terms commute,
\[e^{At}u = e^{\lambda t}e^{(A-\lambda I)t}u = 
e^{\lambda t}\sum_{k=0}^{j}\frac{(A-\lambda I)^k t^k}{k!}u,\]
a finite sum since $(A-\lambda I)^k u = 0$ for $k > j$.

\begin{proposition}
    The general solution of \eqref{eq:ode-system-matrix} is
    \[y(t) = \sum_{i=1}^n c_i\, e^{At}u_i, \quad c_i \in \R,\]
    where $u_1,\ldots,u_n$ are $n$ linearly independent generalized 
    eigenvectors of $A$.
\end{proposition}

\begin{example}
    For $A = \begin{pmatrix}1&1\\0&1\end{pmatrix}$, $\sigma(A) = 
    \{1\}$. From the generalized eigenvector computation above, 
    $u_1 = \begin{pmatrix}1\\0\end{pmatrix}$ (eigenvector) and 
    $u_2 = \begin{pmatrix}0\\1\end{pmatrix}$ (generalized). Then
    \[y_1(t) = e^t u_1, \qquad y_2(t) = e^{At}u_2 = 
    e^t(u_2 + tu_1) = e^t\begin{pmatrix}t\\1\end{pmatrix},\]
    giving the general solution
    \[y(t) = c_1 e^t\begin{pmatrix}1\\0\end{pmatrix} + 
    c_2 e^t\begin{pmatrix}t\\1\end{pmatrix}.\]
\end{example}

\begin{example}
    For $A = \begin{pmatrix}2&4\\-1&-2\end{pmatrix}$: 
    $p(\lambda) = \lambda^2$, so $\sigma(A) = \{0\}$ with 
    multiplicity 2. 
    $k_1 = \ker A = \left\langle\begin{pmatrix}-2\\1
    \end{pmatrix}\right\rangle$.
    Solve $Au_2 = u_1$ for $u_2 \in k_2\setminus k_1$: choosing 
    $b=0$ gives $u_2 = \begin{pmatrix}-1\\0\end{pmatrix}$. Then
    \[y_1(t) = \begin{pmatrix}-2\\1\end{pmatrix}, \qquad 
    y_2(t) = u_2 + tu_1 = 
    \begin{pmatrix}-1-2t\\t\end{pmatrix},\]
    \[y(t) = c_1\begin{pmatrix}-2\\1\end{pmatrix} + 
    c_2\begin{pmatrix}-1-2t\\t\end{pmatrix}, \quad W(t) = 
    \det\begin{pmatrix}-2&-1-2t\\1&t\end{pmatrix} = 1 \neq 0.\]
\end{example}

\subsection{Non-Homogeneous Systems}

Consider
\begin{equation}
    \label{eq:non-homogeneous-ode-system-matrix}
    y'(t) = Ay(t) + r(t),
\end{equation}
with $r: I \to \R^n$ continuous. If $y_1(t)$ and $y_2(t)$ are two solutions, then 
$y_2 - y_1$ solves the homogeneous system, so there exists 
$C \in \R^n$ with $y_2(t) - y_1(t) = Y(t)C$.

\begin{theorem}[General Solution]
    Let $y_p(t)$ be any particular solution of 
    \eqref{eq:non-homogeneous-ode-system-matrix}. Then the general 
    solution is
    \[y(t) = Y(t)C + y_p(t), \quad C \in \R^n.\]
\end{theorem}

\subsubsection{Variation of Parameters}

Look for a particular solution $y_p(t) = Y(t)C_p(t)$, where 
$C_p(t)$ is unknown. Differentiating and substituting into 
\eqref{eq:non-homogeneous-ode-system-matrix}:
\[AY(t)C_p + Y(t)C_p' = AY(t)C_p + r(t) \implies 
Y(t)C_p'(t) = r(t).\]
Since $Y(t)$ is invertible, $C_p'(t) = Y^{-1}(t)r(t)$, giving
\[y_p(t) = Y(t)\int Y^{-1}(t)\,r(t)\,dt.\]
The general solution is therefore
\[y(t) = Y(t)\!\left[C + \int Y^{-1}(t)\,r(t)\,dt\right], 
\quad C \in \R^n.\]

\subsubsection{Mass-Spring Systems}
\marginpar{February 24, 2026}

\begin{center}
    \begin{figure}[H]
        \centering
        \includegraphics[width=0.5\textwidth]{../Images/MSS-setup.jpeg}
    \end{figure}
\end{center}

At equilibrium, $mg + kL = 0$.

\begin{definition}[Hooke's Law]
    The restoring spring force is $F_s = -kx$, where $k > 0$ is 
    the spring constant.
\end{definition}

Let $u(t)$ be the displacement from equilibrium ($u > 0$ 
downward), with damping $F_d = -\gamma u'$ and external force 
$F(t)$. Newton's second law and the equilibrium relation give
\begin{equation}
    \label{eq:mass-spring-system}
    mu'' + \gamma u' + ku = F(t).
\end{equation}
As a first-order system with 
$y = \begin{pmatrix}u\\u'\end{pmatrix}$:
\[y' = \begin{pmatrix}0&1\\-k/m&-\gamma/m\end{pmatrix}y + 
\begin{pmatrix}0\\F(t)/m\end{pmatrix}.\]
The characteristic polynomial of the homogeneous part is
\[p(\lambda) = \lambda^2 + \frac{\gamma}{m}\lambda + \frac{k}{m}, 
\qquad \lambda_\pm = \frac{-\gamma \pm \sqrt{\gamma^2-4km}}{2m}.\]
Setting $\Delta = \gamma^2 - 4km$:
\begin{itemize}
    \item \textbf{Overdamped} ($\Delta > 0$): two distinct real 
    negative eigenvalues,
    \[u(t) = c_1 e^{\lambda_1 t} + c_2 e^{\lambda_2 t} 
    \to 0 \text{ without oscillation.}\]
    \item \textbf{Critically damped} ($\Delta = 0$): repeated 
    eigenvalue $\lambda = -\gamma/(2m)$,
    \[u(t) = (c_1 + c_2 t)e^{\lambda t} \to 0 \text{ as fast 
    as possible without oscillating.}\]
    \item \textbf{Underdamped} ($\Delta < 0$): complex eigenvalues 
    $\lambda = -\frac{\gamma}{2m} \pm i\omega$ with 
    $\omega = \frac{\sqrt{4km-\gamma^2}}{2m}$,
    \[u(t) = e^{-\frac{\gamma}{2m}t}(c_1\cos\omega t + 
    c_2\sin\omega t), \text{ exponentially decaying oscillations.}\]
\end{itemize}

\subsubsection{RLC Circuits}

\begin{center}

        \includegraphics[width=0.5\textwidth]{../Images/circuit-ode.jpeg}
    \end{figure}

Let $Q(t)$ be the capacitor charge and $I(t) = Q'(t)$ the current.

\begin{definition}[Kirchhoff's Voltage Law]
    The sum of voltage drops around a closed circuit equals the 
    applied voltage $E(t)$.
\end{definition}

The voltage drops are: resistor $RI$, capacitor $Q/C$, inductor 
$LI'$. Applying Kirchhoff's law and substituting $I = Q'$:
\begin{equation}
    \label{eq:RLC-circuit}
    LQ'' + RQ' + \frac{1}{C}Q = E(t).
\end{equation}
This is mathematically identical to \eqref{eq:mass-spring-system} 
under the correspondence
\[m \leftrightarrow L, \quad \gamma \leftrightarrow R, \quad 
k \leftrightarrow \frac{1}{C}.\]

\section{Stability, Phase Portraits and Orbits}
\marginpar{March 10, 2026}

Let $y'(t) = f(y(t))$, $f$ locally Lipschitz, and $y_0 \in \R^n$. Denote by
\[
\Gamma_{y_0} = \{y(t) \mid y'(t) = f(y(t)),\ y(0) = y_0,\ t \in J_{\max}\},
\]
the \emph{orbit} of $y_0$, where $J_{\max}$ is the maximal interval of existence. The orbit is a curve in $\R^n$.

%------- 5.1 -------
\subsection{Orbits and Phase Portraits}

\begin{definition}[Phase Portrait]
    The phase portrait is the collection of all orbits of the system $y' = f(y)$.
\end{definition}

\begin{example}
    Let $y' = y(1-y)$. Since this is a separable ODE, the solution is
    \[y(t) = \frac{y_0}{y_0 + (1-y_0)e^{-t}}.\]
    The orbits are:
    \begin{itemize}
        \item $\Gamma_0 = \{0\}$,
        \item $\Gamma_1 = \{1\}$,
        \item $\Gamma_{y_0} = (0,1)$ for $y_0 \in (0,1)$,
        \item $\Gamma_{y_0} = (1,+\infty)$ for $y_0 > 1$,
        \item $\Gamma_{y_0} = (-\infty,0)$ for $y_0 < 0$.
    \end{itemize}
    The values $0$ and $1$ are the equilibrium solutions.
\end{example}

%------- 5.2 -------
\subsection{Equilibrium Solutions}

\begin{definition}[Equilibrium Solution]
    A vector $\tilde{y} \in \R^n$ is called an \emph{equilibrium solution} (or steady state) of $y' = f(y)$ if $f(\tilde{y}) = 0$. Since $\tilde{y}' = 0 = f(\tilde{y})$, the solution satisfying $y(0) = \tilde{y}$ is constant: $\tilde{y}(t) = \tilde{y}$ for all $t \in \R$, and its orbit is the single point $\Gamma_{\tilde{y}} = \{\tilde{y}\}$.
\end{definition}

\begin{example}
    For the linear system $y' = Ay$, equilibrium solutions satisfy $Au = 0$, i.e.\ they lie in $\ker(A)$.
    \begin{itemize}
        \item If $A$ is invertible, then $\ker(A) = \{0\}$, so $\tilde{y} = 0$ is the only equilibrium.
        \item If $A$ is not invertible, $\ker(A)$ contains more equilibria.
    \end{itemize}
\end{example}

\begin{example}
    Let $A = \begin{pmatrix}1&-1\\2&-2\end{pmatrix}$. Solving $Au = 0$ gives $a = b$, so
    \[\ker(A) = \left\langle\begin{pmatrix}1\\1\end{pmatrix}\right\rangle = \left\{c\begin{pmatrix}1\\1\end{pmatrix} \;\middle|\; c\in\R\right\}.\]
    Every point on this line is an equilibrium solution.
\end{example}

\begin{example}[Nonlinear Pendulum]
    From Newton's second law,
    \[\theta'' + \gamma\theta' + \frac{g}{l}\sin(\theta) = 0.\]
    Setting $y_1 = \theta$, $y_2 = \theta'$, the system becomes
    \[\begin{pmatrix}y_1'\\y_2'\end{pmatrix} = \begin{pmatrix}y_2\\-\tfrac{g}{l}\sin(y_1) - \gamma y_2\end{pmatrix} =: f(y).\]
    Equilibria satisfy $f(y) = 0$: we need $y_2 = 0$ and $\sin(y_1) = 0$, giving
    \[y_1 = n\pi,\quad n \in \mathbb{Z}.\]
\end{example}

%------- 5.3 -------
\subsection{Stability}

\subsubsection{Definitions}

\begin{definition}[Stable Equilibrium]
    An equilibrium $\tilde{y}$ of $y' = f(y)$ is \emph{stable} if for every $\epsilon > 0$ there exists $\delta = \delta(\epsilon) > 0$ such that every solution with $\|y(0) - \tilde{y}\| \leq \delta$ satisfies
    \[\|y(t) - \tilde{y}\| \leq \epsilon \quad \forall\, t \geq 0.\]
    Intuitively: solutions that start close to $\tilde{y}$ remain close for all future time.
\end{definition}

\begin{definition}[Unstable Equilibrium]
    An equilibrium $\tilde{y}$ is \emph{unstable} if it is not stable: there exists $\epsilon > 0$ such that for every $\delta > 0$, there is a solution $y(t)$ with $\|y(0) - \tilde{y}\| \leq \delta$ and a time $T > 0$ for which $\|y(T) - \tilde{y}\| > \epsilon$.
\end{definition}

\begin{definition}[Asymptotic Stability]
    An equilibrium $\tilde{y}$ is \emph{asymptotically stable} if it is stable and there exists $\alpha > 0$ such that
    \[\|y(0) - \tilde{y}\| \leq \alpha \implies \lim_{t \to +\infty} \|y(t) - \tilde{y}\| = 0.\]
\end{definition}

\subsubsection{Examples}
\marginpar{March 12, 2026}

\begin{example}[Stable equilibrium]
    Let $y' = -2y$. The unique equilibrium is $\tilde{y} = 0$, and the solution is $y(t) = e^{-2t}y(0)$. For $t \geq 0$,
    \[|y(t) - 0| = e^{-2t}|y(0)| \leq |y(0)|.\]
    Given $\epsilon > 0$, choose $\delta = \epsilon$. Then $|y(0)| \leq \delta$ implies $|y(t)| \leq \epsilon$ for all $t \geq 0$, so $\tilde{y} = 0$ is stable.
\end{example}

\begin{example}[Logistic equation]
    Let $y' = y(1-y)$. The equilibria are $\tilde{y} = 0$ and $\tilde{y} = 1$.

    \textbf{$\tilde{y} = 0$ is unstable.} Take $\epsilon = \tfrac{1}{2}$ and any $0 < \delta < 1$. Choose $y(0) = \delta$; the solution is
    \[y(t) = \frac{\delta}{\delta + (1-\delta)e^{-t}}.\]
    There exists $T = T(\delta)$ such that $y(T) \geq \tfrac{1}{2}$, so $|y(T) - 0| \geq \epsilon$. Hence $\tilde{y} = 0$ is unstable.

    \textbf{$\tilde{y} = 1$ is asymptotically stable.} For any $y_0 > 0$,
    \[|y(t) - 1| = \left|\frac{(y_0-1)e^{-t}}{y_0 + (1-y_0)e^{-t}}\right| = \frac{|y_0-1|}{y_0 e^t + (1-y_0)} \leq |y_0 - 1|,\]
    since $y_0(e^t - 1) + 1 \geq 1$ for $t \geq 0$. Choosing $\delta = \epsilon$ gives stability. Moreover, as $t \to +\infty$ the numerator $|y_0 - 1|e^{-t} \to 0$, so $y(t) \to 1$ and $\tilde{y} = 1$ is asymptotically stable.
\end{example}

\begin{example}[Linear system]
    Let $y' = Ay$ with
    \[A = \begin{pmatrix}-\tfrac{3}{2} & \tfrac{1}{2} \\ \tfrac{1}{2} & -\tfrac{3}{2}\end{pmatrix}.\]
    Since $\det(A) \neq 0$, the only equilibrium is $\tilde{y} = 0$. 
    
    A fundamental matrix solution is
    \[Y(t) = \begin{pmatrix}-e^{-2t} & e^{-t} \\ e^{-2t} & e^{-t}\end{pmatrix},\]
    giving
    \[e^{At} = Y(t)Y(0)^{-1} = \frac{1}{2}\begin{pmatrix}e^{-2t}+e^{-t} & -e^{-2t}+e^{-t} \\ -e^{-2t}+e^{-t} & e^{-2t}+e^{-t}\end{pmatrix}.\]
    Using the induced sup-norm $\|x\|_\infty = \max\{|x_1|,|x_2|\}$ and the corresponding matrix norm
    \[\|M\|_\infty = \max_{1 \leq i \leq n}\sum_{j=1}^{n}|M_{ij}|,\]
    one checks that $\|e^{At}\|_\infty \leq 2e^{-t}$. Therefore
    \[\|y(t)\|_\infty = \|e^{At}y(0)\|_\infty \leq 2e^{-t}\|y(0)\|_\infty.\]
    \textbf{Stability:} given $\epsilon > 0$, choose $\delta = \tfrac{\epsilon}{2}$. Then $\|y(0)\|_\infty \leq \delta$ implies $\|y(t)\|_\infty \leq 2e^{-t}\delta \leq \epsilon$ for all $t \geq 0$.

    \textbf{Asymptotic stability:} for any $\alpha > 0$, if $\|y(0)\|_\infty \leq \alpha$ then $\|y(t)\|_\infty \leq 2\alpha e^{-t} \to 0$ as $t \to +\infty$. Hence $\tilde{y} = 0$ is asymptotically stable.
\end{example}

\subsection{Phase Portraits of Planar Linear Systems}
\marginpar{March 10, 2026}

Let \eqref{eq:ode-system-matrix} hold with $A \in M_2(\R)$ 
and $\det(A) \neq 0$. We classify the phase portrait by the 
eigenstructure of $A$.

\subsubsection{Real Eigenvalues: Diagonalizable Case}

The general solution is
\[y(t) = c_1 e^{\lambda_1 t}u_1 + c_2 e^{\lambda_2 t}u_2.\]

\subsubsection{Complex Eigenvalues}
\marginpar{March 17, 2026}

Suppose $A$ has complex eigenvalues $\lambda = \alpha + i\beta$ 
with $\beta \neq 0$. From Chapter~3, the general solution is
\[y(t) = c_1 e^{\alpha t}(\cos(\beta t)\,p - \sin(\beta t)\,q) 
+ c_2 e^{\alpha t}(\sin(\beta t)\,p + \cos(\beta t)\,q),\]
where $p = \mathrm{Re}(u)$ and $q = \mathrm{Im}(u)$. 
Given constants $c_1, c_2$, there exist $v_1, v_2 \in \R^2$ 
perpendicular such that
\[y(t) = e^{\alpha t}\cos(\beta t)\,v_1 + 
e^{\alpha t}\sin(\beta t)\,v_2.\]
The nature of the equilibrium $\tilde{y} = 0$ depends on $\alpha$:
\begin{itemize}
    \item $\alpha = 0$: $y(t) = \cos(\beta t)v_1 + \sin(\beta t)v_2$ 
    traces a closed ellipse. The origin is called a 
    \emph{stable center}.
    \item $\alpha > 0$: orbits spiral outward. The origin is called 
    an \emph{unstable spiral point}.
    \item $\alpha < 0$: orbits spiral inward. The origin is called 
    an \emph{asymptotically stable spiral point}.
\end{itemize}

\begin{example}
    Let $y' = \begin{pmatrix}-1&6\\-3/2&-1\end{pmatrix}y$ with 
    $\sigma(A) = \{-1 \pm 3i\}$. Since $\alpha = -1 < 0$, the 
    origin is an asymptotically stable spiral point. Note that
    \[A\begin{pmatrix}1\\0\end{pmatrix} = 
    \begin{pmatrix}-1\\-3/2\end{pmatrix}.\]
\end{example}

\subsubsection{Repeated Eigenvalue: Non-Diagonalizable Case}

If $\det(A) \neq 0$, $\sigma(A) = \{\lambda, \lambda\}$, and 
$\ker(A - \lambda I) = \langle u \rangle$ (only one independent 
eigenvector), let $v$ be the generalized eigenvector satisfying 
$(A - \lambda I)v = u$. Then the general solution is
\[y(t) = C_1 e^{\lambda t}u + C_2 e^{\lambda t}(v + tu).\]

\begin{example}[Love Affairs]
    Let $R(t)$ denote the love ($>0$) or hate ($<0$) that Romeo 
    feels toward Juliet, and $J(t)$ the same for Juliet toward Romeo.
    In this story, Romeo echoes Juliet's feelings (the more she 
    loves/hates him, the more he loves/hates her), while Juliet 
    reacts negatively to Romeo's interest. This gives
    \[\frac{dR}{dt} = \alpha J, \quad \alpha > 0, \qquad 
    \frac{dJ}{dt} = -\beta R, \quad \beta > 0,\]
    or in matrix form,
    \[\begin{pmatrix}R'\\J'\end{pmatrix} = 
    \begin{pmatrix}0&\alpha\\-\beta&0\end{pmatrix}
    \begin{pmatrix}R\\J\end{pmatrix}.\]
    The characteristic polynomial is $p(\lambda) = \lambda^2 + 
    \alpha\beta = 0$, giving $\lambda = \pm i\sqrt{\alpha\beta}$. 
    Since $\alpha = 0$, the origin is a \emph{stable center}: 
    their relationship oscillates forever without settling.

    \textbf{Extended model.} Now suppose both Romeo and Juliet 
    echo each other's feelings equally and are equally cautious. 
    With $a < 0 < b$:
    \[A = \begin{pmatrix}a&b\\b&a\end{pmatrix}, \qquad 
    p(\lambda) = (a-\lambda)^2 - b^2 = 0 \implies 
    \lambda = a \pm b.\]
    The eigenpairs are
    \[\lambda_1 = a - b < 0, \quad u_1 = \begin{pmatrix}1\\-1
    \end{pmatrix}; \qquad \lambda_2 = a + b, \quad 
    u_2 = \begin{pmatrix}1\\1\end{pmatrix}.\]
    The general solution is
    \[y(t) = c_1 e^{(a-b)t}\begin{pmatrix}1\\-1\end{pmatrix} + 
    c_2 e^{(a+b)t}\begin{pmatrix}1\\1\end{pmatrix}.\]
    If $\lambda_2 = a + b < 0$, i.e.\ $|a| > |b|$, both 
    eigenvalues are negative and the origin is asymptotically 
    stable: both feelings decay to zero. This corresponds to 
    excessive caution ($|a|$ large) overpowering the willingness 
    to open up ($|b|$ small). If $\lambda_2 > 0$, i.e.\ $|b| > |a|$,
    the relationship grows without bound along the $u_2$ direction.
\end{example}

\subsubsection{Quick Classification Guide for Phase Portraits}

For a $2 \times 2$ linear system $y' = Ay$ with $\det(A) \neq 0$, the phase portrait is determined entirely by the eigenvalues $\lambda_1, \lambda_2$ of $A$. The following guide provides a rapid way to match systems to phase portraits by examining the eigenvalue structure.

\paragraph{Step 1: Calculate the eigenvalues.}
Find $\lambda_1, \lambda_2$ from the characteristic polynomial $\det(A - \lambda I) = 0$.

\paragraph{Step 2: Classify by type and sign.}
\begin{itemize}
    \item Both real, same sign, both negative: Stable node. All orbits approach the origin. Arrows point inward toward the equilibrium.
    
    \item Both real, same sign, both positive: Unstable node. All orbits move away from the origin. Arrows point outward.
    
    \item Real, opposite signs: Saddle point. Hyperbolic structure: orbits approach along one eigenvector direction (stable manifold) and recede along the other (unstable manifold). The phase portrait resembles a stretched X.
    
    \item Complex with negative real part: Stable (asymptotically stable) spiral. Orbits spiral inward toward the origin. Arrows rotate and shrink.
    
    \item Complex with positive real part: Unstable spiral. Orbits spiral outward away from the origin. Arrows rotate and expand.
    
    \item Purely imaginary (real part = 0): Center. Orbits trace closed ellipses. Arrows rotate at constant radius; no convergence or divergence.
\end{itemize}
\subsubsection*{Remark (Degenerate case: $\det(A) = 0$)}
If $\det(A) = 0$, then at least one eigenvalue is $0$, and the above classification no longer applies.

\begin{itemize}
    \item $\lambda_1 = \lambda_2 = 0$, $A = 0$: Every point is an equilibrium. Type: completely degenerate (all equilibria).. Stability: stable (but not asymptotically stable).

    \item $\lambda_1 = \lambda_2 = 0$, $A \neq 0$: Degenerate equilibrium (defective case). Trajectories are straight lines (typically parallel). Type: degenerate node (non-isolated equilibrium behavior). Stability: unstable.

    \item One eigenvalue $0$, the other $\lambda \neq 0$: The origin is non-isolated; there is a line of equilibria.
    \begin{itemize}
        \item If $\lambda < 0$: Type: line of equilibria (degenerate node-like). Stability: stable (but not asymptotically stable).

        \item If $\lambda > 0$: Type: line of equilibria. Stability: unstable.
    \end{itemize}
\end{itemize}
\begin{example}
    If $\sigma(A) = \{-2, 3\}$ (real, opposite signs), look for a phase portrait with a hyperbolic saddle structure. If $\sigma(A) = \{-1 + 2i, -1 - 2i\}$ (complex with $\mathrm{Re}(\lambda) = -1 < 0$), look for a stable spiral spiraling inward.
\end{example}

\subsection{Stability Theory for Linear Systems}

Let \eqref{eq:ode-system-matrix} hold with $A \in M_n(\R)$ 
and $\det(A) \neq 0$, so that $\tilde{y} = 0 \in \R^n$ is 
the only equilibrium solution.

\begin{theorem}[Stability via Eigenvalues]
    Let $A \in M_n(\R)$ be invertible.
    \begin{enumerate}
        \item If $\mathrm{Re}(\lambda) < 0$ for all 
        $\lambda \in \sigma(A)$, then $0$ is an asymptotically 
        stable equilibrium solution of \eqref{eq:ode-system-matrix}.
        \item If there exists $\lambda_0 \in \sigma(A)$ with 
        $\mathrm{Re}(\lambda_0) > 0$, then $0$ is unstable.
    \end{enumerate}
\end{theorem}

\begin{proof}
    \textbf{(1)} Under this hypothesis, there exist constants 
    $C > 0$ and $\gamma > 0$ such that $\|e^{At}\| \leq 
    Ce^{-\gamma t}$. The unique solution of 
    \eqref{eq:ode-system-matrix} with $y(0) = y_0$ is 
    $y(t) = e^{At}y_0$, so
    \[\|y(t)\| = \|e^{At}y_0\| \leq \|e^{At}\|\cdot\|y_0\| 
    \leq Ce^{-\gamma t}\|y_0\|.\]
    \textbf{Stability:} given $\epsilon > 0$, choose 
    $\delta = \epsilon/C$. Then $\|y_0\| < \delta$ implies 
    $\|y(t)\| \leq Ce^{-\gamma t}\cdot\frac{\epsilon}{C} \leq 
    \epsilon$ for all $t \geq 0$.\\
    \textbf{Asymptotic stability:} for any $\alpha > 0$, if 
    $\|y_0\| \leq \alpha$ then 
    $\|y(t)\| \leq C\alpha e^{-\gamma t} \to 0$ as $t \to +\infty$.

    \marginpar{March 19, 2026}
    \textbf{(2)} Let $u \pm iv$ be the eigenvector associated to 
    $\lambda_0 = \alpha_0 + i\beta_0$ with $\alpha_0 > 0$. Then
    \[y(t) = Ce^{\alpha_0 t}(\cos(\beta_0 t)\,u - 
    \sin(\beta_0 t)\,v)\]
    is a solution. Choose $C = \frac{\delta}{\|u\|}$ for any small 
    $\delta > 0$, so that
    \[\|y(0)\| = \left\|\frac{\delta}{\|u\|}u\right\| = \delta.\]
    Let $\epsilon = 1$. Since $\alpha_0 > 0$, there exists 
    $T = T(\delta)$ such that $\|y(T)\| \geq \epsilon = 1$, 
    so $0$ is unstable.
\end{proof}

\subsection{Stability Theory for Nonlinear Systems}
\marginpar{March 19, 2026}

Consider a nonlinear system $y' = f(y)$ with 
$f : D \to \R^n$, $D \subseteq \R^n$ open, and 
let $\tilde{y} \in D$ be an equilibrium solution, i.e.\ 
$f(\tilde{y}) = 0$. For $y$ close to $\tilde{y}$, expand $f$ 
around $\tilde{y}$:
\[f(y) = f(\tilde{y}) + Df(\tilde{y})(y - \tilde{y}) + 
O(\|y-\tilde{y}\|^2) \approx Df(\tilde{y})(y-\tilde{y}).\]
Setting $z = y - \tilde{y}$ (translating $\tilde{y}$ to the 
origin), we get
\[z' = y' = f(y) \approx Df(\tilde{y})z.\]
The solutions of $y' = f(y)$ near $\tilde{y}$ are therefore 
approximated by the solutions of the \emph{linearization}
\[z' = Df(\tilde{y})z\]
near $0$.

\begin{theorem}[Stability via Linearization]
    Let $f : D \to \R^n$ be $C^1$ and let 
    $\tilde{y} \in D$ be an equilibrium solution.
    \begin{enumerate}
        \item If $\mathrm{Re}(\lambda) < 0$ for all 
        $\lambda \in \sigma(Df(\tilde{y}))$, then $\tilde{y}$ is 
        asymptotically stable.
        \item If there exists $\lambda_0 \in \sigma(Df(\tilde{y}))$ 
        with $\mathrm{Re}(\lambda_0) > 0$, then $\tilde{y}$ is 
        unstable.
    \end{enumerate}
\end{theorem}

\begin{definition}[Hyperbolic Equilibrium]
    An equilibrium solution $\tilde{y}$ is called 
    \emph{hyperbolic} if $\mathrm{Re}(\lambda) \neq 0$ for all 
    $\lambda \in \sigma(Df(\tilde{y}))$.
\end{definition}

\begin{theorem}[Hartman--Grobman]
    Near a hyperbolic equilibrium $\tilde{y}$, the solutions of 
    $y' = f(y)$ have the same qualitative structure as the 
    solutions of the linearization $z' = Df(\tilde{y})z$ near $0$.
\end{theorem}

\begin{example}
    Let
    \[f(y) = \begin{pmatrix}-2y_1 - y_1^3 - y_1 y_2^2\\
    -3y_2 + y_1^2 y_2 + y_2^3\end{pmatrix}.\]
    Since $f\begin{pmatrix}0\\0\end{pmatrix} = 
    \begin{pmatrix}0\\0\end{pmatrix}$, the origin is an equilibrium. 
    The Jacobian is
    \[Df(y) = \begin{pmatrix}-2 - 3y_1^2 - y_2^2 & -2y_1 y_2\\
    2y_1 y_2 & -3 + y_1^2 + 3y_2^2\end{pmatrix},\]
    so
    \[Df(0,0) = \begin{pmatrix}-2 & 0\\ 0 & -3\end{pmatrix}, 
    \quad \sigma(Df(0,0)) = \{-2,-3\}.\]
    Since $\mathrm{Re}(\lambda) < 0$ for all 
    $\lambda \in \sigma(Df(0,0))$, the equilibrium 
    $\tilde{y} = \begin{pmatrix}0\\0\end{pmatrix}$ is hyperbolic 
    and asymptotically stable.
\end{example}

\section{Laplace Transform}

\subsection{Definition and Existence}

\begin{definition}[Laplace Transform]
    Given $f: [0,\infty) \to \R$, its Laplace transform is
    \[\Lap[f](s) = F(s) = \int_0^\infty e^{-st} f(t)\,dt = \lim_{A\to+\infty} \int_0^A e^{-st} f(t)\,dt,\]
    provided the improper integral converges.
\end{definition}

\begin{definition}[Piecewise Continuous]
    A function $f: [0,\infty)\to\R$ is piecewise continuous if there exists a partition
    $0 = t_0 < t_1 < \cdots < t_n < t_{n+1} = +\infty$
    such that $f$ is continuous on each $(t_i,\,t_{i+1})$ and the one-sided limits
    $\lim_{t\to t_i^-} f(t)$ and $\lim_{t\to t_i^+} f(t)$ are finite for each $i$.
\end{definition}

\begin{definition}[Exponential Order]
    A function $f:[0,\infty)\to\R$ is of exponential order $a$ if there exist constants $K\geq 0$ and $M\geq 0$ such that
    \[|f(t)| \leq K e^{at} \quad \text{for all } t \geq M.\]
\end{definition}

\begin{remark}
    We abbreviate ``piecewise continuous of exponential order'' as PCEO.
\end{remark}

\begin{example}[Exponential order]
    \begin{itemize}[nosep]
        \item $f(t) = t^n$ is of exponential order $a$ for any $a > 0$, since $t^n$ grows slower than any exponential.
        \item $f(t) = e^{t^2}$ is not of exponential order for any $a$.
    \end{itemize}
\end{example}

\begin{lemma}
    If $\int_0^\infty |g(t)|\,dt$ converges, then $\int_0^\infty g(t)\,dt$ converges.
\end{lemma}

\begin{theorem}[Existence of the Laplace Transform]
    If $f$ is PCEO with order $a$, then $F(s) = \Lap[f](s)$ exists for all $s > a$.
\end{theorem}

\begin{proof}
    \[\int_0^\infty |e^{-st} f(t)|\,dt \leq \int_0^M e^{-st}|f(t)|\,dt + K\int_M^\infty e^{-(s-a)t}\,dt.\]
    The first term is finite since $f$ is piecewise continuous on $[0,M]$. The second term equals
    \[K\cdot\frac{e^{-(s-a)M}}{s-a} < \infty \quad\text{for } s > a. \qedhere\]
\end{proof}

%-----------------------------------------------------------------------------
\subsection{Computing Laplace Transforms}
%-----------------------------------------------------------------------------

\begin{example}[$f(t) = K$]
    Since $|K| \leq K\,e^{0\cdot t}$, the exponential order is $a = 0$. For $s > 0$:
    \[\Lap[K](s) = K\lim_{A\to\infty}\int_0^A e^{-st}\,dt = K\lim_{A\to\infty}\left(\frac{1}{s} - \frac{e^{-sA}}{s}\right) = \frac{K}{s}.\]
\end{example}

\begin{example}[$f(t) = e^{at}$, $a\in\R$]
    With $K = 1$, $M = 0$:
    \[\Lap[e^{at}](s) = \int_0^\infty e^{-(s-a)t}\,dt = \frac{1}{s-a},\quad s > a.\]
\end{example}

\begin{remark}
    By induction: $\Lap[t^n](s) = \dfrac{n!}{s^{n+1}}$ for $n = 0,1,2,\ldots$
\end{remark}

\begin{example}[$f(t) = \cosh(at)$]
    Using $\cosh(at) = \frac{1}{2}(e^{at} + e^{-at})$ and linearity:
    \[\Lap[\cosh(at)](s) = \frac{1}{2}\cdot\frac{1}{s-a} + \frac{1}{2}\cdot\frac{1}{s+a} = \frac{s}{s^2 - a^2},\quad s > |a|.\]
\end{example}

\begin{example}[$f(t) = \cos(\omega t)$]
    Using $\cos(\omega t) = \frac{1}{2}(e^{i\omega t} + e^{-i\omega t})$ and linearity:
    \[\Lap[\cos(\omega t)](s) = \frac{1}{2}\cdot\frac{1}{s - i\omega} + \frac{1}{2}\cdot\frac{1}{s + i\omega} = \frac{s}{s^2 + \omega^2},\quad s > 0.\]
\end{example}

\begin{example}[Piecewise function]
    Let $f(t) = \begin{cases}1,& 0\leq t < 1,\\ K,& t = 1,\\ 0,& t > 1.\end{cases}$
    Since $f(t) = 0$ for $t > 1$:
    \[\Lap[f](s) = \int_0^1 e^{-st}\,dt = \begin{cases}1, & s = 0,\\[4pt] \dfrac{1 - e^{-s}}{s}, & s \neq 0,\end{cases}\]
    which exists for all $s\in\R$.
\end{example}

\subsection{Properties}
%-----------------------------------------------------------------------------

\subsubsection{Linearity}

Let $f_1, f_2$ be PCEO with orders $a_1, a_2$ and set $a = \max(a_1, a_2)$. Then for any constants $c_1, c_2$:
\[\Lap[c_1 f_1 + c_2 f_2](s) = c_1\,\Lap[f_1](s) + c_2\,\Lap[f_2](s),\quad s > a.\]

\subsubsection{Differentiability and the $s$-Derivative Formula}

\begin{proposition}
    Let $f$ be PCEO with order $a$. Then $F:(a,\infty)\to\R$ is $C^\infty$ and
    \[\frac{d^k F}{ds^k}(s) = (-1)^k\,\Lap[t^k f(t)](s),\quad s > a.\]
\end{proposition}

\begin{proof}
    For $k = 1$: since $f$ is PCEO of order $a$, so is $tf(t)$, so $\Lap[tf(t)](s)$ exists for $s > a$. Differentiating under the integral sign:
    \[F'(s) = \frac{d}{ds}\int_0^\infty e^{-st}f(t)\,dt = -\int_0^\infty te^{-st}f(t)\,dt = -\Lap[tf(t)](s).\]
    For $k > 1$, replace $f(t)$ by $t^{k-1}f(t)$ and iterate.
\end{proof}

\subsubsection{Injectivity and the Inverse Transform}

\begin{proposition}[Injectivity]
    If $f,g:[0,\infty)\to\R$ are continuous and of exponential order, and $\Lap[f] = \Lap[g]$, then $f = g$.
\end{proposition}

Injectivity guarantees that the inverse Laplace transform $\Linv[F(s)]$ is well-defined.

\subsubsection{Laplace Transform of Derivatives}

\begin{theorem}
    \label{thm:deriv}
    If $f$ is differentiable and PCEO, then
    \begin{align}
        \Lap[f'](s) &= s\,\Lap[f](s) - f(0), \label{eq:first-deriv}\\
        \Lap[f''](s) &= s^2\,\Lap[f](s) - sf(0) - f'(0). \label{eq:second-deriv}
    \end{align}
\end{theorem}

\begin{proof}
    Integrating by parts:
    \[\Lap[f'](s) = \int_0^\infty e^{-st}f'(t)\,dt = \Big[e^{-st}f(t)\Big]_0^{+\infty} + s\int_0^\infty e^{-st}f(t)\,dt = -f(0) + s\,\Lap[f](s).\]
    Applying this to $f'$:
    \[\Lap[f''](s) = s\,\Lap[f'](s) - f'(0) = s\bigl(s\,\Lap[f](s) - f(0)\bigr) - f'(0) = s^2\Lap[f](s) - sf(0) - f'(0). \qedhere\]
\end{proof}

\begin{example}[Solving an IVP via Laplace transform]
    Solve $2y'' + 5y' - 3y = 2$, with $y(0) = 0$, $y'(0) = 1$.

    Applying $\Lap$ to both sides and denoting $Y(s) = \Lap[y](s)$:
    \[\Lap[y'] = sY(s),\qquad \Lap[y''] = s^2Y(s) - 1.\]
    The equation becomes
    \[2(s^2Y - 1) + 5(sY) - 3Y = \frac{2}{s},\]
    \[(2s^2 + 5s - 3)\,Y(s) = \frac{2}{s} + 2 = \frac{2(s+1)}{s}.\]
    Factoring $2s^2 + 5s - 3 = (2s - 1)(s + 3)$:
    \[Y(s) = \frac{2(s+1)}{s(2s-1)(s+3)} = \frac{s+1}{s(s - \tfrac{1}{2})(s+3)}.\]
    By partial fractions:
    \[Y(s) = -\frac{2}{3}\cdot\frac{1}{s} + \frac{6}{7}\cdot\frac{1}{s - \frac{1}{2}} - \frac{4}{21}\cdot\frac{1}{s+3}.\]
    Taking the inverse Laplace transform:
    \[\boxed{y(t) = -\frac{2}{3} + \frac{6}{7}\,e^{t/2} - \frac{4}{21}\,e^{-3t}.}\]
\end{example}

\subsubsection{Integration in the Time Domain}

\begin{corollary}
    \label{cor:integration}
    Let $f$ be PCEO with $F(s) = \Lap[f](s)$. Then
    \[\Lap\!\left[\int_0^t f(u)\,du\right] = \frac{F(s)}{s}, \qquad\text{equivalently,}\qquad \Linv\!\left[\frac{F(s)}{s}\right] = \int_0^t f(u)\,du.\]
\end{corollary}

\begin{proof}
    Let $g(t) = \int_0^t f(u)\,du$. Then $g(0) = 0$ and $g'(t) = f(t)$. By the derivative theorem:
    \[\Lap[f](s) = \Lap[g'](s) = s\,\Lap[g](s) - g(0) = s\,\Lap[g](s).\]
    Dividing by $s$ gives the result.
\end{proof}

\begin{example}
    Compute $\Linv\!\left[\dfrac{1}{s(s^2+1)}\right]$.

    Write $\dfrac{1}{s(s^2+1)} = \dfrac{1}{s}\cdot\dfrac{1}{s^2+1} = \dfrac{F(s)}{s}$ where $F(s) = \dfrac{1}{s^2+1}$, so $f(t) = \sin t$. By Corollary~\ref{cor:integration}:
    \[\Linv\!\left[\frac{1}{s(s^2+1)}\right] = \int_0^t \sin u\,du = \Big[-\cos u\Big]_0^t = 1 - \cos t.\]
    Alternatively, by partial fractions: $\dfrac{1}{s(s^2+1)} = \dfrac{1}{s} - \dfrac{s}{s^2+1}$, giving the same result.
\end{example}

\subsubsection{Convolution}

\begin{definition}[Convolution]
    The convolution of $f,g:[0,\infty)\to\R$ is
    \[(f\conv g)(t) = \int_0^t f(t-\tau)\,g(\tau)\,d\tau = \int_0^t f(\xi)\,g(t-\xi)\,d\xi.\]
\end{definition}

\begin{theorem}[Convolution Theorem]
    If $f$ and $g$ are PCEO with $F(s) = \Lap[f]$ and $G(s) = \Lap[g]$, then
    \[\Lap[f\conv g](s) = F(s)\,G(s), \qquad\text{equivalently,}\qquad \Linv[F(s)\,G(s)] = (f\conv g)(t).\]
\end{theorem}

\begin{proof}
    \[F(s)\,G(s) = \left(\int_0^\infty f(\xi)\,e^{-s\xi}\,d\xi\right)\!\left(\int_0^\infty g(\tau)\,e^{-s\tau}\,d\tau\right) = \iint_{[0,\infty)^2} f(\xi)\,g(\tau)\,e^{-s(\xi+\tau)}\,d\xi\,d\tau.\]
    Substituting $t = \xi + \tau$ and switching the order of integration:
    \[= \int_0^\infty e^{-st}\left(\int_0^t f(t-\tau)\,g(\tau)\,d\tau\right)dt = \Lap[f\conv g](s). \qedhere\]
\end{proof}

\begin{example}
    Compute $\Linv\!\left[\dfrac{a}{s^2(s^2+a^2)}\right]$.

    Write $\dfrac{a}{s^2(s^2+a^2)} = \dfrac{1}{s^2}\cdot\dfrac{a}{s^2+a^2}$. Since $\Linv[1/s^2] = t$ and $\Linv[a/(s^2+a^2)] = \sin(at)$:
    \[\Linv\!\left[\frac{a}{s^2(s^2+a^2)}\right] = t\conv\sin(at) = \int_0^t (t-\tau)\sin(a\tau)\,d\tau = \frac{at - \sin(at)}{a^2}.\]
\end{example}

\begin{example}[Integro-differential equation]
    Solve $y'(t) + 2y(t) + \displaystyle\int_0^t y(\tau)\,d\tau = \sin t$, with $y(0) = 1$.

    Note that $\int_0^t y(\tau)\,d\tau = (1\conv y)(t)$, so $\Lap[1\conv y] = Y(s)/s$. Let $Y(s) = \Lap[y](s)$ and take the transform:
    \[sY(s) - 1 + 2Y(s) + \frac{Y(s)}{s} = \frac{1}{s^2+1}.\]
    Multiplying by $s$:
    \[(s^2 + 2s + 1)\,Y(s) = s + \frac{s}{s^2+1}, \qquad\Longrightarrow\qquad (s+1)^2\,Y(s) = \frac{s(s^2+2)}{s^2+1},\]
    \[Y(s) = \frac{s(s^2+2)}{(s+1)^2(s^2+1)}.\]
    Using partial fractions and $\Lap[te^{-t}](s) = 1/(s+1)^2$ (from the $s$-derivative formula with $F(s) = 1/(s+1)$):
    \[\boxed{y(t) = e^{-t} - \tfrac{3}{2}\,te^{-t} + \tfrac{1}{2}\sin t.}\]
\end{example}

\subsection{Step Functions and Translation Theorems}
%-----------------------------------------------------------------------------

\begin{definition}[Unit step function / Heaviside function]
    \[u_c(t) = \begin{cases}0, & 0 \leq t < c,\\ 1, & t \geq c.\end{cases}\]
\end{definition}

\begin{example}
    Any piecewise function can be written with step functions:
    \[f(t) = \begin{cases}f_1(t),& 0\leq t<a,\\ f_2(t),& a\leq t<b,\\ f_3(t),& t\geq b\end{cases} \quad\Longrightarrow\quad f = u_0\,f_1 + u_a(f_2-f_1) + u_b(f_3-f_2).\]
\end{example}

\begin{theorem}[Translation Theorem 1]
    \[\Lap[u_c](s) = \int_c^\infty e^{-st}\,dt = \frac{e^{-cs}}{s},\quad s > 0.\]
\end{theorem}

\begin{definition}[Translation operator]
    \[\mathsf{t}_c(f)(t) = \begin{cases}0, & 0\leq t<c,\\ f(t-c), & t\geq c.\end{cases}\]
    This shifts $f$ to the right by $c$ and fills the gap with zero.
\end{definition}

\begin{theorem}[Translation Theorem 2 -- Time Shift]
    \label{thm:TT2}
    Let $f$ be PCEO with order $a$ and let $c > 0$. Then
    \[\Lap[\mathsf{t}_c(f)](s) = e^{-cs}\,\Lap[f](s), \qquad\text{equivalently,}\qquad \Linv[e^{-cs}F(s)] = \mathsf{t}_c(f)(t).\]
\end{theorem}

\begin{proof}
    \[\Lap[\mathsf{t}_c(f)](s) = \int_c^\infty e^{-st} f(t-c)\,dt.\]
    Substituting $u = t - c$:
    \[= \int_0^\infty e^{-s(u+c)}f(u)\,du = e^{-cs}\int_0^\infty e^{-su}f(u)\,du = e^{-cs}\,\Lap[f](s). \qedhere\]
\end{proof}

Key identity for TT2: to apply the time-shift theorem, write the function in the form $u(t-c)\cdot g(t-c)$, then
\[\Lap\{u(t-c)\,g(t-c)\} = e^{-cs}\,\Lap\{g(t)\}.\]

\begin{theorem}[Translation Theorem 3 -- Frequency Shift]
    \label{thm:TT3}
    Let $f$ be PCEO with order $a$ and let $c \in \R$. Then
    \[\Lap[e^{ct}f(t)](s) = F(s-c),\quad s > a + c, \qquad\text{equivalently,}\qquad \Linv[F(s-c)] = e^{ct}\,\Linv[F(s)].\]
\end{theorem}

\begin{proof}
    \[\Lap[f](s-c) = \int_0^\infty e^{-(s-c)t}f(t)\,dt = \int_0^\infty e^{-st}\bigl[e^{ct}f(t)\bigr]\,dt = \Lap[e^{ct}f(t)](s). \qedhere\]
\end{proof}

\begin{example}[TT2: piecewise with cosine]
    Let $f(t) = \begin{cases}\sin t, & 0\leq t < \pi/4,\\ \sin t + \cos(t - \pi/4), & t\geq\pi/4.\end{cases}$

    Rewrite: $f(t) = \sin t + u(t - \tfrac{\pi}{4})\cos(t - \tfrac{\pi}{4})$. By TT2:
    \[\Lap\{f\} = \frac{1}{s^2+1} + e^{-\frac{\pi}{4}s}\cdot\frac{s}{s^2+1}.\]
\end{example}

\begin{example}[TT2: three-piece function]
    Let $f(t) = \begin{cases}2, & 0\leq t<1,\\ 1+t, & 1\leq t<4,\\ t^2, & t\geq 4.\end{cases}$

    Write using step functions:
    \[f(t) = 2 + u(t-1)(t-1) + u(t-4)(t^2-t-1).\]
    Set $g_1(t) = t$ so that $u(t-1)(t-1) = u(t-1)\,g_1(t-1)$, and $g_2(t) = t^2 + 7t + 11$ so that $u(t-4)(t^2-t-1) = u(t-4)\,g_2(t-4)$. By TT2:
    \[\Lap\{f\} = \frac{2}{s} + e^{-s}\cdot\frac{1}{s^2} + e^{-4s}\left(\frac{2}{s^3} + \frac{7}{s^2} + \frac{11}{s}\right).\]
\end{example}

\begin{example}[Inverse transform with TT2]
    Compute $\Linv\!\left\{\dfrac{1-e^{-2s}}{s^2}\right\}$.

    Split: $\dfrac{1-e^{-2s}}{s^2} = \dfrac{1}{s^2} - e^{-2s}\cdot\dfrac{1}{s^2}$. Since $\Linv\{1/s^2\} = t$, by TT2:
    \[f(t) = t - u(t-2)(t-2) = \begin{cases}t, & 0\leq t<2,\\ 2, & t\geq 2.\end{cases}\]
\end{example}

\begin{example}[TT3: completing the square]
    Find $\Linv\!\left\{\dfrac{1}{s^2-4s+5}\right\}$.

    Complete the square: $s^2-4s+5 = (s-2)^2+1$, so
    \[\frac{1}{s^2-4s+5} = \frac{1}{(s-2)^2+1} = F(s-2),\quad\text{where } F(s) = \frac{1}{s^2+1}.\]
    Since $\Linv\{F(s)\} = \sin t$, by TT3:
    \[\Linv\!\left\{\frac{1}{s^2-4s+5}\right\} = e^{2t}\sin t.\]
\end{example}

\subsection{Impulse Functions}
%-----------------------------------------------------------------------------

Consider the rectangular pulse of width $2\tau$ and unit area:
\[d_\tau(t) = \begin{cases}\dfrac{1}{2\tau}, & |t| < \tau,\\[4pt] 0, & |t|\geq\tau.\end{cases} \qquad\text{Note: } \int_{-\infty}^{\infty}d_\tau(t)\,dt = \frac{2\tau}{2\tau} = 1.\]
The impulse function is the limiting ``function'' $\delta(t) = \lim_{\tau\to 0} d_\tau(t)$. For $t\neq 0$, $\delta(t) = 0$; at $t = 0$, $\delta(t) = \lim_{\tau\to 0} 1/(2\tau) = \infty$. Crucially, $\lim_{\tau\to 0}\int_{-\infty}^{\infty} d_\tau(t)\,dt = 1$.

\begin{definition}[Dirac delta function]
    The impulse function (Dirac delta function), with impulse at $t = t_0$, is defined by:
    \begin{enumerate}[nosep]
        \item $\delta(t-t_0) = 0$ for all $t \neq t_0$.
        \item $\displaystyle\int_{-\infty}^{\infty}\delta(t-t_0)\,dt = 1$.
    \end{enumerate}
\end{definition}

\begin{theorem}
    For $t_0 > 0$:
    \[\Lap\{\delta(t-t_0)\} = e^{-st_0}.\]
    In particular, $\Lap\{\delta(t)\} = 1$.
\end{theorem}

\begin{proof}
    \begin{align*}
        \Lap\{\delta(t-t_0)\}
        &= \lim_{\tau\to 0}\Lap\{d_\tau(t-t_0)\}
        = \lim_{\tau\to 0}\int_{t_0-\tau}^{t_0+\tau}e^{-st}\cdot\frac{1}{2\tau}\,dt\\
        &= \lim_{\tau\to 0}\frac{1}{2\tau}\cdot\frac{e^{-s(t_0-\tau)} - e^{-s(t_0+\tau)}}{s}
        = e^{-st_0}\lim_{\tau\to 0}\frac{e^{s\tau} - e^{-s\tau}}{2s\tau}.
    \end{align*}
    By L'H\^opital's rule:
    \[\lim_{\tau\to 0}\frac{e^{s\tau} - e^{-s\tau}}{2s\tau} = \lim_{\tau\to 0}\frac{s(e^{s\tau} + e^{-s\tau})}{2s} = 1.\]
    Therefore $\Lap\{\delta(t-t_0)\} = e^{-st_0}$.
\end{proof}

\begin{example}
    Solve $y'' + 2y' + y = \delta(t-5)$, with $y(0) = y'(0) = 0$.

    Taking the Laplace transform:
    \[(s^2+2s+1)\,Y(s) = e^{-5s} \quad\Longrightarrow\quad Y(s) = \frac{e^{-5s}}{(s+1)^2}.\]
    By TT2: $y(t) = \mathsf{t}_5\!\left[\Linv\!\left\{\dfrac{1}{(s+1)^2}\right\}\right]$.

    By TT3: $\Linv\!\left\{\dfrac{1}{(s+1)^2}\right\} = e^{-t}\,\Linv\!\left\{\dfrac{1}{s^2}\right\} = te^{-t}$.

    Therefore:
    \[\boxed{y(t) = u_5(t)\,(t-5)\,e^{-(t-5)} = \begin{cases}0, & 0\leq t<5,\\ (t-5)\,e^{-(t-5)}, & t\geq 5.\end{cases}}\]
\end{example}

\subsection{Second-Order Non-Autonomous Linear ODEs}

Consider the general equation
\begin{equation}
    \label{eq:general}
    P(x)\,y''(x) + Q(x)\,y'(x) + R(x)\,y(x) = 0.
\end{equation}

\begin{example}[Important special cases]\leavevmode
    \begin{itemize}[nosep]
        \item Bessel equation: $x^2y'' + xy' + (x^2 - \nu^2)y = 0$.
        \item Legendre equation: $(1-x^2)y'' - 2xy' + \alpha(\alpha+1)y = 0$.
        \item Chebyshev equation: $(1-x^2)y'' - xy' + \alpha^2 y = 0$.
    \end{itemize}
\end{example}

\begin{definition}[Ordinary point]
    A point $x = x_0$ is an ordinary point of (\ref{eq:general}) if $P(x_0)\neq 0$.
\end{definition}

Near an ordinary point, $P\neq 0$, so we can divide through to get the standard form
\begin{equation}
    \label{eq:standard}
    y'' + p(x)\,y' + q(x)\,y = 0,\qquad p(x) = \frac{Q(x)}{P(x)},\quad q(x) = \frac{R(x)}{P(x)}.
\end{equation}

Key assumption: $p$ and $q$ are analytic at $x_0$, i.e.\ they have power series expansions
\[p(x) = \sum_{n\geq 0}p_n(x-x_0)^n,\qquad q(x) = \sum_{n\geq 0}q_n(x-x_0)^n\]
with positive radii of convergence. By the Cauchy--Kovalevskaya theorem (originally for PDEs), if $p$ and $q$ are analytic at $x_0$, then $y(x)$ is analytic at $x = x_0$.

%-----------------------------------------------------------------------------
\subsection{The Power Series Method}
%-----------------------------------------------------------------------------

Since the solution is analytic, write $y(x) = \sum_{n=0}^\infty a_n(x-x_0)^n$, with $a_0 = y(x_0)$ and $a_1 = y'(x_0)$. The derivatives are:
\begin{align*}
    y'(x) &= \sum_{n=0}^\infty (n+1)\,a_{n+1}(x-x_0)^n, \\
    y''(x) &= \sum_{n=0}^\infty (n+2)(n+1)\,a_{n+2}(x-x_0)^n.
\end{align*}
The products $p(x)y'(x)$ and $q(x)y(x)$ are computed via the Cauchy product:
\[p(x)y'(x) = \sum_{n\geq 0}(p\conv\tilde{a})_n\,(x-x_0)^n,\qquad q(x)y(x) = \sum_{n\geq 0}(q\conv a)_n\,(x-x_0)^n,\]
where $\tilde{a}_k = (k+1)a_{k+1}$ and $(p\conv\tilde{a})_n = \sum_{k=0}^n (k+1)\,a_{k+1}\,p_{n-k}$.

Substituting into (\ref{eq:standard}) and matching coefficients gives the recurrence relation:
\[\boxed{a_{n+2} = \frac{-1}{(n+2)(n+1)}\Big[(p\conv\tilde{a})_n + (q\conv a)_n\Big],\quad n\geq 0.}\]

\begin{example}[$y'' + y = 0$]
    Here $p = 0$, $q = 1$, $x_0 = 0$. The recurrence simplifies to:
    \[(n+2)(n+1)\,a_{n+2} + a_n = 0 \quad\Longrightarrow\quad a_{n+2} = -\frac{a_n}{(n+2)(n+1)}.\]
    With $a_0$ and $a_1$ free, this produces the familiar series for $\cos$ and $\sin$.
\end{example}

\begin{example}[Airy's equation: $y'' - xy = 0$, $y(0) = 1$, $y'(0) = 0$]
    Take $x_0 = 0$. Substituting:
    \[y''(x) = \sum_{n=0}^\infty (n+2)(n+1)\,a_{n+2}\,x^n,\qquad xy(x) = \sum_{n=1}^\infty a_{n-1}\,x^n.\]
    Setting $y'' - xy = 0$:
    \[2a_2 + \sum_{n=1}^\infty \Big[(n+2)(n+1)\,a_{n+2} - a_{n-1}\Big]x^n = 0.\]
    This gives $a_2 = 0$ and the recurrence:
    \[a_{n+2} = \frac{a_{n-1}}{(n+2)(n+1)},\quad n\geq 1.\]
    With $a_0 = 1$, $a_1 = 0$: non-zero coefficients occur at indices $n = 0, 3, 6, 9, \ldots$ The Airy function is:
    \[y(x) = \sum_{k=0}^\infty \frac{x^{3k}}{(2\cdot 3)(5\cdot 6)\cdots(3k-1)(3k)}.\]
    By the ratio test:
    \[\left|\frac{b_{k+1}}{b_k}\right| = \frac{|x|^3}{(3k+2)(3k+3)} \to 0 \quad\text{as } k\to\infty,\]
    so the radius of convergence is $\rho = +\infty$.
\end{example}

\begin{problem}
    Redo Airy's equation with $y(0) = 0$, $y'(0) = 1$ (this gives the second linearly independent solution).
\end{problem}

\subsection{Power Series for IVPs in Nonlinear Systems}
%-----------------------------------------------------------------------------

Consider the IVP
\[\dot{y} = f(y),\qquad y(0) = y_0 \in U \subseteq \R^n,\]
where $f: U \to \R^n$ is analytic and $U$ is open. By the Cauchy--Kovalevskaya theorem, the solution is analytic, so we seek $y(t) = \sum_{n=0}^\infty a_n t^n$ with $a_0 = y_0$.

Substituting into the ODE:
\[\sum_{n=0}^\infty (n+1)\,a_{n+1}\,t^n = f\!\left(\sum_{n=0}^\infty a_n t^n\right) = \sum_{n=0}^\infty \Phi_n(a)\,t^n,\]
giving the recurrence:
\[\boxed{a_{n+1} = \frac{\Phi_n(a)}{n+1},\quad n\geq 0,\qquad a_0 = y_0.}\]

\begin{example}[Logistic equation: $y' = y(1-y)$, $y(0) = 1/2$]
    Here $f(y) = y - y^2$. Expanding $y^2$ via the Cauchy product:
    \[y^2(t) = \sum_{n=0}^\infty (a\conv a)_n\,t^n,\qquad (a\conv a)_n = \sum_{k=0}^n a_k\,a_{n-k}.\]
    The recurrence becomes:
    \[a_{n+1} = \frac{1}{n+1}\Big[a_n - (a\conv a)_n\Big],\quad n\geq 0.\]
    With $a_0 = 1/2$: $a_1 = 1/4$, $a_2 = 0$, $a_3 = -1/48$, $a_4 = 0$, $a_5 = 1/480$, so
    \[y(t) = \frac{1}{2} + \frac{1}{4}\,t - \frac{1}{48}\,t^3 + \frac{1}{480}\,t^5 + \cdots \approx \sum_{n=0}^N a_n t^n.\]
\end{example}

%-----------------------------------------------------------------------------
\subsection{Numerical Method via Power Series}
%-----------------------------------------------------------------------------

Given $y' = f(y)$, $y(0) = y_0$, fix an order $N$ and compute $a_0, a_1, \ldots, a_N$ from the recurrence. This gives the numerical approximation of order $N$:
\[y^N(t) = \sum_{n=0}^N a_n\,t^n.\]

Case $N = 1$ (Euler's method):
\[y^1(t) = a_0 + a_1 t = y_0 + f(y_0)\,t.\]
Picking a step size $h > 0$ and iterating:
\[\boxed{y_{n+1} = y_n + h\,f(y_n),\qquad t_{n+1} = t_n + h.}\]

Case $N > 1$ (Taylor method of order $N$): compute $y_1 = \sum_{n=0}^N a_n h^n$ and apply the method iteratively, achieving higher-order accuracy.

\subsection{Picard Iterations Convergence}
\begin{proof}[Proof of Picard Iterations Convergence]
    \label{pf:picard-convergence}
    Pick $t \in [t_0, t_0 + \epsilon]$ (the proof is similar for $t \in [t_0 - \epsilon, t_0]$).
    The goal is to show that $\{y_k(t)\}_{k=0}^{\infty}$ is a Cauchy sequence\footnote{Cauchy sequence is a sequence that has a limit in a metric space $\R^n$.} in $\R^n$.\footnote{Evaluated on compacted cylinder, understanding epsilon, the L Lipschitz constant coming from somewhere, not on the analysis background such as the Banach fixed point theorem.}
    We prove by induction that
    \[(**)\quad ||y_m(t) - y_{m - 1}(t)|| \leq L^{m - 1}M_{\alpha,\delta} \frac{(t - t_0)^m}{m!},\:\forall m \geq 1.\]
    \textbf{Base case \(m=1\):}
    \[||y_1(t) - y_0(t)|| = \left\| \int_{t_0}^{t} f(y_0,s)ds \right\| \leq \int_{t_0}^{t}||f(y_0,s)||ds \leq M_{\alpha,\delta}|t - t_0| \leq \alpha.\]
    \textbf{Induction step:}
    \[||y_{m + 1}(t) - y_m(t)|| \leq \int_{t_0}^{t}||f(y_m(s),s) - f(y_{m-1}(s),s)||ds.\]
    Since $D_{\alpha,\delta}$ is compact and $f$ is Lipschitz on $D_{\alpha,\delta}$, there exists $L$ such that
    \[||f(x,t) - f(y,t)|| \leq L||x - y||.\]
    Thus,
    \[\leq L \int_{t_0}^{t}||y_m(s) - y_{m - 1}(s)||ds.\]
    Using (**),
    \[\leq L^m M_{\alpha,\delta} \frac{1}{(m-1)!} \int_{t_0}^{t}(s - t_0)^{m-1}ds
    = L^m M_{\alpha,\delta} \frac{(t - t_0)^m}{m!}.\]
    Hence (**) holds.
    In particular, for all $\rho \geq 1$,
    \[||y_\rho(t) - y_{\rho-1}(t)|| \leq M_{\alpha,\delta} \frac{(L(t - t_0))^{\rho}}{(\rho)!} < \frac{M_{\alpha,\delta}}{L}\frac{(L\epsilon)^\rho}{\rho!}.\]
    Let $m,p \geq 1$:
    \[||y_{m+p}(t) - y_{m+1}(t)|| \leq \sum_{k=1}^{p-1}||y_{m+k+1}(t) - y_{m+k}(t)||.\]
    So,
    \[< \frac{M_{\alpha,\delta}}{L}\sum_{j=m+2}^{m+p}\frac{(L\epsilon)^j}{j!}.\]
    Since \(e^{L\epsilon} = \sum_{j=0}^{\infty}\frac{(L\epsilon)^j}{j!}\) converges,
    \[\to_{m,p \to +\infty} 0.\]
    Thus $\{y_k(t)\}$ is Cauchy and converges to $y(t)$.
    Taking limits in the iteration,
    \[y(t) = y_0 + \int_{t_0}^{t} f(y(s),s)ds.\]
\end{proof}

\begin{solution}[Existence and Uniqueness Theorem]
    Let $f(y,t) = t^2+y^2$. This function is continuous and has continuous partial derivatives $\forall(t,y) \in \R^2$, ensuring a unique local solution. To find the guaranteed interval of existence $\epsilon$, we define a rectangular domain $D$ centered at the origin:
    \[D_{\alpha,\delta} = \{(y,t) : \|y\| \leq \alpha, |t| \leq \delta\}.\]
    Now we need to bound the function. We find the maximum value $M$ of $|f(t, y)|$ on this domain:
    \[M = \sup_{(t, y) \in D_{\alpha, \delta}} |t^2 + y^2| = \delta^2 + \alpha^2.\]
    Now we determine the Existence Interval. According to the theorem, the solution is guaranteed to exist for $|t| \leq \epsilon$, where:
    \[\epsilon = \min \left(\delta, \frac{\alpha}{M}\right) = \min \left(\delta, \frac{\alpha}{\delta^2 + \alpha^2}\right).\]
    To find the largest possible value for $\epsilon$, we first maximize $h(\alpha) = \frac{\alpha}{\delta^2 + \alpha^2}$ with respect to $\alpha$ for a fixed $\delta$:
    \[h'(\alpha) = \frac{(\delta^2 + \alpha^2)(1) - \alpha(2\alpha)}{(\delta^2 + \alpha^2)^2} = \frac{\delta^2 - \alpha^2}{(\delta^2 + \alpha^2)^2}.\]
    Setting $h'(\alpha) = 0$ gives $\alpha = \delta$.Substituting $\alpha = \delta$ back into our expression for $\epsilon$:
    \[\epsilon = \min \left( \delta, \frac{\delta}{\delta^2 + \delta^2} \right) = \min \left( \delta, \frac{1}{2\delta} \right).\]
    The maximum value of this minimum occurs when the two terms are equal:
    \[\delta = \frac{1}{2\delta} \implies \delta^2 = \frac{1}{2} \implies \delta = \frac{1}{\sqrt{2}}.\]
    The largest guaranteed interval of existence $\epsilon$ using this rectangular domain is:
    \[\epsilon = \frac{1}{\sqrt{2}} \approx 0.707.\]
\end{solution}

\begin{solution}[Complex eigenvalues and real-valued solutions]
    We are given the system
    \[\vec{y'} = A\vec{y},\qquad A = \begin{pmatrix}-3 & -2 \\ 5 & 3\end{pmatrix}.\]
    \textbf{(a) Eigenvalues and eigenvectors.}
    The characteristic polynomial is
    \[p(\lambda) = \det(A-\lambda I) = \begin{vmatrix}
    -3-\lambda & -2 \\ 5 & 3-\lambda\end{vmatrix} = (-3-\lambda)(3-\lambda)+10 = \lambda^2 + 1.\]
    Setting $p(\lambda)=0$ gives us the eigenvalues $\lambda_1 = i$ and $\lambda_2 = -i$.
    
    For $\lambda_1=i$, we solve $(A-iI)u=0$:
    \[\begin{pmatrix}-3-i & -2 \\ 5 & 3-i\end{pmatrix}\begin{pmatrix}a \\ b\end{pmatrix}=0.\]
    From the first row, $(-3-i)a - 2b = 0$ implies $a = \frac{2}{-3-i} b$. Choosing $b=1$, we obtain our first solution
    \[u_1 = \begin{pmatrix}\dfrac{2}{-3-i}\\1\end{pmatrix}.\]

    For $\lambda_2=-i$, we similarly solve $(A+iI)u=0$:
    \[\begin{pmatrix}-3+i & -2 \\ 5 & 3+i\end{pmatrix}\begin{pmatrix}a \\ b\end{pmatrix}=0,\]
    which gives $a = \frac{2}{-3+i} b$. Again choosing $b=1$, we obtain our second solution
    \[u_2 = \begin{pmatrix}\dfrac{2}{-3+i}\\1\end{pmatrix}.\]

    Hence,
    \[\boxed{\lambda_1 = i,\quad \vec v_1 = \begin{pmatrix}
    \dfrac{2}{-3-i}\\1\end{pmatrix}, \qquad \lambda_2 = -i,\quad \vec v_2 = \begin{pmatrix}\dfrac{2}{-3+i} \\ 1\end{pmatrix}.}\]
    \textbf{(b) Real-valued solution of the initial value problem.}\\
    From the eigenpair $(\lambda_1,u_1)$\footnote{Since the matrix has real coefficients, the second eigenpair $(\lambda_2,u_2)$ is simply the complex conjugate of $(\lambda_1,u_1)$, so it produces the same two real solutions after taking real and imaginary parts. Therefore, working with one eigenpair is sufficient.}, a complex-valued solution is
    \[\vec y(t) = e^{it} u_1.\]
    Using Euler's formula \eqref{eq:EulerFormula}, we write
    \[\vec y(t) = (\cos t + i\sin t)\begin{pmatrix}\dfrac{2}{-3-i}\\1\end{pmatrix} = (\cos t + i\sin t)\begin{pmatrix}-\dfrac{3}{5} + \dfrac{1}{5}i\\1\end{pmatrix}.\]
    Separating real and imaginary parts yields two real-valued solutions:
    \[\vec y_1(t) = \begin{pmatrix}-3\cos t - \sin t \\ 5 \cos t\end{pmatrix}, \qquad \vec y_2(t) = \begin{pmatrix} \cos t - 3 \sin t \\ 5 \sin t\end{pmatrix}.\]
    Therefore, the general real solution is
    \[\vec y(t) = c_1 \vec y_1(t) + c_2 \vec y_2(t).\]
    Applying the initial conditions $y_(0)=\begin{pmatrix}7\\-15\end{pmatrix}$, we compute
    \[\vec y_1(0)=\begin{pmatrix}-3\\5\end{pmatrix}, \qquad \vec y_2(0)=\begin{pmatrix}1\\0\end{pmatrix}.\]
    Thus,
    \[\begin{pmatrix}7\\-15\end{pmatrix} = c_1\begin{pmatrix}-3\\5\end{pmatrix} + c_2\begin{pmatrix}1\\0\end{pmatrix}.\]
    Solving gives
    \[c_1 = -3, \qquad c_2 = -2.\]
    Therefore, the real-valued solution of the initial value problem is
    \[\vec y(t) = c_1 \vec y_1(t) + c_2 \vec y_2(t) = -3 \begin{pmatrix}-3\cos t - \sin t \\ 5 \cos t \end{pmatrix}-2 \begin{pmatrix}-3\sin t + \cos t \\ 5 \sin t \end{pmatrix}.\]
    You can simplify each component if desired:
    \[\vec y(t) = \begin{pmatrix}9\cos t + 3 \sin t + 6 \sin t - 2 \cos t \\ -15 \cos t - 10 \sin t\end{pmatrix} = \begin{pmatrix}7 \cos t + 9 \sin t \\ -15 \cos t - 10 \sin t\end{pmatrix}.\]
\end{solution}
