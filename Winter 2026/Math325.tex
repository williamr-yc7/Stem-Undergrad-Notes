\section{Prerequisite Knowledge}
\subsection{Real Analysis}
\subsubsection{Norms}
\begin{definition}[Norm]
    A norm $\|\cdot\|$ on $\mathbb{R}^n$ is a way of measuring the 
    ``size'' or ``length'' of a vector $x \in \mathbb{R}^n$. It must satisfy:
    \begin{itemize}
        \item $\|x\| \geq 0$, and $\|x\| = 0$ only if $x = 0$,
        \item $\|\alpha x\| = |\alpha|\,\|x\|$ for any scalar $\alpha$,
        \item $\|x + y\| \leq \|x\| + \|y\|$ \quad (triangle inequality).
    \end{itemize}
\end{definition}
\begin{example}
    The three most common norms on $\mathbb{R}^n$ are:
    \begin{itemize}
        \item \textbf{Euclidean norm:} $\|x\|_2 = \sqrt{x_1^2 + \cdots + x_n^2}$ 
        (standard geometric length),
        \item \textbf{Supremum norm:} $\|x\|_\infty = \max\{|x_1|,\ldots,|x_n|\}$ 
        (size of the largest component),
        \item \textbf{$\ell^1$ norm:} $\|x\|_1 = |x_1| + \cdots + |x_n|$ 
        (sum of all components, also called the taxicab norm).
    \end{itemize}
\end{example}
\begin{remark}
    In this course, any norm can be used unless stated otherwise. The choice 
    of norm does not affect whether a solution exists or is unique.
\end{remark}

\subsubsection{Sequences and Convergence}
\begin{definition}[Cauchy Sequence]
    A sequence of vectors $\{x_k\} \subset \mathbb{R}^n$ is called a Cauchy 
    sequence if its terms eventually get arbitrarily close to one another:
    for every $\varepsilon > 0$, there exists $N \in \mathbb{N}$ such that
    \[\|x_m - x_n\| < \varepsilon \quad \text{for all } m, n \geq N.\]
\end{definition}
\begin{theorem}
    Every Cauchy sequence in $\mathbb{R}^n$ converges to a point in $\mathbb{R}^n$.
\end{theorem}
\begin{remark}
    This is the key fact used in Picard iteration: we show that the sequence 
    of approximate solutions $\{y_k\}$ is Cauchy, and therefore it must 
    converge to an actual solution.
\end{remark}

\subsubsection{Uniform Convergence}
\begin{definition}[Uniform Convergence]
    A sequence of functions $f_k : J \to \mathbb{R}^n$ converges 
    \emph{uniformly} to $f : J \to \mathbb{R}^n$ if
    \[\lim_{k \to \infty} \sup_{t \in J} \|f_k(t) - f(t)\| = 0.\]
    In plain terms: the approximation error $\|f_k(t) - f(t)\|$ becomes 
    small \emph{simultaneously} for all $t \in J$, not just pointwise.
\end{definition}
\begin{theorem}
    If $f_k \to f$ uniformly on $J$ and each $f_k$ is continuous, then $f$ 
    is also continuous. Moreover,
    \[\lim_{k\to\infty}\int_J f_k(t)\,dt = \int_J f(t)\,dt.\]
\end{theorem}
\begin{remark}
    The second statement says that for uniformly convergent sequences, 
    you can swap the limit and the integral. This is used in the Picard 
    iteration proof to pass the limit inside the integral and confirm that 
    the limit function actually solves the IVP.
\end{remark}

\subsubsection{The Mean Value Inequality}
\label{MVI}
\begin{theorem}[Mean Value Inequality]
    Let $U \subseteq \mathbb{R}^n$ be open and $f : U \to \mathbb{R}^n$ 
    be $C^1$. For any compact set $K \subset U$ and all $x, y \in K$,
    \[\|f(y) - f(x)\| \leq \left(\sup_{z \in K}\|Df(z)\|\right)\|y - x\|,\]
    where $Df$ denotes the Jacobian matrix of $f$.
\end{theorem}
\begin{remark}
    This generalizes the familiar scalar fact $|f(b)-f(a)| \leq 
    \sup|f'|\cdot|b-a|$ to higher dimensions. Its main consequence for 
    this course is that any $C^1$ function is locally Lipschitz: on a 
    compact set $K$, the sup of $\|Df\|$ is finite, providing a Lipschitz 
    constant. This is why the hypothesis ``$f$ is $C^1$'' is often used 
    as a sufficient condition for the Existence and Uniqueness Theorem.
\end{remark}

\subsection{Intro}
\marginpar{January 06, 2026.}
\begin{definition}[Differential Equation]
    A differential equation (DE) is a relation that involves an unknown function and some of its derivatives.
\end{definition}
\begin{definition}[Ordinary Differential Equation]
    An ordinary differential equation (ODE) is a differential equation whose unknown function depends on one independent variable only.
\end{definition}
\begin{definition}[Partial Differential Equation]
    A partial differential equation (PDE) is a differential equation whose unknown function depends on more than one independent variable. \textbf{Will not be taught in this course.}
\end{definition}

\begin{example}[Motion under gravity and linear drag]
    A ball of mass $m$ falls subject to gravity and linear air resistance. Let $v(t)$ be its velocity (positive downward). The net force gives
    \[m\frac{dv}{dt} = mg - \lambda v,\]
    where $\lambda \geq 0$ is the drag coefficient. Writing $v = y'$, this becomes the second-order ODE
    \[my'' + \lambda y' = mg.\]
\end{example}
\begin{example}[The Malthusian Growth Model]
    Let $N(t)$ be a population size. Assuming the rate of change is 
    proportional to the current population,
    \[\frac{dN}{dt} = rN,\]
    where $r$ is the growth rate ($r>0$) or decay rate ($r<0$). Separating variables and applying the initial condition $N(0)=N_0$ gives the unique solution
    \[N(t) = N_0 e^{rt}.\]
\end{example}
\begin{example}[The Logistic Equation]
    If the growth rate depends on the population itself, with carrying capacity $K$, the simplest choice $R(N) = r(1-N/K)$ gives
    \[\frac{dN}{dt} = rN\!\left(1 - \frac{N}{K}\right),\]
    called the logistic equation. It models growth when $N < K$ and decay when $N > K$.
\end{example}
\begin{example}[Heat Equation]
    The Heat Equation is an example of a PDE. Let \(u = u(x,t)\), \(\frac{\partial u}{\partial t} = \frac{\partial^2 u}{\partial x^2}\). This PDE denotes the temperature of a body at time t and at position x.
\end{example}

\subsection{Classification}
\marginpar{January 08, 2026.}
\subsubsection{The Order}
\begin{definition}
    The order of an ODE is the order of the highest derivative that appears in the equation.
\end{definition}
\begin{example}
    $N' = rN$ is first order. \quad $y'' + 2y' = e^t$ is second order.
\end{example}

\subsubsection{Scalar vs.\ System}
\begin{definition}[Scalar ODE]
    A scalar ODE has one unknown function.
\end{definition}
\begin{example}[Scalar ODE]
    \[y'' + y = 0\]
\end{example}
\begin{definition}[System of ODEs]
    A system of ODEs has several unknown functions.
\end{definition}
\begin{example}[System of ODEs]
    \[\begin{cases}
        y_1' = y_2\\
        y_2' = -y_1
    \end{cases}\]
\end{example}
\begin{definition}[System of first order ODEs]
    A first order system has the form
    \[y'(t)=f(y(t),t),\]
    where \(y(t)\in\mathbb{R}^n\) and \(f:\mathbb{R}^n\times\mathbb{R}\to\mathbb{R}^n\).
    In components,
    \[y_i'(t)=f_i(y_1,\dots,y_n,t), \quad i=1,\dots,n.\]
    Any \(n\)th-order ODE
    \[y^{(n)}=G(t,y,y',\dots,y^{(n-1)})\]
    can be written as a first order system by setting
    \[y_1=y,\; y_2=y',\; \dots,\; y_n=y^{(n-1)},\]
    which gives
    \[\begin{cases}y_1'=y_2,\\y_2'=y_3,\\\vdots\\y_{n-1}'=y_n,\\y_n'=G(t,y_1,\dots,y_n).\end{cases}\]
\end{definition}
\begin{example}[System of First Order ODEs]
    Consider the second-order ODE
    \[y'' + 2y' + y = e^t.\]
    Define new variables
    \[y_1 = y, \qquad y_2 = y'.\]
    Then the system becomes
    \[\begin{pmatrix}y_1' \\ y_2'\end{pmatrix} = \begin{pmatrix}0 & 1 \\ -1 & -2\end{pmatrix} \begin{pmatrix}y_1 \\ y_2\end{pmatrix} + \begin{pmatrix} 0 \\ e^t \end{pmatrix}.\]
    Let
    \[A = \begin{pmatrix} 0 & 1 \\ -1 & -2 \end{pmatrix},\quad r(t) = \begin{pmatrix} 0 \\ e^t \end{pmatrix}.\]
\end{example}
\begin{problem}[System of First Order ODEs]
    Rewrite this third order ODE as a first order system:
    \[y''' + 4y' - y = 0\]
\end{problem}

\subsubsection{Linearity}
\begin{definition}[Linearity]
    An ODE is linear if the unknown function and its derivatives appear only to the first power, are not multiplied together, and are not composed with nonlinear functions. They may be multiplied by functions of the independent variable.
\end{definition}
\begin{example}[Linearity]
    Consider the following ODEs:
    \begin{itemize}
        \item Linear: $y' +3y = 0$
        \item Linear: $y'' - 2xy' + y = cos(x)$
        \item Non-linear: $y' + y^2 = 0$
        \item Non-linear: $y'' + sin(y) = 0$
    \end{itemize}
\end{example}
\begin{example}[Lorenz system]
    \label{Lorenzsystem}
    \[\begin{aligned}y_1' &= \sigma (y_2-y_1), \\ y_2' &= \rho y_1 - y_2 - y_1y_3, \\ y_3' &= y_1y_2 - \beta y_3,\end{aligned}\]
    where \(\sigma,\rho,\beta\) are parameters. This is a nonlinear first order system in \(\mathbb{R}^3\).
\end{example}

\subsubsection{Autonomy}
\begin{definition}[Autonomy]
    The \(n\)\textsuperscript{th} order ODE \(F(t, y, y', \ldots, y^{(n)}) = 0\) is autonomous if \(F\) does not depend explicitly on t, that is, it can be written as \(F(y, y', \ldots, y^{(n)}) = 0\). Otherwise, it is non-autonomous.
\end{definition}
\begin{example}[Autonomy]
    Consider the following ODEs:
    \begin{itemize}
        \item Non-autonomous: \(y'' + 2y' + y - e^t = 0\)
        \item Autonomous: \(N'(t) = rN(t)\)
        \item Non-autonomous: \(y'(t) = ty(t)\)
    \end{itemize}
\end{example}
\begin{example}
    A classic, simple example of an autonomous first-order system is the linear growth/decay model:
    \[\frac{dy}{dt} = ky\]
    Here, \(f(y,t) = ky\), which depends only on \(y\) (where \(k\) is a constant), making it autonomous.
\end{example}
\begin{remark}
    The Lorenz system \eqref{Lorenzsystem} is an example of an autonomous system.
\end{remark}

\subsection{Solutions and Initial Value Problems}
\begin{definition}[Solution]
    A solution of $y'(t) = f(y(t), t)$ on an interval $J$ is a 
    differentiable function $y : J \to D$ such that
    \[y'(t) = f(y(t), t) \quad \forall\, t \in J.\]
\end{definition}
\begin{example}[Explicit Solution]
    Consider the following ODE $y' + y = 1$. We can verify that \(y(t) = e^{-t} + 1\), and therefore \(y'(t) = -e^{-t}\), is a solution on \(\mathbb{R}\). Indeed,
    \[y' + y = -e^{-t} + (e^{-t} + 1) = 1.\]
\end{example}
\begin{example}[Implicit Solution]
    Consider the ODE $y \frac{dy}{dx} = x$. This is a nonautonomous, nonlinear, first-order scalar ODE. Separating $y\,dy = x\,dx$ and integrating gives the implicit solution $x^2 - y^2 = C, \quad C \in \mathbb{R}$.
\end{example}
\marginpar{January 13, 2026}
\begin{definition}[Initial Value Problem (IVP)]
    An initial value problem consists of an ODE together with a condition $y(t_0)=y_0$, where the value of the solution is prescribed at an initial time \(t_0\).
\end{definition}

\subsection{Existence and Uniqueness Theorem}
\subsubsection{Lipschitz continuity}
\begin{definition}[Lipschitz Continuity]
    A function $f : D \to \mathbb{R}^n$ is Lipschitz continuous on $D \subseteq \mathbb{R}^n$ if there exists $L \geq 0$ such that
    \[z||f(y_1) - f(y_2)|| \le L||y_1 - y_2|| \quad \forall y_1,y_2 \in D, z \in \mathbb{R}.\]
    The smallest such $L$ is called the Lipschitz constant, denoted $\mathrm{Lip}(f)$.
\end{definition}
\begin{remark}[How to find a Lipschitz constant]
    If $f$ is differentiable on $D \subset \mathbb{R}$ and 
    $|f'(y)| \leq M$ for all $y \in D$, then $f$ is Lipschitz with 
    $\mathrm{Lip}(f) = M$.
\end{remark}
\begin{example}
    Let \(f(y) = 4y - 5\), \(D = \mathbb{R}\), and \(||\cdot|| = |\cdot|\).
    Then
    \[|f(y_1) - f(y_2)| = |4y_1 - 4y_2| = 4|y_1 - y_2|.\]
    So \(f\) is Lipschitz on \(\mathbb{R}\) with \(Lip(f) = 4\).
\end{example}
\begin{example}
    Let
    \[f(y) = \frac{1}{y - 1}, \quad f'(y) = -\frac{1}{(y - 1)^2}, \quad D = (1, +\infty).\]
    This function is not Lipschitz continuous on \(D\) since as \(y \to 1^+\), the derivative \(f'(y)\) approaches \(-\infty\), and so the ratio \(\frac{|f(y_2) - f(y_1)|}{|y_2 - y_1|}\) approaches \(\infty\) for any \(y_1,y_2 \in D\). Therefore, there is no Lipschitz constant for \(f\) on \(D\).\\
    To make this function Lipschitz continuous, we fix \(\delta > 1\) and define \(D_\delta = (\delta, +\infty)\). For \(y_1,y_2 \in D_\delta\), by the Mean Value Theorem, there exists \(z \in (y_1,y_2)\) such that
    \[f(y_2) - f(y_1) = f'(z)(y_2 - y_1).\]
    So
    \[|f(y_2) - f(y_1)| \le \frac{1}{(z-1)^2}|y_2 - y_1|\le \frac{1}{(\delta-1)^2}|y_2 - y_1|.\]
    Thus \(f\) is Lipschitz on \(D_\delta\) with
    \[Lip(f) = \frac{1}{(\delta - 1)^2}.\]
\end{example}
\begin{example}
    Suppose \(f\) is differentiable and satisfies
    \[-4 \le f'(y) \le 9 \quad \text{for all } y \in \mathbb{R}.\]
    Then
    \[|f'(y)| \le 9 \quad \forall y,\]
    so \(f\) is Lipschitz on \(\mathbb{R}\) with Lipschitz constant \(L = 9\).
\end{example}

\subsubsection{Local Lipschitz continuity}
\begin{definition}[Locally Lipschitz]
    Let $D \subseteq \mathbb{R}^n$ be open. A function $f : D \to \mathbb{R}^n$ is \emph{locally Lipschitz} if for every compact set $K \subset D$ there exists $L > 0$ such that
    \[\|f(y_1) - f(y_2)\| \leq L\|y_1 - y_2\| \quad \forall\, y_1, y_2 \in K.\]
\end{definition}
\begin{remark}
    Lipschitz $\Rightarrow$ locally Lipschitz, but not conversely.  
    Any $C^1$ function is locally Lipschitz (by the Mean Value 
    Inequality from Section \eqref{MVI}).
\end{remark}
\begin{example}[Locally Lipschitz but not Lipschitz]
    \[f(y) = y^2 \in \mathbb{R}.\]
    Since \(f'(y)=2y\) is unbounded on \(\mathbb{R}\), no single constant works on the whole domain, so \(f\) is not Lipschitz.  
    However, on any bounded set \(K=[-M,M]\), \(|f'(y)| \le 2M\), so \(f\) is locally Lipschitz.
\end{example}
\begin{example}[Lipschitz (hence locally Lipschitz)]
    \[f(y) = \sin y.\]
    Since \(|f'(y)| = |\cos y| \leq 1\), $\forall y$, $f$ is Lipschitz on \(\mathbb{R}\), with \(Lip(f) = 1\).
\end{example}
\begin{example}[Continuous but not locally Lipschitz]
    \[f(y) = \sqrt{|y|}.\]
    The derivative is unbounded near \(y = 0\), so no Lipschitz constant exists even locally. Hence \(f\) is not locally Lipschitz.
\end{example}

\subsubsection{Integral Form of the IVP}
\marginpar{January 13, 2026}
\begin{lemma}
    A function \(y\) solves the IVP if and only if
    \[y(t) = y_0 + \int_{t_0}^{t} f(y(s),s)ds.\]
\end{lemma}
\begin{proof}
    If \(y' = f(y,t)\) and \(y(t_0) = y_0\), then by the Fundamental Theorem of Calculus,
    \[y(t) = y_0 + \int_{t_0}^{t} f(y(s),s)ds.\]
    Conversely, differentiating the right-hand side gives
    \[y'(t) = f(y(t),t), \quad y(t_0) = y_0.\]
\end{proof}

\subsubsection{Picard operator}
The integral form motivates defining the following operator. Since $(y_0, t_0) \in D \times (a,b)$ and the set is open, choose $\alpha, \delta > 0$ such that
\[D_{\alpha,\delta} = \{(y,t) : \|y-y_0\| \leq \alpha,\; |t-t_0| \leq \delta\} \subset D \times (a,b),\]
and set
\[M_{\alpha,\delta} = \sup_{(y,t)\in D_{\alpha,\delta}}\|f(y,t)\| < +\infty, \qquad \varepsilon = \min\!\left(\delta,\frac{\alpha}{M_{\alpha,\delta}}\right), \qquad J = (t_0-\varepsilon, t_0+\varepsilon).\]
\begin{definition}[Picard Operator]
    For any $y : J \to \mathbb{R}^n$ with $y(t_0) = y_0$ and 
    $(y(t),t) \in D_{\alpha,\delta}$ for all $t \in J$, define
    \[T(y)(t) = y_0 + \int_{t_0}^{t} f(y(s),s)\, ds.\]
    A function $y$ solves the IVP if and only if $T(y) = y$, i.e.\ 
    $y$ is a \emph{fixed point} of $T$.
\end{definition}
\begin{lemma}[Invariance]
    If \(y(t_0)=y_0\) and \((y(t),t) \in D_{\alpha,\delta}\) \(\forall t \in J\), then \(T(y)(t_0)=y_0\) and \((T(y)(t),t) \in D_{\alpha,\delta}\) for all \(t \in J\).\footnote{Basically, If a function stays inside the box, then applying $T$ keeps it inside the box.}
\end{lemma}
\begin{proof}
    Clearly \(T(y)(t_0)=y_0\). For \(t \in J\),
    \[\|T(y)(t) - y_0\| \leq \int_{t_0}^{t} \|f(y(s),s)\|ds \leq M_{\alpha,\delta}|t - t_0| \leq M_{\alpha,\delta}\epsilon \leq \alpha.\]
    Hence \((T(y)(t),t) \in D_{\alpha,\delta}\).
\end{proof}
\begin{example}[Picard operator]
    Consider the IVP $y' = y$, $y(0) = 1$. It's integral form is $y(t) = 1 + \int_0^t y(s)ds$. Start with the simplest function that satisfies the initial condition: $y_0(t) = 1$. Now iterate:
    \[y_1 = T(y_0) = 1 + \int_0^t 1ds = 1 + t,\]
    \[y_2 = T(y_1) = 1 + \int_0^t (1+s)ds = 1+t+\frac{t^2}{2},\]
    \[y_3 = T(y_2) = 1 + t + \frac{t^2}{2} + \frac{t^3}{6}.\]
    Each step adds to the next power of t. We can express this relation as
    \[y(t) = \sum_{k=0}^\infty \frac{t^k}{k!} = e^t.\]
    Indeed, $T(y)(t) = 1 + \int_0^t e^s ds = e^t$, so $T(y)=y$. Therefore, $T$ is an operator that maps a function to a new one, and only the true solution is a fixed point of $T$.
\end{example}

\subsubsection{Statement of the Theorem}
\label{EUT}
\begin{theorem}[Existence and Uniqueness]
    Let $D \subseteq \mathbb{R}^n$ be open and $(a,b)$ an open interval containing $t_0$. Suppose $f : D \times (a,b) \to \mathbb{R}^n$\,\footnote{\(D \times (a,b)\) denotes the set of all pairs \((y,t)\) with \(y \in D \subset \mathbb{R}^n\) and \(t \in (a,b)\), i.e., all allowed state-time inputs of \(f\).} is continuous and locally Lipschitz in $y$. If $y_0 \in D$, then the IVP
    \[\begin{cases}y' = f(y,t),\\y(t_0) = y_0\end{cases}\]
    has a unique solution on some open interval $J$ containing $t_0$.
\end{theorem}
\marginpar{January 20, 2026}
\begin{remark}[Choosing $\varepsilon$]
    To apply the Existence and Uniqueness Theorem, we choose \(\varepsilon\) so that the solution remains inside the domain \(D \times (a,b)\).
    Consider the IVP
    \[\begin{cases}y' = f(y,t), \\ y(t_0) = y_0.\end{cases}\]
    The theorem guarantees that there exists a unique solution 
    \[y : (t_0-\varepsilon, t_0+\varepsilon) \to \mathbb{R}^n.\]
    Choose \(\alpha > 0\) and \(\delta > 0\) such that
    \[D_{\alpha,\delta} := \{(y,t) : \|y-y_0\| \le \alpha,\ |t-t_0| \le \delta\} \subset D \times (a,b).\]
    Since \(f\) is continuous, define
    \[M_{\alpha,\delta} = \sup_{(y,t)\in D_{\alpha,\delta}} \|f(y,t)\| < \infty.\]
    Then one may take
    \[\varepsilon = \min\!\left(\delta, \frac{\alpha}{M_{\alpha,\delta}}\right).\]
\end{remark}

    Determine the largest value of $\epsilon$ for which the Theorem of Existence and Uniqueness ensures the existence of a solution $y: (-\epsilon,\epsilon) \to \mathbb{R}$.
\end{problem}

\subsubsection{Picard Iteration and Existence}
\marginpar{January 15, 2026}
Define $y_0(t) \equiv y_0$ (constant). For $k \geq 1$ set
\[y_k(t) = T(y_{k-1})(t) = y_0 + \int_{t_0}^{t} f(y_{k-1}(s), s)\, ds.\]
By the invariance lemma, each $y_k$ stays inside $D_{\alpha,\delta}$. 

The sequence $\{y_k\}$ is Cauchy in $\mathbb{R}^n$ (see Appendix~\ref{pf:picard-convergence}) and therefore converges uniformly to a function $y: J \to \mathbb{R}^n$ which satisfies the integral equation, hence solves the IVP.

\subsubsection{Uniqueness}
\begin{proof}[Proof of Uniqueness]
    Assume $y(t)$ and $z(t)$ are two solutions of the IVP with the same initial condition. By the Lipschitz property of $f$,
    \[\|y(t)-z(t)\| = \|\int_{t_0}^t f(y(s),s) - f(z(s),s) ds\| \leq \int_{t_0}^t L\|y(s) - z(s)\|ds,\]
    for some Lipschitz constant $L>0$. Define 
    \[g(t) = \int_{t_0}^{t} \|y(s)-z(s)\|ds,\]
    so $g'(t) \leq L g(t)$ and $g(t_0)=0$. Grönwall's idea\footnote{Multiplying by $e^{-L(t-t_0)}$ gives $\frac{d}{dt}(e^{-L(t-t_0)}g(t)) \le 0$, which shows $e^{-L(t-t_0)}g(t)$ is decreasing and thus $g(t) = 0$} tells us this forces $g(t) = 0$ for all $t$, so $\|y(t) - z(t)\| = 0$. Hence, $y(t) = z(t)$, meaning two solutions with the same initial condition cannot diverge, this proves the solution is unique.
\end{proof}

\section{First-Order Scalar Equation}
\marginpar{January 22, 2026}
A first-order scalar ODE involves a single unknown function $y(t)$ 
and its first derivative:
\[y' = f(y,t).\]
We study three main classes: linear, separable, and exact equations.
\subsection{First Order Linear Equations}
\begin{definition}[First Order Linear Equation]
    A first-order linear equation has the form
    \[a_0(t)y' + a_1(t)y = g(t), \quad a_0(t) \neq 0.\]
    Dividing by $a_0(t)$ gives the standard form
    \[y' + P(t)y = Q(t).\]
\end{definition}

\subsubsection{Integrating Factor Method}
To solve $y' + P(t)y = Q(t)$, multiply both sides by a function $\mu(t)$ chosen so that the left-hand side becomes an exact derivative:
\[\mu y' + \mu P(t)y = \frac{d}{dt}(\mu y).\]
This requires $\mu'(t) = \mu(t)P(t)$, which gives
\[\mu(t) = e^{\int P(t)\,dt}.\]
The equation then becomes
\[\frac{d}{dt}(\mu y) = \mu Q(t),\]
and integrating both sides yields
\[y(t) = \frac{1}{\mu(t)}\left(\int \mu(t)Q(t)\,dt + C\right), \quad C \in \mathbb{R}.\]
\begin{example}
    Solve $y' - 2y = 3e^t$ with $y(0) = 1$.\\
    Here $P(t) = -2$, $Q(t) = 3e^t$, so $\mu(t) = e^{-2t}$. 
    Multiplying through:
    \[\frac{d}{dt}(e^{-2t}y) = 3e^{-t}.\]
    Integrating:
    \[e^{-2t}y = -3e^{-t} + C \implies y(t) = -3e^t + Ce^{2t}.\]
    Applying $y(0) = 1$ gives $C = 4$, so
    \[y(t) = -3e^t + 4e^{2t}.\]
\end{example}
\begin{remark}[Quick Solution Method]
    A quick method to solve a first-order linear equation (not presented by the professor) is
    \[y(t) = \frac{\int \mu(t) Q(t)}{\mu(t)}.\]
\end{remark}

\subsubsection{Existence and Uniqueness for First Order Linear Equations}
\begin{theorem}[Existence \& Uniqueness]
    If $P$ and $Q$ are continuous on an open interval containing $t_0$, then the IVP
    \[y' + P(t)y = Q(t), \quad y(t_0) = y_0\]
    has a unique solution on that entire interval.
\end{theorem}
\begin{remark}
    Unlike the general E\&U theorem from Chapter~\eqref{EUT}, this result gives a global solution on the full interval where $P$ and $Q$ are continuous, not just locally.
\end{remark}
\begin{example}
    For the IVP
    \[\begin{cases}y' + \dfrac{1}{t-1}y = \dfrac{1}{\cos t},\\y(0) = 1,\end{cases}\]
    $P(t)$ is discontinuous at $t = 1$ and $Q(t)$ is discontinuous 
    at $t = \pm\frac{\pi}{2}$. The largest interval containing 
    $t_0 = 0$ on which both are continuous is
    \[J_{\max} = \left(-\frac{\pi}{2},\, 1\right).\]
\end{example}

\subsection{Separable Equations}
\marginpar{January 27, 2026}
\begin{definition}
    The ODE $\frac{dy}{dx} = f(x,y)$ is \emph{separable} if $f(x,y) = f_1(x)f_2(y)$ for some functions $f_1$ and $f_2$.
\end{definition}
\begin{example}
    $\dfrac{dy}{dx} = -\dfrac{x}{y}$ is separable with $f_1(x) = x$, $f_2(y) = -\dfrac{1}{y}$.\\
    $\dfrac{dy}{dx} = x^2 + y^2$ is not separable.
\end{example}

\subsubsection{General Method}
Writing the separable ODE as
\begin{equation}
    \label{SeparableODE}
    M(x) + N(y)\frac{dy}{dx} = 0,
\end{equation}
where $M(x) = -f_1(x)$ and $N(y) = \frac{1}{f_2(y)}$, and letting $H_1$, $H_2$ be antiderivatives of $M$, $N$ respectively, the chain rule gives
\[\frac{d}{dx}(H_1(x) + H_2(y(x))) = 0,\]
so the implicit general solution is
\[H_1(x) + H_2(y) = C, \quad C \in \mathbb{R},\]
or equivalently in integral form,
\[\int \frac{dy}{f_2(y)} = \int f_1(x)\,dx + C.\]
\begin{example}[Logistic Equation]
    The logistic equation
    \[\frac{dN}{dt} = rN\!\left(1 - \frac{N}{K}\right)\]
    is separable. Using partial fractions,
    \[\int\frac{dN}{N} + \int\frac{dN}{N-K} = \int r\,dt \implies \ln\left|\frac{N}{N-K}\right| = rt + C.\]
    Solving with $N(0) = N_0$ gives
    \[N(t) = \frac{KN_0}{N_0 + (K-N_0)e^{-rt}}.\]
    As $t\to\infty$, $N(t)\to K$ (the carrying capacity), provided $N_0 \neq 0$.
\end{example}

\subsection{Exact Equations}
\marginpar{January 29, 2026}
\subsubsection{Potential Functions and Integral Curves}
\begin{definition}[Potential Function]
    A function $\Psi(x,y)$ is a potential function for the equation \eqref{SeparableODE} if
    \[\frac{\partial\Psi}{\partial x} = M \quad \text{and} \quad \frac{\partial\Psi}{\partial y} = N.\]
    The general solution is then given implicitly by the level curves
    \[\Psi(x,y) = C.\]
\end{definition}
\begin{definition}[Integral Curve]
    An integral curve is the level curve of $\Psi$ passing through a specific initial condition $(x_0,y_0)$:
    \[\Gamma_{(x_0,y_0)} = \{(x,y)\mid \Psi(x,y) = \Psi(x_0,y_0)\}.\]
\end{definition}

\begin{remark}
    For a separable equation, the general solution $\Psi(x,y) = C$ describes all possible solution curves, and picking a particular $C$ gives the integral curve passing through a given point.
\end{remark}
\begin{example}
    For $\frac{dy}{dx} = -\frac{x}{y}$, $y \neq 0$, separating gives $y\,dy + x\,dx = 0$, so the potential function is
    \[\Psi(x,y) = \frac{x^2}{2} + \frac{y^2}{2}.\]
    The general solution is the family of circles $x^2+y^2 = C$. For $y(0)=2$, $\Psi(0,2)=2$, so the unique integral curve is
    \[y(x) = \sqrt{4-x^2}, \quad J_{\max} = (-2,2).\]
\end{example}

\subsubsection{Definition and Exactness Criterion}
\begin{definition}[Exact Equation]
    The equation \eqref{SeparableODE} is exact on a domain $D$ if there exists a potential function $\Psi$ such that $\Psi_x = M$ and $\Psi_y = N$. Along any solution $y(x)$,
    \[\frac{d}{dx}\Psi(x,y(x)) = 0,\]
    so $\Psi$ is constant along solutions.
\end{definition}
\begin{theorem}
    Let $D\subset\mathbb{R}^2$ be open and simply connected\footnote{Meaning there are no holes in the domain.}. Suppose $M$, $N$, $M_y$, $N_x$ are continuous on $D$. Then \eqref{SeparableODE} is exact on $D$ if and only if
    \[M_y(x,y) = N_x(x,y) \quad \forall\,(x,y)\in D.\]
\end{theorem}
\begin{proof}
    ($\Rightarrow$) Suppose it is exact, then $M = \Psi_x$ and $N = \Psi_y$, so $M_y = \Psi_{xy} = \Psi_{yx} = N_x$ by equality of mixed partials.

    ($\Leftarrow$) If $M_y = N_x$ on a simply connected domain and 
    partials are continuous, one can construct $\Psi$ with 
    $\Psi_x = M$ and $\Psi_y = N$ explicitly by integration.\footnote{The construction is shown for rectangles in the notes; the general case follows similarly.}
\end{proof}
\begin{example}
    Solve
    \[y\cos x + 2xe^y + (\sin x + x^2e^y - 1)\,\frac{dy}{dx} = 0.\]

    Identify $M = y\cos x + 2xe^y$ and $N = \sin x + x^2e^y - 1$. Check:
    \[M_y = \cos x + 2xe^y = N_x. \quad \checkmark\]
    So the equation is exact. Integrate $\Psi_x = M$ in $x$:
    \[\Psi = y\sin x + x^2e^y + c(y).\]
    Differentiate in $y$ and match with $N$:
    \[\Psi_y = \sin x + x^2e^y + c'(y) = \sin x + x^2e^y - 1 
    \implies c'(y) = -1 \implies c(y) = -y.\]
    The general solution is
    \[\Psi(x,y) = y\sin x + x^2e^y - y = C.\]
\end{example}
\begin{remark}
    Every separable equation is exact on a simply connected domain. Writing $M(x)\,dx + N(y)\,dy = 0$ gives $M_y = 0 = N_x$ automatically, so exactness always holds. The potential function is simply $\Psi(x,y) = H_1(x) + H_2(y)$.
\end{remark}

\subsubsection{Integrating Factors for Non-Exact Equations}
\marginpar{February 3, 2026}
If $M_y \neq N_x$, we seek a function $\mu(x,y)$ such that $\mu M\,dx + \mu N\,dy = 0$ is exact. The condition $(\mu M)_y = (\mu N)_x$ expands to
\[\mu_y M + \mu M_y = \mu_x N + \mu N_x.\]

\textbf{Case: $\mu = \mu(x)$ only.} Then $\mu_y = 0$ and the condition reduces to
\[\frac{d\mu}{dx} = \mu\,\frac{M_y - N_x}{N}.\]
If $\frac{M_y - N_x}{N}$ depends only on $x$, this is a separable ODE for $\mu(x)$ that can be solved directly.
\begin{example}
    Solve
    \[(3xy + y^2)\,dx + (x^2 + xy)\,dy = 0.\]

    Here $M_y = 3x+2y$ and $N_x = 2x+y$, so $M_y \neq N_x$. Compute:
    \[\frac{M_y - N_x}{N} = \frac{x+y}{x(x+y)} = \frac{1}{x},\]
    which depends only on $x$. Solving $\frac{d\mu}{dx} = \frac{\mu}{x}$ gives $\mu(x) = x$. Multiplying through by $x$:
    \[(3x^2y+xy^2)\,dx + (x^3+x^2y)\,dy = 0.\]
    Now $M_y = 3x^2+2xy = N_x$. \checkmark \; Integrating:
    \[\Psi = \int(3x^2y+xy^2)\,dx = x^3y + \frac{x^2y^2}{2} + c(y).\]
    Matching $\Psi_y = N$ gives $c'(y) = 0$, so the general solution is
    \[x^3y + \frac{x^2y^2}{2} = C.\]
\end{example}

\subsection{Applications}
\begin{example}[Falling Object with Air Resistance]
    An object of mass $m$ falls under gravity with damping 
    $\gamma > 0$. Setting $v(0) = 0$, the equation
    \[v' + \frac{\gamma}{m}v = g\]
    has integrating factor $\mu = e^{\frac{\gamma}{m}t}$, giving
    \[v(t) = \frac{gm}{\gamma}\left(1 - e^{-\frac{\gamma}{m}t}
    \right).\]
    As $t\to\infty$, the velocity approaches the terminal value 
    $v_\infty = \frac{gm}{\gamma}$.
\end{example}
\begin{example}[Mixing Problem]
    A tank holds 100 L of brine with $y_0$ g of salt. Pure salt 
    solution enters at 50 g/L at rate $R$ L/s, and the well-mixed 
    solution exits at the same rate. The salt amount $y(t)$ satisfies
    \[y' + \frac{R}{100}y = 50R, \quad y(0) = y_0.\]
    Using integrating factor $\mu = e^{\frac{R}{100}t}$,
    \[y(t) = 5000 + (y_0-5000)e^{-\frac{R}{100}t}.\]
    Regardless of $y_0$, the salt amount converges to 5000 g as 
    $t\to\infty$.
\end{example}


\section{Systems of Linear Equations}
\marginpar{February 3, 2026}

Many problems involve several unknown functions evolving together. 
Instead of a single ODE, we study systems of first-order linear ODEs, 
treating the unknowns as a vector and applying matrix algebra.

We consider systems of the form
\begin{equation}
    \label{eq:ode-system}
    y'(t) = A(t)y(t) + r(t), \quad t \in Q,
\end{equation}
where
\[y(t) = \begin{pmatrix}y_1(t)\\\vdots\\y_n(t)\end{pmatrix} \in 
\mathbb{R}^n, \quad A(t) \in M_n(\mathbb{R}), \quad
r(t) = \begin{pmatrix}r_1(t)\\\vdots\\r_n(t)\end{pmatrix}.\]

%------- 3.1 -------
\subsection{Reduction to First-Order Systems}

Any $n$th-order scalar ODE can be written as a first-order system 
by introducing new variables for each derivative.

\begin{example}
    Convert $y''' - 2y'' + 4y' - 5y = \sin t + t^3$ into a 
    first-order system. Set $y_1 = y$, $y_2 = y'$, $y_3 = y''$:
    \[\begin{pmatrix}y_1'\\y_2'\\y_3'\end{pmatrix} = 
    \begin{pmatrix}0&1&0\\0&0&1\\5&-4&2\end{pmatrix}
    \begin{pmatrix}y_1\\y_2\\y_3\end{pmatrix} + 
    \begin{pmatrix}0\\0\\\sin t + t^3\end{pmatrix},\]
    which is in the form $y' = Ay + r$ with $J_{\max} = \mathbb{R}$.
\end{example}

%------- 3.2 -------
\subsection{Existence, Uniqueness, and Solution Structure}

\begin{theorem}[Existence and Uniqueness]
    If every entry of $A(t)$ and $r(t)$ is continuous on $Q$, then 
    for any $t_0 \in Q$ and $y_0 \in \mathbb{R}^n$, the IVP
    \[\begin{cases}y'(t) = A(t)y(t) + r(t),\\ y(t_0) = y_0\end{cases}\]
    has a unique solution on all of $Q$. Hence $J_{\max} = Q$.
\end{theorem}

\begin{definition}
    If $r(t) = 0$, the system is \emph{homogeneous}; otherwise it 
    is \emph{non-homogeneous}.
\end{definition}

We now focus on the homogeneous system
\begin{equation}
    \label{eq:ode-system-matrix}
    y'(t) = A(t)y(t).
\end{equation}

\begin{definition}[Principle of Superposition]
    \label{POS}
    If $y_1(t), \dots, y_n(t)$ are solutions of 
    \eqref{eq:ode-system-matrix}, then any linear combination 
    is also a solution.
\end{definition}

\begin{definition}[Linear Independence]
    Solutions $y_1, \dots, y_n$ of \eqref{eq:ode-system-matrix} 
    are linearly independent if
    \[c_1 y_1(t) + \cdots + c_n y_n(t) = 0 \;\; \forall\, t \in Q 
    \implies c_1 = \cdots = c_n = 0.\]
\end{definition}

\begin{theorem}[Solution Space]
    The solution set $S$ of \eqref{eq:ode-system-matrix} is an 
    $n$-dimensional vector subspace of $C^1(Q)$.
\end{theorem}

\begin{proof}
    \textbf{Step 1: Vector space.} If $y_1, y_2 \in S$ and 
    $\alpha_1, \alpha_2 \in \mathbb{R}$,
    \[(\alpha_1 y_1 + \alpha_2 y_2)' = \alpha_1 y_1' + 
    \alpha_2 y_2' = A(t)(\alpha_1 y_1 + \alpha_2 y_2),\]
    so any linear combination is in $S$, and the zero function 
    is in $S$.

    \textbf{Step 2: Basis of dimension $n$.} Let $e_1,\dots,e_n$ 
    be the standard basis of $\mathbb{R}^n$, and let $y_i$ be the 
    unique solution of
    \[\begin{cases}y' = A(t)y,\\ y(t_0) = e_i\end{cases}\]
    for $i = 1,\dots,n$. By uniqueness, every $z \in S$ is a linear 
    combination of the $y_i$, so $\{y_1,\dots,y_n\}$ spans $S$. 
    Evaluating at $t_0$ shows they are independent. Hence 
    $\dim S = n$.
\end{proof}

\subsubsection{Matrix Solutions and the Wronskian}

\begin{definition}[Matrix Solution]
    A matrix solution of \eqref{eq:ode-system-matrix} is an 
    $n \times n$ matrix whose columns are solutions of 
    \eqref{eq:ode-system-matrix}.
\end{definition}

\begin{proposition}
    Let $y(t)$ be a matrix solution of \eqref{eq:ode-system-matrix}. 
    Either $\det(y(t)) \neq 0$ for all $t \in Q$, or 
    $\det(y(t)) = 0$ for all $t \in Q$.
\end{proposition}

\begin{proof}
    Suppose $\det(y(t_0)) = 0$ for some $t_0$. Then there exists a 
    nonzero $v \in \mathbb{R}^n$ such that $y(t_0)v = 0$. Define 
    $y_1(t) = y(t)v$; since each column of $y(t)$ is a solution, 
    $y_1 \in S$. Also define $y_2(t) = 0 \in S$. Since 
    $y_1(t_0) = 0 = y_2(t_0)$, uniqueness gives $y_1(t) = y_2(t)$ 
    for all $t$, so $y(t)v = 0$ for all $t$. Hence the columns of 
    $y(t)$ are linearly dependent and $\det(y(t)) = 0$ for all 
    $t \in Q$.
\end{proof}

\begin{definition}[Wronskian]
    The Wronskian of $y_1,\ldots,y_n$ is
    \[W(t) = W(y_1,\ldots,y_n)(t) = \det(y(t)).\]
\end{definition}

\begin{example}
    Let $y_1(t) = \begin{pmatrix}\cos 2t\\-\sin 2t\end{pmatrix}$, 
    $y_2(t) = \begin{pmatrix}-2\sin 2t\\-2\cos 2t\end{pmatrix}$.
    \[W(t) = \det\begin{pmatrix}\cos 2t & -2\sin 2t\\
    -\sin 2t & -2\cos 2t\end{pmatrix} = -2 \neq 0,\]
    so $y_1$ and $y_2$ are linearly independent.
\end{example}

\begin{definition}[Fundamental Matrix Solution]
    A matrix solution with nonvanishing Wronskian is called a 
    \emph{fundamental matrix solution}.
\end{definition}

\begin{theorem}[Liouville's Formula]
    Let $y(t)$ be a matrix solution of \eqref{eq:ode-system-matrix} 
    with $y(t_0) = y_0$. Then
    \[\det(y(t)) = \det(y_0)\,e^{\int_{t_0}^t \operatorname{tr}
    (A(s))\,ds},\]
    where $\operatorname{tr}(M)$ denotes the sum of the diagonal 
    entries of $M$.
\end{theorem}

\subsubsection{Recovering the Coefficient Matrix and Solving IVPs}

\begin{proposition}[Recovering $A$]
    \label{prop:recover-A}
    Let $\Phi(t)$ be a fundamental matrix solution of 
    \eqref{eq:ode-system-matrix}. Then $\Phi(t)$ is invertible for 
    all $t \in Q$ and
    \[A(t) = \Phi'(t)\,\Phi(t)^{-1}.\]
\end{proposition}

\begin{proof}
    Since $\Phi'(t) = A(t)\Phi(t)$ and $\det\Phi(t) \neq 0$, 
    multiplying on the right by $\Phi(t)^{-1}$ gives the result.
\end{proof}

\begin{example}
    Given $\Phi(t) = \begin{pmatrix}e^{2t}&0\\0&e^{-t}\end{pmatrix}$,
    \[\Phi'(t) = \begin{pmatrix}2e^{2t}&0\\0&-e^{-t}\end{pmatrix}, 
    \quad \Phi(t)^{-1} = \begin{pmatrix}e^{-2t}&0\\0&e^t
    \end{pmatrix},\]
    so $A = \Phi'(t)\Phi(t)^{-1} = \begin{pmatrix}2&0\\0&-1
    \end{pmatrix}$.
\end{example}

Since $y_1,\ldots,y_n$ are linearly independent, the general 
solution is
\[y(t) = c_1 y_1(t) + \cdots + c_n y_n(t).\]
Given $y(t_0) = y_0$, the constants $c = (c_1,\ldots,c_n)^T$ 
are uniquely determined by
\[\begin{pmatrix}|&&|\\ y_1(t_0)&\cdots&y_n(t_0)\\|&&|
\end{pmatrix}c = y_0 \implies c = \Phi(t_0)^{-1} y_0.\]

%------- 3.3 -------
\subsection{Homogeneous Systems with Constant Coefficients}
\label{sec:const-coeff}

Consider \eqref{eq:ode-system-matrix} with $A \in M_n(\mathbb{R})$ 
constant. We look for solutions of the form
\begin{equation}
    \label{eq:exp-ansatz}
    y(t) = e^{\lambda t}u,
\end{equation}
where $u$ is a constant vector. Substituting into 
\eqref{eq:ode-system-matrix},
\[e^{\lambda t}(\lambda u) = e^{\lambda t}(Au) \implies Au = 
\lambda u.\]
So $\lambda$ must be an eigenvalue of $A$ and $u$ a corresponding 
eigenvector.

\begin{definition}[Kernel]
    $\ker A = \{u \mid Au = 0\}$ is the kernel (null space) of $A$.
\end{definition}

\begin{definition}[Characteristic Polynomial, Eigenvalue, Spectrum]
    The \emph{characteristic polynomial} of $A \in M_n(\mathbb{R})$ 
    is $p(\lambda) = \det(A - \lambda I)$. An \emph{eigenvalue} is a 
    root of $p(\lambda)$, and the \emph{spectrum} is
    \[\sigma(A) = \{\lambda \mid p(\lambda) = 0\}.\]
\end{definition}

\subsubsection{Diagonalizable Case: Real Eigenvalues}

\begin{definition}[Diagonalizable Matrix]
    $A \in M_n(\mathbb{R})$ is \emph{diagonalizable} if there exists 
    an invertible $P$ such that $P^{-1}AP$ is diagonal.
\end{definition}

\begin{theorem}[Diagonalization]
    $A$ is diagonalizable if and only if it has a basis of 
    eigenvectors. In particular, if all eigenvalues of $A$ are real 
    and distinct, then $A$ is diagonalizable.
\end{theorem}

\begin{proof}
    Let $\lambda_1,\ldots,\lambda_n$ be real distinct eigenvalues 
    with eigenvectors $u_1,\ldots,u_n$. Distinct eigenvalues imply 
    the eigenvectors are linearly independent, so they form a basis 
    of $\mathbb{R}^n$. Let $P = [u_1 \cdots u_n]$. Then
    \[AP = PD, \quad D = \operatorname{diag}(\lambda_1,\ldots,
    \lambda_n),\]
    so $P^{-1}AP = D$.
\end{proof}

\begin{theorem}[General Solution via Diagonalization]
    If $A$ is diagonalizable with eigenpairs 
    $(\lambda_i, u_i)_{i=1}^n$, the general solution of 
    \eqref{eq:ode-system-matrix} is
    \[y(t) = \sum_{i=1}^n c_i e^{\lambda_i t} u_i.\]
\end{theorem}

\begin{example}[$2\times 2$ system]
    Let $y' = Ay$ with $A = \begin{pmatrix}1&3\\3&1\end{pmatrix}$.

    \textbf{Eigenvalues:} $p(\lambda) = (1-\lambda)^2 - 9 = 0$ gives 
    $\lambda_1 = -2$, $\lambda_2 = 4$, so $\sigma(A) = \{-2, 4\}$.

    \textbf{Eigenvectors:} For $\lambda_1 = -2$: $(A+2I)u = 0$ gives 
    $u_1 = \begin{pmatrix}1\\-1\end{pmatrix}$. For $\lambda_2 = 4$: 
    $u_2 = \begin{pmatrix}1\\1\end{pmatrix}$.

    \textbf{Wronskian:} $W(0) = \begin{vmatrix}1&1\\-1&1
    \end{vmatrix} = 2 \neq 0$, so the solutions are independent.

    \textbf{General solution:}
    \[y(t) = c_1 e^{-2t}\begin{pmatrix}1\\-1\end{pmatrix} + 
    c_2 e^{4t}\begin{pmatrix}1\\1\end{pmatrix}.\]

    \textbf{With} $y(0) = \begin{pmatrix}1\\2\end{pmatrix}$:
    \[\begin{pmatrix}c_1\\c_2\end{pmatrix} = 
    \begin{pmatrix}1&1\\-1&1\end{pmatrix}^{-1}
    \begin{pmatrix}1\\2\end{pmatrix} = 
    \begin{pmatrix}1/2 & -1/2\\1/2&1/2\end{pmatrix}
    \begin{pmatrix}1\\2\end{pmatrix} = 
    \begin{pmatrix}-1/2\\3/2\end{pmatrix}.\]
\end{example}

\begin{example}[$3\times 3$ system]
    \marginpar{February 10, 2026}
    Let $y' = Ay$ with 
    $A = \begin{pmatrix}3&1&-2\\-1&5&-2\\-3&3&-2\end{pmatrix}$.

    \textbf{Eigenvalues:} $p(\lambda) = -(\lambda+2)(\lambda-4)^2$, 
    so $\lambda_1 = -2$ and $\lambda_2 = \lambda_3 = 4$.

    \textbf{Eigenvectors:}
    For $\lambda_1 = -2$: $(A+2I)u = 0$ gives 
    $u_1 = \begin{pmatrix}1\\1\\3\end{pmatrix}$.
    For $\lambda = 4$: row reduction of $(A-4I)u = 0$ yields two 
    free variables ($b$ and $c$ free, $a = b-2c$), giving
    \[u_2 = \begin{pmatrix}1\\1\\0\end{pmatrix}, \quad 
    u_3 = \begin{pmatrix}-2\\0\\1\end{pmatrix}.\]
    Since $\dim\ker(A-4I) = 2 =$ algebraic multiplicity, $A$ is 
    diagonalizable.

    \textbf{Wronskian:} $W(0) = \det\begin{pmatrix}1&1&-2\\1&1&0\\
    3&0&1\end{pmatrix} = 6 \neq 0$.

    \textbf{General solution:}
    \[y(t) = c_1 e^{-2t}\begin{pmatrix}1\\1\\3\end{pmatrix} + 
    c_2 e^{4t}\begin{pmatrix}1\\1\\0\end{pmatrix} + 
    c_3 e^{4t}\begin{pmatrix}-2\\0\\1\end{pmatrix}.\]
\end{example}

\begin{example}[Non-diagonalizable]
    Let $A = \begin{pmatrix}1&1\\0&1\end{pmatrix}$. Since $A$ is 
    upper triangular, $\lambda_1 = \lambda_2 = 1$ (multiplicity 2). 
    Solving $(A-I)u = 0$ gives only one independent eigenvector 
    $u = \begin{pmatrix}1\\0\end{pmatrix}$, so $A$ is 
    \emph{not} diagonalizable.
\end{example}

\subsubsection{Diagonalizable Case: Complex Eigenvalues}

If $A \in M_n(\mathbb{R})$ has a complex eigenpair 
$(\lambda, u)$ with $\lambda = \alpha + i\beta$ and 
$u = p + iq$, then a complex solution is
\begin{equation}
    \label{eq:exp-ansatz-complex}
    y(t) = e^{(\alpha+i\beta)t}(p+iq).
\end{equation}

\begin{definition}[Euler's Formula]
    \begin{equation}
        \label{eq:EulerFormula}
        e^{ix} = \cos x + i\sin x, \quad x \in \mathbb{R}.
    \end{equation}
\end{definition}

Expanding \eqref{eq:exp-ansatz-complex} using Euler's formula and 
separating real and imaginary parts yields two real solutions:
\begin{equation}
    \label{eq:exp-ansatz-complex-sol}
    y_1(t) = e^{\alpha t}(p\cos\beta t - q\sin\beta t), \qquad 
    y_2(t) = e^{\alpha t}(p\sin\beta t + q\cos\beta t).
\end{equation}

\begin{proposition}[Complex Conjugate Eigenpairs]
    \label{prop:CCE}
    Let $A \in M_n(\mathbb{R})$. If $(\lambda, u)$ is an eigenpair 
    with $\lambda \in \mathbb{C}$, then $(\bar\lambda, \bar u)$ is 
    also an eigenpair.
\end{proposition}

\begin{proof}
    From $Au = \lambda u$, taking conjugates and using 
    $\bar A = A$ gives $A\bar u = \bar\lambda\,\bar u$.
\end{proof}

\begin{proposition}
    \marginpar{February 12, 2026}
    If $\lambda = \alpha + i\beta \in \sigma(A)$ with $\beta \neq 0$ 
    and eigenvector $u = p + iq$, then the functions 
    \eqref{eq:exp-ansatz-complex-sol} are real-valued linearly 
    independent solutions of \eqref{eq:ode-system-matrix}.
\end{proposition}

\begin{example}
    Let $A = \begin{pmatrix}2&-1\\1&2\end{pmatrix}$.

    \textbf{Eigenvalues:} $p(\lambda) = (\lambda-2)^2 + 1 = 0$ 
    gives $\lambda_1 = 2+i$, $\lambda_2 = 2-i$.

    \textbf{Eigenvector for $\lambda_1 = 2+i$:} 
    $(A-(2+i)I)u = 0$ gives $b = ia$, so taking $a=1$:
    \[u_1 = \begin{pmatrix}1\\i\end{pmatrix} = 
    \underbrace{\begin{pmatrix}1\\0\end{pmatrix}}_{p} + 
    i\underbrace{\begin{pmatrix}0\\1\end{pmatrix}}_{q}.\]
    By Proposition~\ref{prop:CCE}, $u_2 = \bar u_1 = 
    \begin{pmatrix}1\\-i\end{pmatrix}$.
\end{example}

        \item Find the eigenvalues and eigenvectors.
        \item Find the real-valued solution to the IVP with

    \end{enumerate}
\end{problem}

\subsubsection{Nondiagonalizable Matrices and Generalized Eigenvectors}

\begin{remark}
    If $A$ does not admit a basis of eigenvectors, there are not 
    enough solutions of the form $e^{\lambda t}u$ to span the 
    solution space.
\end{remark}

\begin{definition}[Algebraic and Geometric Multiplicity]
    Let $\lambda_0 \in \sigma(A)$.
    \begin{enumerate}
        \item The \emph{algebraic multiplicity} $m$ is the largest 
        integer such that $(\lambda - \lambda_0)^m$ divides 
        $p(\lambda)$.
        \item The \emph{geometric multiplicity} is 
        $\dim\ker(A - \lambda_0 I)$ (the number of independent 
        eigenvectors for $\lambda_0$).
    \end{enumerate}
    The \emph{defect} is $d_{\lambda_0} = m - \dim\ker(A-\lambda_0 I) 
    \geq 0$. We call $\lambda_0$ \emph{defective} if 
    $d_{\lambda_0} > 0$.
\end{definition}

\begin{example}
    Let $A = \begin{pmatrix}1&1\\0&1\end{pmatrix}$, 
    $p(\lambda) = (\lambda-1)^2$. Algebraic multiplicity $m=2$; 
    $\ker(A-I) = \operatorname{span}\!\left\{
    \begin{pmatrix}1\\0\end{pmatrix}\right\}$ has dimension 1. 
    Defect $= 1 > 0$, so $\lambda_0 = 1$ is defective.
\end{example}

\begin{example}
    Let $A = \begin{pmatrix}0&1&0&1\\0&-2&-1&-1\\1&1&0&0\\
    0&1&0&1\end{pmatrix}$, $p(\lambda) = \lambda^3(\lambda+1)$. 
    For $\lambda_0 = 0$: algebraic multiplicity $m=3$, 
    $\ker A = \operatorname{span}\!\left\{
    \begin{pmatrix}1\\-1\\1\\1\end{pmatrix}\right\}$ has dimension 1. 
    Defect $= 2$.
\end{example}

\begin{example}
    For $A = \begin{pmatrix}3&1&-2\\-1&5&-2\\-3&3&-2\end{pmatrix}$, 
    $p(\lambda) = (\lambda+2)(\lambda-4)^2$. For $\lambda_0 = 4$: 
    $m = 2$, $\dim\ker(A-4I) = 2$. Defect $= 0$, so 
    $\lambda_0 = 4$ is \emph{not} defective.
\end{example}

\begin{definition}[Generalized Eigenvector]
    Let $\lambda \in \sigma(A)$ with algebraic multiplicity $m$. 
    A \emph{generalized eigenvector} satisfies 
    $(A - \lambda I)^m u = 0$.
\end{definition}

\begin{theorem}[Generalized Eigenvector Basis]
    If $\lambda \in \sigma(A)$ has algebraic multiplicity $m$, then
    $\dim\ker(A-\lambda I)^m = m$. Consequently, every 
    $A \in M_n(\mathbb{R})$ has a basis of generalized eigenvectors.
\end{theorem}

Denote $k_j = \ker(A-\lambda I)^j$. Then
\[k_0 = \{0\} \subsetneq k_1 \subsetneq k_2 \subsetneq 
\cdots \subsetneq k_m.\]

\begin{lemma}
    $k_j \subseteq k_{j+1}$ for all $j$. Moreover, if 
    $k_j = k_{j+1}$ for some $j$, then 
    $k_j = k_{j+1} = \cdots = k_m$.
\end{lemma}

\begin{proof}
    The inclusion is immediate. For the second part: assume 
    $k_j = k_{j+1}$ and let $w \in k_{j+2}$. Setting 
    $u = (A-\lambda I)w \in k_{j+1} = k_j$ gives 
    $(A-\lambda I)^{j+1}u = (A-\lambda I)^{j+2}w = 0$, hence 
    $(A-\lambda I)^j u = 0$, so $(A-\lambda I)^{j+1}w = 0$, meaning 
    $w \in k_{j+1}$. Iterate to get the result.
\end{proof}

\begin{algorithm}[Computing Generalized Eigenvectors]
    Given a basis of $k_i$ with $\dim(k_i) < m$, find 
    $u \in k_{i+1} \setminus k_i$ by solving
    $(A - \lambda I)u = w$ for each basis vector $w \in k_i$.
\end{algorithm}

\begin{example}
    For $A = \begin{pmatrix}1&1\\0&1\end{pmatrix}$: 
    $k_1 = \operatorname{span}\!\left\{
    \begin{pmatrix}1\\0\end{pmatrix}\right\}$. To find 
    $u \in k_2 \setminus k_1$, solve
    \[\begin{pmatrix}0&1\\0&0\end{pmatrix}
    \begin{pmatrix}a\\b\end{pmatrix} = 
    \begin{pmatrix}1\\0\end{pmatrix}.\]
    This gives $b=1$, $a$ free. Pick $a=0$: 
    $u_2 = \begin{pmatrix}0\\1\end{pmatrix}$.
    Check: $(A-I)^2 u_2 = (A-I)\begin{pmatrix}1\\0\end{pmatrix} = 0$. 
    \checkmark
\end{example}

%------- 3.4 -------
\subsection{The Matrix Exponential}
\marginpar{February 17, 2026}

\begin{definition}[Matrix Exponential]
    For $A \in M_n(\mathbb{R})$,
    \[e^A = \sum_{k=0}^{\infty} \frac{A^k}{k!} = I + A + 
    \frac{A^2}{2!} + \cdots\]
\end{definition}

\begin{definition}[Induced Matrix Norm]
    \[\|A\| = \sup_{x \neq 0}\frac{\|Ax\|}{\|x\|} = 
    \sup_{\|x\|=1}\|Ax\|.\]
\end{definition}

\begin{proposition}[Norm Properties]
    For all $A, B \in M_n(\mathbb{R})$ and $x \in \mathbb{R}^n$:
    \[\|Ax\| \leq \|A\|\cdot\|x\|, \qquad 
    \|AB\| \leq \|A\|\cdot\|B\|.\]
\end{proposition}

\begin{proof}
    For any nonzero $x$: $\|Ax\| = \|A(x/\|x\|)\|\cdot\|x\| 
    \leq \|A\|\cdot\|x\|$. For submultiplicativity, let $\|x\|=1$:
    $\|ABx\| \leq \|A\|\cdot\|Bx\| \leq \|A\|\cdot\|B\|$. Take the 
    supremum over unit vectors.
\end{proof}

\begin{proposition}[Convergence]
    The series defining $e^A$ converges absolutely for every 
    $A \in M_n(\mathbb{R})$.
\end{proposition}

\begin{proof}
    By the triangle inequality, 
    $\|\sum_{k=0}^N A^k/k!\| \leq \sum_{k=0}^N \|A\|^k/k!$, 
    which converges to $e^{\|A\|}$.
\end{proof}

\begin{proposition}[Basic Properties]
    For all $A \in M_n(\mathbb{R})$ and $t, s \in \mathbb{R}$:
    \[e^{A\cdot 0} = I, \quad e^{A(t+s)} = e^{At}e^{As}, \quad 
    e^{-At} = (e^{At})^{-1}, \quad \frac{d}{dt}e^{At} = Ae^{At} 
    = e^{At}A.\]
    If $AB = BA$, then $e^{A+B} = e^A e^B$.
\end{proposition}

\begin{proof}[Proof of the derivative formula]
    \[\frac{d}{dt}e^{At} = \frac{d}{dt}\sum_{k=0}^\infty 
    \frac{A^k t^k}{k!} = \sum_{k=1}^\infty \frac{A^k t^{k-1}}
    {(k-1)!} = A\sum_{k=0}^\infty \frac{A^k t^k}{k!} = Ae^{At}.\]
\end{proof}

\begin{theorem}[Fundamental Matrix Solution]
    $e^{At}$ is a fundamental matrix solution of 
    \eqref{eq:ode-system-matrix}. The general solution is 
    $y(t) = e^{At}C$, and the unique solution to the IVP 
    $y(0) = y_0$ is $y(t) = e^{At}y_0$.
\end{theorem}

\begin{proof}
    $\frac{d}{dt}e^{At} = Ae^{At}$ shows each column satisfies 
    the system. $e^{A\cdot 0} = I$ implies $\det(e^{A\cdot 0}) = 
    1 \neq 0$, so $e^{At}$ is invertible.
\end{proof}

\subsubsection{Computing $e^{At}$ from a Fundamental Matrix}
\marginpar{February 19, 2026}

Let $Y(t)$ be any fundamental matrix solution of $y' = Ay$. Since 
$e^{At}$ is also a fundamental matrix, there exists invertible 
$C$ such that $e^{At} = Y(t)C$. Evaluating at $t=0$:
\[I = Y(0)C \implies C = Y(0)^{-1}, \quad \therefore \quad 
e^{At} = Y(t)\,Y(0)^{-1}.\]

\begin{example}
    For $A = \begin{pmatrix}1&3\\3&1\end{pmatrix}$ with 
    $\lambda_1=-2$, $u_1=\begin{pmatrix}-1\\1\end{pmatrix}$ and 
    $\lambda_2=4$, $u_2=\begin{pmatrix}1\\1\end{pmatrix}$:
    \[Y(t) = \begin{pmatrix}-e^{-2t}&e^{4t}\\e^{-2t}&e^{4t}
    \end{pmatrix}, \quad Y(0)^{-1} = \frac{1}{2}
    \begin{pmatrix}-1&1\\1&1\end{pmatrix},\]
    \[e^{At} = Y(t)Y(0)^{-1} = \begin{pmatrix}
    \frac{1}{2}(e^{-2t}+e^{4t}) & \frac{1}{2}(-e^{-2t}+e^{4t})\\
    \frac{1}{2}(-e^{-2t}+e^{4t}) & \frac{1}{2}(e^{-2t}+e^{4t})
    \end{pmatrix}.\]
    Verify: $e^{A\cdot 0} = I$. \checkmark
\end{example}

\subsubsection{Action of $e^{At}$ on Eigenvectors and Generalized 
Eigenvectors}

\textbf{Eigenvectors.} If $Au = \lambda u$, then
\[e^{At}u = \sum_{k=0}^\infty \frac{A^k t^k}{k!}u = 
\sum_{k=0}^\infty \frac{\lambda^k t^k}{k!}u = e^{\lambda t}u.\]

\textbf{Generalized eigenvectors.} If $(A-\lambda I)^{j+1}u=0$, 
write $A = (A-\lambda I) + \lambda I$. Since these terms commute,
\[e^{At}u = e^{\lambda t}e^{(A-\lambda I)t}u = 
e^{\lambda t}\sum_{k=0}^{j}\frac{(A-\lambda I)^k t^k}{k!}u,\]
a finite sum since $(A-\lambda I)^k u = 0$ for $k > j$.

\begin{proposition}
    The general solution of \eqref{eq:ode-system-matrix} is
    \[y(t) = \sum_{i=1}^n c_i\, e^{At}u_i, \quad c_i \in \mathbb{R},\]
    where $u_1,\ldots,u_n$ are $n$ linearly independent generalized 
    eigenvectors of $A$.
\end{proposition}

\begin{example}
    For $A = \begin{pmatrix}1&1\\0&1\end{pmatrix}$, $\sigma(A) = 
    \{1\}$. From the generalized eigenvector computation above, 
    $u_1 = \begin{pmatrix}1\\0\end{pmatrix}$ (eigenvector) and 
    $u_2 = \begin{pmatrix}0\\1\end{pmatrix}$ (generalized). Then
    \[y_1(t) = e^t u_1, \qquad y_2(t) = e^{At}u_2 = 
    e^t(u_2 + tu_1) = e^t\begin{pmatrix}t\\1\end{pmatrix},\]
    giving the general solution
    \[y(t) = c_1 e^t\begin{pmatrix}1\\0\end{pmatrix} + 
    c_2 e^t\begin{pmatrix}t\\1\end{pmatrix}.\]
\end{example}

\begin{example}
    For $A = \begin{pmatrix}2&4\\-1&-2\end{pmatrix}$: 
    $p(\lambda) = \lambda^2$, so $\sigma(A) = \{0\}$ with 
    multiplicity 2. 
    $k_1 = \ker A = \left\langle\begin{pmatrix}-2\\1
    \end{pmatrix}\right\rangle$.
    Solve $Au_2 = u_1$ for $u_2 \in k_2\setminus k_1$: choosing 
    $b=0$ gives $u_2 = \begin{pmatrix}-1\\0\end{pmatrix}$. Then
    \[y_1(t) = \begin{pmatrix}-2\\1\end{pmatrix}, \qquad 
    y_2(t) = u_2 + tu_1 = 
    \begin{pmatrix}-1-2t\\t\end{pmatrix},\]
    \[y(t) = c_1\begin{pmatrix}-2\\1\end{pmatrix} + 
    c_2\begin{pmatrix}-1-2t\\t\end{pmatrix}, \quad W(t) = 
    \det\begin{pmatrix}-2&-1-2t\\1&t\end{pmatrix} = 1 \neq 0.\]
\end{example}

\subsection{Non-Homogeneous Systems}

Consider $y'(t) = Ay(t) + r(t)$ with $r: I \to \mathbb{R}^n$ 
continuous. If $y_1(t)$ and $y_2(t)$ are two solutions, then 
$y_2 - y_1$ solves the homogeneous system, so there exists 
$C \in \mathbb{R}^n$ with $y_2(t) - y_1(t) = Y(t)C$.

\begin{theorem}[General Solution]
    Let $y_p(t)$ be any particular solution of 
    \eqref{eq:non-homogeneous-ode-system-matrix}. Then the general 
    solution is
    \[y(t) = Y(t)C + y_p(t), \quad C \in \mathbb{R}^n.\]
\end{theorem}

\subsubsection{Variation of Parameters}

Look for a particular solution $y_p(t) = Y(t)C_p(t)$, where 
$C_p(t)$ is unknown. Differentiating and substituting into 
\eqref{eq:non-homogeneous-ode-system-matrix}:
\[AY(t)C_p + Y(t)C_p' = AY(t)C_p + r(t) \implies 
Y(t)C_p'(t) = r(t).\]
Since $Y(t)$ is invertible, $C_p'(t) = Y^{-1}(t)r(t)$, giving
\[y_p(t) = Y(t)\int Y^{-1}(t)\,r(t)\,dt.\]
The general solution is therefore
\[y(t) = Y(t)\!\left[C + \int Y^{-1}(t)\,r(t)\,dt\right], 
\quad C \in \mathbb{R}^n.\]

\subsubsection{Mass-Spring Systems}
\marginpar{February 24, 2026}

\begin{center}
    \begin{figure}[H]
        \centering
        \includegraphics[width=0.5\textwidth]{../Images/MSS-setup.jpeg}
    \end{figure}
\end{center}

At equilibrium, $mg + kL = 0$.

\begin{definition}[Hooke's Law]
    The restoring spring force is $F_s = -kx$, where $k > 0$ is 
    the spring constant.
\end{definition}

Let $u(t)$ be the displacement from equilibrium ($u > 0$ 
downward), with damping $F_d = -\gamma u'$ and external force 
$F(t)$. Newton's second law and the equilibrium relation give
\begin{equation}
    \label{eq:mass-spring-system}
    mu'' + \gamma u' + ku = F(t).
\end{equation}
As a first-order system with 
$y = \begin{pmatrix}u\\u'\end{pmatrix}$:
\[y' = \begin{pmatrix}0&1\\-k/m&-\gamma/m\end{pmatrix}y + 
\begin{pmatrix}0\\F(t)/m\end{pmatrix}.\]
The characteristic polynomial of the homogeneous part is
\[p(\lambda) = \lambda^2 + \frac{\gamma}{m}\lambda + \frac{k}{m}, 
\qquad \lambda_\pm = \frac{-\gamma \pm \sqrt{\gamma^2-4km}}{2m}.\]
Setting $\Delta = \gamma^2 - 4km$:
\begin{itemize}
    \item \textbf{Overdamped} ($\Delta > 0$): two distinct real 
    negative eigenvalues,
    \[u(t) = c_1 e^{\lambda_1 t} + c_2 e^{\lambda_2 t} 
    \to 0 \text{ without oscillation.}\]
    \item \textbf{Critically damped} ($\Delta = 0$): repeated 
    eigenvalue $\lambda = -\gamma/(2m)$,
    \[u(t) = (c_1 + c_2 t)e^{\lambda t} \to 0 \text{ as fast 
    as possible without oscillating.}\]
    \item \textbf{Underdamped} ($\Delta < 0$): complex eigenvalues 
    $\lambda = -\frac{\gamma}{2m} \pm i\omega$ with 
    $\omega = \frac{\sqrt{4km-\gamma^2}}{2m}$,
    \[u(t) = e^{-\frac{\gamma}{2m}t}(c_1\cos\omega t + 
    c_2\sin\omega t), \text{ exponentially decaying oscillations.}\]
\end{itemize}

\section{Stability, Phase Portraits and Orbits}
\marginpar{March 10, 2026}

Let $y'(t) = f(y(t))$, $f$ locally Lipschitz, and $y_0 \in \mathbb{R}^n$. Denote by
\[
\Gamma_{y_0} = \{y(t) \mid y'(t) = f(y(t)),\ y(0) = y_0,\ t \in J_{\max}\},
\]
the \emph{orbit} of $y_0$, where $J_{\max}$ is the maximal interval of existence. The orbit is a curve in $\mathbb{R}^n$.

%------- 5.1 -------
\subsection{Orbits and Phase Portraits}

\begin{definition}[Phase Portrait]
    The phase portrait is the collection of all orbits of the system $y' = f(y)$.
\end{definition}

\begin{example}
    Let $y' = y(1-y)$. Since this is a separable ODE, the solution is
    \[y(t) = \frac{y_0}{y_0 + (1-y_0)e^{-t}}.\]
    The orbits are:
    \begin{itemize}
        \item $\Gamma_0 = \{0\}$,
        \item $\Gamma_1 = \{1\}$,
        \item $\Gamma_{y_0} = (0,1)$ for $y_0 \in (0,1)$,
        \item $\Gamma_{y_0} = (1,+\infty)$ for $y_0 > 1$,
        \item $\Gamma_{y_0} = (-\infty,0)$ for $y_0 < 0$.
    \end{itemize}
    The values $0$ and $1$ are the equilibrium solutions.
\end{example}

%------- 5.2 -------
\subsection{Equilibrium Solutions}

\begin{definition}[Equilibrium Solution]
    A vector $\tilde{y} \in \mathbb{R}^n$ is called an \emph{equilibrium solution} (or steady state) of $y' = f(y)$ if $f(\tilde{y}) = 0$. Since $\tilde{y}' = 0 = f(\tilde{y})$, the solution satisfying $y(0) = \tilde{y}$ is constant: $\tilde{y}(t) = \tilde{y}$ for all $t \in \mathbb{R}$, and its orbit is the single point $\Gamma_{\tilde{y}} = \{\tilde{y}\}$.
\end{definition}

\begin{example}
    For the linear system $y' = Ay$, equilibrium solutions satisfy $Au = 0$, i.e.\ they lie in $\ker(A)$.
    \begin{itemize}
        \item If $A$ is invertible, then $\ker(A) = \{0\}$, so $\tilde{y} = 0$ is the only equilibrium.
        \item If $A$ is not invertible, $\ker(A)$ contains more equilibria.
    \end{itemize}
\end{example}

\begin{example}
    Let $A = \begin{pmatrix}1&-1\\2&-2\end{pmatrix}$. Solving $Au = 0$ gives $a = b$, so
    \[\ker(A) = \left\langle\begin{pmatrix}1\\1\end{pmatrix}\right\rangle = \left\{c\begin{pmatrix}1\\1\end{pmatrix} \;\middle|\; c\in\mathbb{R}\right\}.\]
    Every point on this line is an equilibrium solution.
\end{example}

\begin{example}[Nonlinear Pendulum]
    From Newton's second law,
    \[\theta'' + \gamma\theta' + \frac{g}{l}\sin(\theta) = 0.\]
    Setting $y_1 = \theta$, $y_2 = \theta'$, the system becomes
    \[\begin{pmatrix}y_1'\\y_2'\end{pmatrix} = \begin{pmatrix}y_2\\-\tfrac{g}{l}\sin(y_1) - \gamma y_2\end{pmatrix} =: f(y).\]
    Equilibria satisfy $f(y) = 0$: we need $y_2 = 0$ and $\sin(y_1) = 0$, giving
    \[y_1 = n\pi,\quad n \in \mathbb{Z}.\]
\end{example}

%------- 5.3 -------
\subsection{Stability}

\subsubsection{Definitions}

\begin{definition}[Stable Equilibrium]
    An equilibrium $\tilde{y}$ of $y' = f(y)$ is \emph{stable} if for every $\epsilon > 0$ there exists $\delta = \delta(\epsilon) > 0$ such that every solution with $\|y(0) - \tilde{y}\| \leq \delta$ satisfies
    \[\|y(t) - \tilde{y}\| \leq \epsilon \quad \forall\, t \geq 0.\]
    Intuitively: solutions that start close to $\tilde{y}$ remain close for all future time.
\end{definition}

\begin{definition}[Unstable Equilibrium]
    An equilibrium $\tilde{y}$ is \emph{unstable} if it is not stable: there exists $\epsilon > 0$ such that for every $\delta > 0$, there is a solution $y(t)$ with $\|y(0) - \tilde{y}\| \leq \delta$ and a time $T > 0$ for which $\|y(T) - \tilde{y}\| > \epsilon$.
\end{definition}

\begin{definition}[Asymptotic Stability]
    An equilibrium $\tilde{y}$ is \emph{asymptotically stable} if it is stable and there exists $\alpha > 0$ such that
    \[\|y(0) - \tilde{y}\| \leq \alpha \implies \lim_{t \to +\infty} \|y(t) - \tilde{y}\| = 0.\]
\end{definition}

\subsubsection{Examples}
\marginpar{March 12, 2026}

\begin{example}[Stable equilibrium]
    Let $y' = -2y$. The unique equilibrium is $\tilde{y} = 0$, and the solution is $y(t) = e^{-2t}y(0)$. For $t \geq 0$,
    \[|y(t) - 0| = e^{-2t}|y(0)| \leq |y(0)|.\]
    Given $\epsilon > 0$, choose $\delta = \epsilon$. Then $|y(0)| \leq \delta$ implies $|y(t)| \leq \epsilon$ for all $t \geq 0$, so $\tilde{y} = 0$ is stable.
\end{example}

\begin{example}[Logistic equation]
    Let $y' = y(1-y)$. The equilibria are $\tilde{y} = 0$ and $\tilde{y} = 1$.

    \textbf{$\tilde{y} = 0$ is unstable.} Take $\epsilon = \tfrac{1}{2}$ and any $0 < \delta < 1$. Choose $y(0) = \delta$; the solution is
    \[y(t) = \frac{\delta}{\delta + (1-\delta)e^{-t}}.\]
    There exists $T = T(\delta)$ such that $y(T) \geq \tfrac{1}{2}$, so $|y(T) - 0| \geq \epsilon$. Hence $\tilde{y} = 0$ is unstable.

    \textbf{$\tilde{y} = 1$ is asymptotically stable.} For any $y_0 > 0$,
    \[|y(t) - 1| = \left|\frac{(y_0-1)e^{-t}}{y_0 + (1-y_0)e^{-t}}\right| = \frac{|y_0-1|}{y_0 e^t + (1-y_0)} \leq |y_0 - 1|,\]
    since $y_0(e^t - 1) + 1 \geq 1$ for $t \geq 0$. Choosing $\delta = \epsilon$ gives stability. Moreover, as $t \to +\infty$ the numerator $|y_0 - 1|e^{-t} \to 0$, so $y(t) \to 1$ and $\tilde{y} = 1$ is asymptotically stable.
\end{example}

    \[\|y(t)\|_\infty = \|e^{At}y(0)\|_\infty \leq 2e^{-t}\|y(0)\|_\infty.\]

\end{example}

\begin{proof}[Proof of Picard Iterations Convergence]
    \label{pf:picard-convergence}
    Pick $t \in [t_0, t_0 + \epsilon]$ (the proof is similar for $t \in [t_0 - \epsilon, t_0]$).
    The goal is to show that $\{y_k(t)\}_{k=0}^{\infty}$ is a Cauchy sequence\footnote{Cauchy sequence is a sequence that has a limit in a metric space $\mathbb{R}^n$.} in $\mathbb{R}^n$.\footnote{Evaluated on compacted cylinder, understanding epsilon, the L Lipschitz constant coming from somewhere, not on the analysis background such as the Banach fixed point theorem.}
    We prove by induction that
    \[(**)\quad ||y_m(t) - y_{m - 1}(t)|| \leq L^{m - 1}M_{\alpha,\delta} \frac{(t - t_0)^m}{m!},\:\forall m \geq 1.\]
    \textbf{Base case \(m=1\):}
    \[||y_1(t) - y_0(t)|| = \left\| \int_{t_0}^{t} f(y_0,s)ds \right\| \leq \int_{t_0}^{t}||f(y_0,s)||ds \leq M_{\alpha,\delta}|t - t_0| \leq \alpha.\]
    \textbf{Induction step:}
    \[||y_{m + 1}(t) - y_m(t)|| \leq \int_{t_0}^{t}||f(y_m(s),s) - f(y_{m-1}(s),s)||ds.\]
    Since $D_{\alpha,\delta}$ is compact and $f$ is Lipschitz on $D_{\alpha,\delta}$, there exists $L$ such that
    \[||f(x,t) - f(y,t)|| \leq L||x - y||.\]
    Thus,
    \[\leq L \int_{t_0}^{t}||y_m(s) - y_{m - 1}(s)||ds.\]
    Using (**),
    \[\leq L^m M_{\alpha,\delta} \frac{1}{(m-1)!} \int_{t_0}^{t}(s - t_0)^{m-1}ds
    = L^m M_{\alpha,\delta} \frac{(t - t_0)^m}{m!}.\]
    Hence (**) holds.
    In particular, for all $\rho \geq 1$,
    \[||y_\rho(t) - y_{\rho-1}(t)|| \leq M_{\alpha,\delta} \frac{(L(t - t_0))^{\rho}}{(\rho)!} < \frac{M_{\alpha,\delta}}{L}\frac{(L\epsilon)^\rho}{\rho!}.\]
    Let $m,p \geq 1$:
    \[||y_{m+p}(t) - y_{m+1}(t)|| \leq \sum_{k=1}^{p-1}||y_{m+k+1}(t) - y_{m+k}(t)||.\]
    So,
    \[< \frac{M_{\alpha,\delta}}{L}\sum_{j=m+2}^{m+p}\frac{(L\epsilon)^j}{j!}.\]
    Since \(e^{L\epsilon} = \sum_{j=0}^{\infty}\frac{(L\epsilon)^j}{j!}\) converges,
    \[\to_{m,p \to +\infty} 0.\]
    Thus $\{y_k(t)\}$ is Cauchy and converges to $y(t)$.
    Taking limits in the iteration,
    \[y(t) = y_0 + \int_{t_0}^{t} f(y(s),s)ds.\]
\end{proof}

\begin{solution}[Existence and Uniqueness Theorem]
    Let $f(y,t) = t^2+y^2$. This function is continuous and has continuous partial derivatives $\forall(t,y) \in \mathbb{R}^2$, ensuring a unique local solution. To find the guaranteed interval of existence $\epsilon$, we define a rectangular domain $D$ centered at the origin:
    \[D_{\alpha,\delta} = \{(y,t) : \|y\| \leq \alpha, |t| \leq \delta\}.\]
    Now we need to bound the function. We find the maximum value $M$ of $|f(t, y)|$ on this domain:
    \[M = \sup_{(t, y) \in D_{\alpha, \delta}} |t^2 + y^2| = \delta^2 + \alpha^2.\]
    Now we determine the Existence Interval. According to the theorem, the solution is guaranteed to exist for $|t| \leq \epsilon$, where:
    \[\epsilon = \min \left(\delta, \frac{\alpha}{M}\right) = \min \left(\delta, \frac{\alpha}{\delta^2 + \alpha^2}\right).\]
    To find the largest possible value for $\epsilon$, we first maximize $h(\alpha) = \frac{\alpha}{\delta^2 + \alpha^2}$ with respect to $\alpha$ for a fixed $\delta$:
    \[h'(\alpha) = \frac{(\delta^2 + \alpha^2)(1) - \alpha(2\alpha)}{(\delta^2 + \alpha^2)^2} = \frac{\delta^2 - \alpha^2}{(\delta^2 + \alpha^2)^2}.\]
    Setting $h'(\alpha) = 0$ gives $\alpha = \delta$.Substituting $\alpha = \delta$ back into our expression for $\epsilon$:
    \[\epsilon = \min \left( \delta, \frac{\delta}{\delta^2 + \delta^2} \right) = \min \left( \delta, \frac{1}{2\delta} \right).\]
    The maximum value of this minimum occurs when the two terms are equal:
    \[\delta = \frac{1}{2\delta} \implies \delta^2 = \frac{1}{2} \implies \delta = \frac{1}{\sqrt{2}}.\]
    The largest guaranteed interval of existence $\epsilon$ using this rectangular domain is:
    \[\epsilon = \frac{1}{\sqrt{2}} \approx 0.707.\]
\end{solution}
