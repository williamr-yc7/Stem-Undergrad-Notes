\subsection{Introduction to Proof}
\subsubsection{Direct Proof}
The basic idea of a direct proof of $P \implies Q$ is:
\begin{enumerate}
    \item Assume $P$ is true,
    \item Use $P$ to show that $Q$ is true (the hard part).
\end{enumerate}
\begin{example}
    An integer number $n$ is odd if and only if there exists an integer $k$ such that $n=2k+1$.
    \begin{proof}
        Assume $a,b$ are consecutive integers, meaning $b = a + 1$. So 
        \[a+b = a+(a+1) = 2a+1\]
        Thus $b$ is odd if and only if $a$ is odd.
    \end{proof}
\end{example}

\subsubsection{Proof by Contradiction}
The idea here is that a proposition is either true or false, but not both. We get a contradiction when we can show a statement is both true and false, showing our initial assumptions are inconsistent. We use this to show $P \implies Q$ by assuming both $P$ and not $Q$ are simultaneously true and deriving a contradiction. Hopefully, we have either proved or assumed $P$ to be true so that by default, not $Q$ is false.

\begin{example}
    An integer number $n$ is odd if and only if there exists an integer $k$ such that $n=2k+1$.
    \begin{proof}
        Assume $a$ and $b$ are consecutive integers. Assume that $(a+b)$ is not odd. If $(a+b)$ is not odd, then no integer $k$ such that $(a+b)=2k+1$. But $a+b = a+(a+1)=2a+1$. Shown that $(a+b) \neq 2k+1$ but $(a+b)=2a+1$, showing a contradiction.        
    \end{proof}
\end{example}

\subsubsection{Proof by Contrapositive}
The idea behind a proof by contrapositive is to prove $P \implies Q$ by treating it the other way around meaning proving not $Q \implies$ not $P$.
\begin{example}
    If $n^2$ is even, then $n$ is even.
    \begin{proof}
        To prove this statement, we need to follow the proof by contrapositive, which states that we need to show if $n$ is not even, $n^2$ is not even too. Assume $n$ is odd, $\exists k_1 \in \mathbb{Z}, n = 2k_1+1$. So $n^2 = (2k_1+1)^2 = 4k_1^2+4k_1+1=2(2k_1^2+2k_1)+1$. Let $k_2 = 2k_1^2+2k_1$. So $n^2 = 2k_2 + 1$. Thus $n^2$ is odd.
    \end{proof}
\end{example}

\subsection{Complex Algebra}
\subsubsection{Complex Numbers}
\begin{definition}[Complex Number]
    A complex number is of the form: \(z = x + iy\) where \(x,y \in \mathbb{R}\) and \(i\) is the imaginary unit \(i = \sqrt{-1}\).
\end{definition}
\begin{theorem}[Fundamental Theorem of Algebra]
    The FTA states that any non-constant, single-variable polynomial\footnote{Polynomial is a function such as: \(f = a_n x^n + a_{n-1}x^{n-1} + ... + a_1 x + a_0\) where \(a_i \in \mathbb{R}\) or \(\mathbb{C}\) and \(n \in \mathbb{N}\).} with complex coefficients has at least one root in \(\mathbb{C}\).
\end{theorem}
\begin{remark}
    If we have a polynomial \(f\) of degree \(n\), then it has \(n\) roots, where each root can have a multiplicity\footnote{The multiplicity of a root represents how many times the root occurs in the polynomial.}.
\end{remark}
\begin{example}
    If we have a polynomial \((x-1)^2\), it has a degree of 2 but only one root, which is 1, with a multiplicity of 2.
\end{example}
We can factorize a polynomial in the form of \(f = a_n x^n + ... + a_1 x + a_0\) into a linear factor: \(f = a(x - z_1)(x - z_2)...(x - z_n)\) where \(z_i\) are the roots of \(f\) in \(\mathbb{C}\). Therefore, the FTA implies that \(f\) has a root \(f(z) = a(z-z) = 0\).
\subsubsection{Complex Operations}
\label{subsec:CO}
We can define operations on complex numbers as follows:
\begin{itemize}
    \item Addition: \(z + z' = (x + x') + i(y + y')\), where \(x,x',y,y' \in \mathbb{R}\).
    \item Multiplication: \(zz' = (x + iy)(x' + iy') = (xx' - yy') + i(xy' + yx')\).
    \item Inverse: \(\frac{1}{z} = \frac{\overline{z}}{z\overline{z}} = \frac{x - iy}{x^2 + y^2} = \frac{x}{x^2 + y^2} + i\frac{-y}{x^2 + y^2}\)
\end{itemize}
Multiplying by a complex number \(z\) corresponds geometrically to
\[\begin{cases}\text{a rotation by some angle } \theta,\\\text{a rescaling by the factor } |z|.\end{cases}\]
\subsubsection{Complex Conjugate}
\begin{definition}[Complex Conjugate]
    A complex conjugate is a way to "flip" the imaginary part of a complex number. For example, if we have a complex number \(z = x + iy\), then the complex conjugate of \(z\) is \(\overline{z} = x - iy\).
\end{definition}
Some basic properties of complex conjugates are:
\begin{itemize}
    \item \(\overline{z} = z\)
    \item \(\overline{z + z'} = \overline{z} + \overline{z'}\)
    \item \(\overline{z \cdot z'} = \overline{z} \cdot \overline{z'}\)
\end{itemize}

\subsubsection{Modulus}
\begin{definition}[Modulus]
    The modulus of a complex number \(z\) is defined by
    \[|z| = \sqrt{z\overline{z}} = \sqrt{x^2 + y^2}.\]
    Geometrically, \(|z|\) is the distance from the origin to the point \((x,y)\). Recall $r$ in polar coordinates, which is the same as \(|z|\).
\end{definition}

\subsubsection{Polar Form of Complex Numbers}
\begin{definition}[Euler's formula]
    \begin{equation}
        \label{eq:EulerFormula}
        e^{i\theta} = \cos\theta + i\sin\theta
    \end{equation}
\end{definition}
\begin{definition}[Polar and exponential form]
    If $z=x+iy$ with $r=\sqrt{x^2+y^2}$, then
    \[x=r\cos\theta, \quad y=r\sin\theta,\]
    so
    \[z=r(\cos\theta+i\sin\theta)=re^{i\theta}.\]
\end{definition}
\begin{definition}[Multiplication in polar form]
    \[z = re^{i\theta}, \quad z' = r'e^{i\theta'}, \quad zz' = rr'e^{i(\theta + \theta')}.\]
\end{definition}
\begin{example}
    \[(1 + i)^{32} = (\sqrt{2}e^{i\pi/4})^{32} = (\sqrt{2})^{32} e^{i8\pi} = 2^{16}(\cos 8\pi + i \sin 8\pi) = 2^{16}.\]
\end{example}

\subsubsection{Roots in $\mathbb{C}$}
\begin{definition}[\(n^{\text{th}}\) roots in $\mathbb{C}$]
    For any complex number $z$, an $n^{\text{th}}$ root of $z$ is a complex number $w$ such that
    \[w^n = z.\]
\end{definition}
\begin{example}
    If \(z = re^{i\theta}\), then any solution of \(w^n = z\) must satisfy
    \[w_k = r^{1/n} e^{i(\theta + 2\pi k)/n}, \quad k = 0,1,2,\dots,n-1,\]
    where the $n^{\text{th}}$ roots of $z$ are equally spaced on a circle of radius $r^{1/n}$ centered at the origin.
\end{example}
\begin{definition}[Roots of unity]
The \(n^{\text{th}}\) roots of unity are the solutions of a special case where $z = 1$.
\end{definition}
\begin{example}
    If $z = 1$, then any solution of $w^n = z$ must satisfy
    \[w_k = e^{i2\pi k/n}, \quad k = 0,1,2,\dots,n-1.\]
    Geometrically, they lie on the unit circle and are equally spaced.
\end{example}

\subsubsection{Hermitian}
\begin{definition}[Hermitian Matrix]
    A complex square matrix $A \in M_n(\mathbb{C})$ is called Hermitian if it is equal to its conjugate transpose:
    \[A = A^\ast\]
    where $A^\ast = \overline{A}^T$.
\end{definition}
Equivalently, a matrix $A=(a_{ij})$ is Hermitian if
\[a_{ij} = \overline{a_{ji}} \quad \text{for all } i,j.\]
\begin{example}
    \[A = \begin{pmatrix}2 & 1+i \\ 1-i & 3\end{pmatrix}\]
    is Hermitian since
    \[A^\ast = \begin{pmatrix}2 & 1+i \\ 1-i & 3\end{pmatrix} = A.\]
\end{example}
\begin{remark}
    All eigenvalues of a Hermitian matrix are real and eigenvectors corresponding to distinct eigenvalues are orthogonal.
\end{remark}
\begin{problem}[Hermitian Matrix]
    Let $A,B \in M_n(\mathbb{C})$. Prove that $(AB)^\ast = B^\ast A^\ast$.
\end{problem}

\subsection{Basic Algebraic structures}
\subsubsection{Sets with Multiplication}
\begin{definition}[Set with multiplication]
A set $M$ is called a set with multiplication if you can multiply any two elements of $M$, and the result is still in $M$.
In other words, for any $a,b \in M$, the product $ab$ is also in $M$.
\end{definition}
\begin{example}
    An example of a set with multiplication is the set of all \(2 \times 2\) complex matrices: \(M = M_2(\mathbb{C})\). Another example is the nonzero set of all real numbers \(\mathbb{R}\) with ordinary multiplication: \(\mathbb{R}^* = \mathbb{R} \setminus \{0\}\).
\end{example}
\begin{example}
Let $M = \mathbb{R}$.  
If $a,b \in \mathbb{R}$, then $ab \in \mathbb{R}$.  
So the real numbers $\mathbb{R}$ form a set with multiplication.
\end{example}

\subsubsection{Invertibility}
\begin{definition}[Condition for Invertibility]
    Let \(A \in M\) be an \(n \times n\) matrix, and suppose that there exists an \(n \times n\) matrix \(B\) such that \(AB = I_n\) or \(BA = I_n\). Where \(I_n\) is the \(n \times n\) identity matrix\footnote{An identity matrix is a square matrix with 1s on its main diagonal and 0s everywhere else. It represents no change in linear transformations, and it's used in finding matrix inverses.}$\begin{bmatrix}1&0&0\\0&\ldots&0\\0&0&1\end{bmatrix}$. Then A is invertible, and \(B\) is called the inverse of \(A\) and is denoted by \(B = A^{-1}\).
\end{definition}
\begin{remark}
    If \(A\) is invertible, then \(A^{-1}\) exists and is unique\footnote{Unique means there is exactly one such element.}.
\end{remark}

To determine if an element \(A\) in a set with multiplication \(M\) is invertible, we can use the following examples:
\begin{example}
    Let \(M = \mathbb{Z} = \{...-2,-1,0,1,2,...\}\) and \(A = 2\). Is \(A\) invertible in \(M\)?\\
    Solution: No, because \(\frac{1}{2} \notin \mathbb{Z}\).
\end{example}
\begin{example}
    Let \(M = \mathbb{R}\) and \(A = 2\), is \(A\) invertible in \(M\)?\\
    Solution: Yes, because \(\frac{1}{2} \in \mathbb{R}\).
\end{example}
\begin{example}
    Is \(1 + i\) invertible in \(\mathbb{C}\)?\\
    Solution: Yes, using our previous definition of inverse \eqref{subsec:CO}, we get that \[\frac{1}{1 + i} = \frac{1 - i}{2} \in \mathbb{C}.\]
\end{example}
\begin{problem}[Invertibility]
    Show that if an inverse of A in \(\mathbb{M}\) exists, then it is unique.
\end{problem}
\begin{problem}[Invertibility 2]
    Let \(K\) be a field. Prove that this matrix $\begin{bmatrix}0&1\\0&0\end{bmatrix}$ \(\in M_2(K)\) is not invertible.
\end{problem}

\subsubsection{Ring}
\begin{definition}[Ring]
    \label{def:Ring}
    A ring is a set $R$ where you can add and multiply elements, and the following are true:

    \begin{enumerate}
        \item You can add any two elements and stay in $R$. There is a zero, every element has a negative, and addition is commutative\footnote{Property which focuses on changing order of addition, i.e $a + b = b + a$.} and associative\footnote{Property which focuses on changing grouping of addition, i.e $a + (b + c) = (a + b) + c$.}.
        \item You can multiply any two elements and stay in $R$. Multiplication is associative, and there is a $1$.
        \item Multiplication distributes over addition:
        \[a(b+c) = ab + ac \quad \text{and} \quad (a+b)c = ac + bc.\]
    \end{enumerate}
\end{definition}
\begin{example}
    The set of integers $\mathbb{Z}$ with the usual addition and multiplication is a ring. First, $\mathbb{Z}$ is closed under addition and multiplication. If $a,b \in \mathbb{Z}$, then $a+b \in \mathbb{Z}$ and $ab \in \mathbb{Z}$. There is a zero element $0$ such that
    \[a+0=a \quad \text{for all } a\in\mathbb{Z},\]
    and every integer has an additive inverse since
    \[a+(-a)=0.\]
    Addition is commutative and associative:
    \[a+(b+c)=(a+b)+c.\]
    Multiplication is associative and has identity $1$:
    \[a(bc)=(ab)c,\]
    \[a\cdot1=a.\]
    Finally, multiplication distributes over addition:
    \[(a+b)c=ac+bc.\]
    Therefore, $(\mathbb{Z},+,\cdot)$ is a ring.
\end{example}

\subsubsection{Field}
\begin{definition}[Field]
    A field is a ring \textup{\eqref{def:Ring}} in which every nonzero element has a multiplicative inverse.
\end{definition}
\begin{example}
    The set of rational numbers $\mathbb{Q}$ with the usual addition and multiplication is a field. First, $(\mathbb{Q},+,\cdot)$ is a ring because rational numbers are closed under addition and multiplication, addition is commutative and associative, there is a zero element $0$, and multiplication is associative with identity $1$. In addition, every nonzero element has a multiplicative inverse. If $q \in \mathbb{Q}$ and $q \neq 0$, then there exists $q^{-1}$ such that
    \[q\cdot q^{-1}=1.\]
    For example,
    \[\frac{3}{5}\cdot\frac{5}{3}=1.\]
    Therefore, $(\mathbb{Q},+,\cdot)$ is a field.
\end{example}
\begin{remark}
    The integers $\mathbb{Z}$ form a ring but not a field because most nonzero elements do not have a multiplicative inverse in $\mathbb{Z}$ (for example, there is no integer $x$ such that $2x=1$).
\end{remark}

\section{Vector Spaces}
\begin{definition}[Vector Space]
    Let $k$ be a field. A vector space over $k$ (or K-vectorspace) is a set $V$ with addition and scalar multiplication which satisfies the following axioms
    \begin{enumerate}
        \item Closure under addition: $v_1+v_2\in V$
        \item Commutativity of addition: \(v_1 + v_2 = v_2 + v_1\)
        \item Associativity of addition: \((v_1 + v_2) + v_3 = v_1 + (v_2 + v_3)\)
        \item Identity Element of Addition: There exists a zero vector $0 \in V$ such that $v+0=v$
        \item Existence of additive inverses: \(\forall v \in V, \exists -v \in V\) with \(v + (-v) = 0\)
        \item Closure under scalar multiplication: $\lambda v\in V$
        \item Compatibility of scalar multiplication with field multiplication: \(\lambda(\mu v) = (\lambda\mu)v\)
        \item Identity element of scalar multiplication: \(1 v = v\)
        \item Distributivity of scalar multiplication over vector addition: \(\lambda(v_1 + v_2) = \lambda v_1 + \lambda v_2\)
        \item Distributivity of scalar multiplication over field addition: \((\lambda + \mu)v = \lambda v + \mu v\)
    \end{enumerate}
\end{definition}
\begin{remark}
    We often use the term abstract vector space to emphasize that the elements of $V$ are not restricted to arrows\footnote{An arrow is a function or mapping between two sets.} or tuples\footnote{A tuple is an ordered sequence where duplicates are allowed, whereas a set is an unordered collection of unique elements.} in $\mathbb{R}^n$, but can be any objects (like functions or tensors) that satisfy the axioms.
\end{remark}

\subsection{Vectors}
\subsubsection{Vector operations}
Vector operations are defined as follows:\\
\begin{itemize}
    \item Addition: \(\begin{pmatrix}x_1\\\vdots\\x_n\end{pmatrix} + \begin{pmatrix}y_1\\\vdots\\y_n\end{pmatrix} = \begin{pmatrix}x_1 + y_1\\\vdots\\x_n + y_n\end{pmatrix}\).
    \item Scalar multiplication: \(\lambda \begin{pmatrix}x_1\\\vdots\\x_n\end{pmatrix} = \begin{pmatrix}\lambda x_1\\\vdots\\\lambda x_n\end{pmatrix}\), where $\lambda \in \mathbb{R}$.
\end{itemize}
\begin{definition}[Linear combination]
    A linear combination of vectors \(v_1, \ldots, v_n\) is a vector \(v\) of the form \(v = \lambda_1v_1 + \ldots + \lambda_nv_n\), where \(\lambda_1, \ldots, \lambda_n \in \mathbb{R}\).
\end{definition}

\subsection{Span}
\marginpar{January 12, 2026.}
\begin{definition}[Span]
    Let \(A \subset \mathbb{R}^n\). The span of \(A\), denoted \(\Span(A)\), is the set of all linear combinations of elements of \(A\). In particular, if \(A=\{v_1,\ldots,v_k\}\) (where \(k\) is the number of vectors in the set), then
    \[\Span(A) = \{\lambda_1 v_1+\cdots+\lambda_k v_k : \lambda_i \in \mathbb{R}\}.\]
\end{definition}
When working in \(\mathbb{R}^n\), the span describes all points you can reach by scaling and adding the given vectors. Depending on the vectors, the span can be a line (if the vectors are dependent), a plane, or a higher-dimensional subspace.
\begin{remark}
    If \(A=\{v\}\subset \mathbb{R}^2\) with \(v\neq 0\), then
    \[\Span(v)=\{\lambda v : \lambda \in \mathbb{R}\}.\]
    This set consists of all scalar multiples of \(v\), which form a line in the direction of \(v\). In particular, taking \(\lambda=0\) gives \(0\in\Span(v)\), so the line passes through the origin.
\end{remark}

\begin{example}
    Let \(A = \begin{pmatrix}1\\1\end{pmatrix}\). Then $\Span(A) = \left\{ t\begin{pmatrix}1\\1\end{pmatrix} \mid t \in \mathbb{R} \right\},$ which is a line in \(\mathbb{R}^2\).
\end{example}
\begin{problem}[Span]
    Let \(v_1 = \begin{pmatrix}3\\1\end{pmatrix}\), \(v_2 = \begin{pmatrix}1\\3\end{pmatrix}\). Find $\Span(v_1,v_2)$.
\end{problem}
Span is a generalization of lines in \(\mathbb{R}^2\). For example, let $v_1$ and $v_2$ be two linearly independent vectors in \(\mathbb{R}^2\).
Any vector in \(\Span(v_1,v_2)\) has the form \(\lambda_1 v_1+\lambda_2 v_2\). Geometrically, this can be illustrated as the
sum of two scaled vectors.
\begin{center}
    \begin{tikzpicture}[scale=1, >=Stealth]
        \draw[->] (-0.5,0) -- (5,0);
        \draw[->] (0,-0.5) -- (0,4);
        \coordinate (v1) at (3,1);
        \coordinate (v2) at (1,3);
        \draw[->, thick] (0,0) -- (v1) node[right] {$\lambda_1 v_1$};
        \draw[->, thick] (0,0) -- (v2) node[left] {$\lambda_2 v_2$};
        \draw[dashed] (v1) -- ($(v1)+(v2)$);

\subsubsection{Span in \(\mathbb{C}^n\)}
\begin{definition}[Span in $\mathbb{C}^n$]
    The span over \(\mathbb{C}\) of vectors \(v_1,\ldots,v_n \in \mathbb{C}^n\) is
    \[\Span_{\mathbb{C}}(v_1,\ldots,v_n) = \{\lambda_1 v_1 + \cdots + \lambda_n v_n : \lambda_i \in \mathbb{C}\}.\]
\end{definition}
\begin{example}
    Let
    \[v_1=\begin{pmatrix}1\\0\end{pmatrix}, \quad v_2=\begin{pmatrix}0\\i\end{pmatrix} \in \mathbb{C}^2.\]
    Find $\Span_{\mathbb{C}}(v_1,v_2)$. For \(\lambda_1,\lambda_2 \in \mathbb{C}\),
    \[\lambda_1 v_1+\lambda_2 v_2 = \lambda_1\begin{pmatrix}1\\0\end{pmatrix} + \lambda_2\begin{pmatrix}0\\i\end{pmatrix} = \begin{pmatrix}\lambda_1\\ i\lambda_2\end{pmatrix}.\]
    Since \(\lambda_1\) is arbitrary and \(i\lambda_2\) can represent any complex number, every vector in \(\mathbb{C}^2\) can be written in this form. Hence
    \[\Span_{\mathbb{C}}(v_1,v_2) = \mathbb{C}^2.\]
\end{example}

\subsection{Standard Basis}
\begin{definition}[Standard basis of \(\mathbb{R}^n\)]
    The standard basis of \(\mathbb{R}^n\) is the set \(\{e_1,\ldots,e_n\}\), where each vector \(e_i\) has a \(1\) in the \(i\)-th coordinate and \(0\) in all other coordinates:
    \[e_1=\begin{pmatrix}1\\0\\\vdots\\0\end{pmatrix},\;e_2=\begin{pmatrix}0\\1\\\vdots\\0\end{pmatrix},\;\ldots,\;e_n=\begin{pmatrix}0\\0\\\vdots\\1\end{pmatrix}.\]
    Equivalently, the vectors \(e_1,\ldots,e_n\) are the columns of the \(n \times n\) identity matrix $I_n$.
\end{definition}
\begin{proposition}
    Every vector \(x=\begin{pmatrix}x_1\\ \vdots \\ x_n\end{pmatrix}\in\mathbb{R}^n\) can be written uniquely as $x = x_1 e_1 + \cdots + x_n e_n.$
\end{proposition}
\begin{remark}
    To verify that the standard basis of \(\mathbb{R}^n\) is indeed a basis of \(\mathbb{R}^n\), we must check two properties: it spans \(\mathbb{R}^n\) and it is linearly independent.\footnote{To prove the standard basis is a basis, please refer to the Appendix \eqref{subsec:SB}.}
\end{remark}

The standard basis also helps clarify the difference between real and complex vector spaces. In particular, the same vectors can generate very different spans depending on whether the scalars are real or complex.
\begin{example}[Real vs complex span of the same vectors]
    Let
    \[e_1=\begin{pmatrix}1\\0\end{pmatrix},\quad
    e_2=\begin{pmatrix}0\\1\end{pmatrix}.\]
    \textbf{Over \(\mathbb{C}\):}
    Using complex scalars, any vector in \(\mathbb{C}^2\) can be written as
    \[\lambda_1 e_1 + \lambda_2 e_2 = \begin{pmatrix}\lambda_1 \\ \lambda_2\end{pmatrix}, \quad \lambda_1,\lambda_2 \in \mathbb{C}.\]
    Thus \(\Span_{\mathbb{C}}(e_1,e_2) = \mathbb{C}^2\), and only 2 vectors are needed:
    \[\dim_{\mathbb{C}} \mathbb{C}^2 = 2.\]
    \textbf{Over \(\mathbb{R}\):}  
    Now only real scalars are allowed. Then
    \[a_1 e_1 + a_2 e_2 = \begin{pmatrix}a_1 \\ a_2\end{pmatrix}, \quad a_1,a_2 \in \mathbb{R}.\]
    Thus $\Span_{\mathbb{R}}(e_1,e_2) = \mathbb{R}^2$, and only 2 vectors are needed: $\dim_{\mathbb{R}} \mathbb{R}^2 = 2$.\\
    However, the standard basis of \(\mathbb{R}^2\) is not enough to span \(\mathbb{C}^2\), since it cannot produce vectors with imaginary parts. To span all of \(\mathbb{C}^2\) over \(\mathbb{R}\), we also need
    \[i e_1 = \begin{pmatrix}i\\0\end{pmatrix}, \quad i e_2 = \begin{pmatrix}0\\i\end{pmatrix}.\]
    Now any vector in \(\mathbb{C}^2\) can be expressed as a real linear combination of the 4 vectors
    \[ae_1 + be_2 + c(i e_1) + d(i e_2) = \begin{pmatrix}a+ci\\b+di\end{pmatrix}, \quad a,b,c,d\in\mathbb{R}.\]
    Thus $\Span_{\mathbb{R}}(e_1,e_2,i e_1,i e_2) = \mathbb{C}^2$ and therefore
    \[\dim_{\mathbb{R}} \mathbb{C}^2 = 4.\]
\end{example}

\subsection{Coordinate Spaces}
\marginpar{14 January, 2026}
\begin{definition}[Coordinate Spaces]
    A coordinate space of dimension \(n\) over a field \(k\) (\(k=\mathbb{R}\) or \(k=\mathbb{C}\)) is defined as
    \[k^n = \left\{\begin{pmatrix}x_1\\\vdots\\x_n\end{pmatrix} \mid x_i \in k\right\}.\]
    The standard basis for a coordinate space is $\{e_1,\ldots,e_n\}$, where
    \[e_1=\begin{pmatrix}1\\0\\\vdots\\0\end{pmatrix}, \ \ldots \ , e_n=\begin{pmatrix}0\\\vdots\\0\\1\end{pmatrix}.\]
    The dimension of \(k^n\) is $\dim{k^n} = n$.
\end{definition}

\subsection{Polynomial Spaces}
\begin{definition}[Polynomial Spaces]
    A polynomial space of degree at most \(n\) over a field \(k\) is
    \[P_n(k) = \{ a_n x^n + \cdots + a_1 x + a_0 \mid a_i \in k \},\]
    which forms a vector space over \(k\) with standard basis $\{1, x, x^2, \ldots, x^n\}$. Under the identification \(P_n(k)\cong k^{n+1}\), these basis elements have coordinate vectors
    \[1 \leftrightarrow \begin{pmatrix}1\\0\\\vdots\\0\end{pmatrix},\quad x \leftrightarrow \begin{pmatrix}0\\1\\\vdots\\0\end{pmatrix},\quad \ldots,\quad x^n \leftrightarrow \begin{pmatrix}0\\\vdots\\0\\1\end{pmatrix}.\]
    The dimension of \(P_n(k)\) is $\dim{P_n(k)} = n+1$.
\end{definition}
\begin{example}
    Examples of polynomials in these spaces are
    \[1 + x^2 \in P_2(\mathbb{R}), \qquad 1 + i x^3 \in P_3(\mathbb{C}).\]
    In each case the degree of the polynomial satisfies \(\deg(f)\le n\).
\end{example}
\begin{remark}
    All polynomials over \(k\) form the space
    \[P_\infty(k)=\{a_nx^n+\cdots+a_1x+a_0 : a_i\in k,\ n\ge0\},\]
    which is an infinite-dimensional vector space over \(k\).
\end{remark}
\begin{remark}
    The polynomial spaces form a nested sequence
    \[P_0(k) \subseteq P_1(k) \subseteq P_2(k) \subseteq \cdots \subseteq P_\infty(k).\]
\end{remark}

\subsection{Matrix Spaces}
\begin{definition}[Matrix Spaces]
    The set of all \(n \times n\) matrices over a field \(k\) is
    \[M_n(k) = \left\{ \begin{pmatrix}
    a_{11} & \cdots & a_{1n} \\
    \vdots & \ddots & \vdots \\
    a_{n1} & \cdots & a_{nn}
    \end{pmatrix} \mid a_{ij} \in k \right\},\]
    which forms a vector space over \(k\).
    A standard basis for \(M_n(k)\) is the set of matrices \(\{e_{ij} : 1 \le i,j \le n\}\), where \(e_{ij}\) has a 1 in the \((i,j)\)-th entry and 0 elsewhere. For example,
    \[e_{11}=\begin{pmatrix}1 & 0 & \cdots & 0\\0 & 0 & \cdots & 0\\\vdots & \vdots & \ddots & \vdots\\0 & 0 & \cdots & 0\end{pmatrix},\quad e_{12}=\begin{pmatrix}0 & 1 & \cdots & 0\\0 & 0 & \cdots & 0\\\vdots & \vdots & \ddots & \vdots\\0 & 0 & \cdots & 0\end{pmatrix},\quad e_{21}=\begin{pmatrix}0 & 0 & \cdots & 0\\1 & 0 & \cdots & 0\\\vdots & \vdots & \ddots & \vdots\\0 & 0 & \cdots & 0\end{pmatrix}.\]
    The dimension of \(M_n(k)\) is \(\dim M_n(k)=n^2\).
\end{definition}
\begin{example}[Matrix Spaces]
    For \(M_2(\mathbb{R})\),
    \[\begin{pmatrix}1 & 2 \\ 3 & 4\end{pmatrix} = 1\begin{pmatrix}1 & 0 \\ 0 & 0\end{pmatrix} + 2\begin{pmatrix}0 & 1 \\ 0 & 0\end{pmatrix} + 3\begin{pmatrix}0 & 0 \\ 1 & 0\end{pmatrix} + 4\begin{pmatrix}0 & 0 \\ 0 & 1\end{pmatrix} = 1 e_{11} + 2 e_{12} + 3 e_{21} + 4 e_{22}.\]
\end{example}

\subsection{Function Spaces}
\marginpar{16 January, 2026}
\begin{definition}[Function spaces]
    Let $D$ be a set and $k$ a field. Define
    \[F(D,k) = \{ f : D \to k \}.\]
    With pointwise operations
    \[(f+g)(x) = f(x)+g(x), \quad (\lambda f)(x) = \lambda f(x),\]
    $F(D,k)$ is a vector space over $k$.
\end{definition}
A function \(f \in F(D,k)\) is completely determined by its values on \(D\). If \(D=\{1,\dots,n\}\), this gives \(n\) numbers \(f(1),\dots,f(n)\). To construct a basis, we introduce functions that isolate each of these values.
\paragraph{1. Finite case}
Assume $D = \{1,\dots,n\}$. For each $i \in D$, define the Kronecker delta function
\[\delta_i(j) = \begin{cases}1, & i=j,\\0, & i \ne j.\end{cases}\]
Then $\{\delta_1,\dots,\delta_n\}$ is a basis of $F(D,k)$. Moreover, every $f \in F(D,k)$ can be written uniquely as
\[f = \sum_{i=1}^n f(i)\,\delta_i.\]
Indeed, evaluating at $j$ gives
\[\sum_{i=1}^n f(i)\delta_i(j) = f(j),\]
since \(\delta_i(j)=0\) unless \(i=j\). Hence the functions span, and linear independence follows by evaluating a linear relation at each point of \(D\). Therefore
\[\dim F(D,k)=n.\]
\paragraph{2. Infinite case}
If $D$ is infinite, $F(D,k)$ has no finite basis.  
The family $\{\delta_x : x \in D\}$ is linearly independent but does not span $F(D,k)$.
\begin{example}[Vector space of functions]
    Let $D = \{1,2,3\}$ and $k = \mathbb{R}$. Consider the vector space $F(D,\mathbb{R}) = \{ f : D \to \mathbb{R} \}$. Find a basis of $F(D,\mathbb{R})$ and express $f(1)=2, f(2)=-1, f(3)=3$ as a linear combination of the basis functions.\\
    1. Define the standard (Kronecker delta) functions: 
    \[\delta_1(j) = \begin{cases}1,& j=1\\0,& j\ne 1\end{cases}, \quad \delta_2(j) = \begin{cases}1,& j=2\\0,& j\ne 2\end{cases}, \quad \delta_3(j) = \begin{cases}1,& j=3\\0,& j\ne 3\end{cases}.\]
    Then $\{\delta_1,\delta_2,\delta_3\}$ is a basis of $F(D,\mathbb{R})$.\\
    2. Write $f$ as a linear combination:
    \[f = f(1)\delta_1 + f(2)\delta_2 + f(3)\delta_3 = 2\delta_1 - 1\delta_2 + 3\delta_3.\]
    3. Evaluating at each point:
    \[f(1) = 2\cdot 1 + (-1)\cdot 0 + 3\cdot 0 = 2, \quad f(2) = 2\cdot 0 + (-1)\cdot 1 + 3\cdot 0 = -1,\]
    \[\quad f(3) = 2\cdot 0 + (-1)\cdot 0 + 3\cdot 1 = 3.\]
    Hence the decomposition is correct.
\end{example}

\subsection{Subspaces}
\marginpar{January 19, 2026}
\begin{definition}[Subspace]
    A subspace is a subset of a larger vector space that is itself a vector space. A subspace of $V$ is a subset $U$ of $V$ which is a vector space (over the same field $k$) with respect to the operations of addition and scalar multiplication induced from $V$.
\end{definition}
\begin{proposition}[Subspace criterion]
    Let $V$ be a vector space over a field $k$ and let $U \subseteq V$. Then $U$ is a vector subspace of $V$ if and only if:
    \begin{enumerate}
    \item $0 \in U$,
    \item $u,v \in U \Rightarrow u+v \in U$,
    \item $u \in U,\ \lambda \in k \Rightarrow \lambda u \in U$.
    \end{enumerate}
\end{proposition}
\begin{problem}[Subspace criterion]
    Let $A \in M_n(k)$ be a fixed matrix. Prove that
    \[U = \{Ax = \vec{0} \mid x \in k^n\}\]
    is a subspace, null space or kernel.
\end{problem}
\begin{problem}[Subspace criterion 2]
    Let $A \in M_n(k)$. Show that
    \[U = \{Ax \mid x \in K^n\}\]
    is a subspace of $k^n$. The set $U$ is called the image (or range) of $A$.
\end{problem}
\begin{problem}[Subspace criterion 3]
    Let $V = k$, viewed as a vector space over $k$. Show that the only subspaces of $V$ are $\{0\}$ and $V$ itself.
\end{problem}
\begin{problem}[Subspace criterion 4]
Let $V$ be a vector space over $k$ and let $v_1,\dots,v_n \in V$. Show that
\[\Span(v_1,\dots,v_n) = \{\lambda_1 v_1 + \cdots + \lambda_n v_n \mid \lambda_1,\dots,\lambda_n \in k\}\]
is a subspace of $V$.
\end{problem}

\subsubsection{Operations on subspaces}
Let $W$ be a vector space and $U,V \subseteq W$ subspaces.
\begin{definition}[Intersection]
    \[U \cap V = \{w \in W \mid w \in U \text{ and } w \in V\}.\]
    The intersection of subspaces is a subspace. In particular, the smallest possible intersection is $\{0\}$.
\end{definition}
\begin{definition}[Union]
    \[U \cup V = \{ w \in W \mid w \in U \text{ or } w \in V \}.\]
    The union of subspaces is not necessarily a subspace.
\end{definition}
\begin{proposition}
    In general, $U \cup V$ is a subspace of $W$ only if one subspace is contained in the other, i.e., $U \subseteq V$ or $V \subseteq U$.
\end{proposition}
\begin{proof}
    First assume \(U \subseteq V\). Then
    \[U \cup V = V,\]
    which is a subspace of \(W\). Similarly, if \(V \subseteq U\), then
    \[U \cup V = U,\]
    which is also a subspace. Conversely, suppose neither \(U \subseteq V\) nor \(V \subseteq U\). Then there exist \(u \in U\) with \(u \notin V\), and \(v \in V\) with \(v \notin U\).
    Thus \(u,v \in U \cup V\). If \(U \cup V\) were a subspace, it would be closed under addition, so
    \[u+v \in U \cup V.\]
    If \(u+v \in U\), then since \(U\) is a subspace and \(u \in U\),
    \[v = (u+v) - u \in U,\]
    a contradiction. If \(u+v \in V\), then since \(V\) is a subspace and \(v \in V\),
    \[u = (u+v) - v \in V,\]
    also a contradiction. Therefore \(u+v \notin U \cup V\), so \(U \cup V\) is not closed under addition. Hence \(U \cup V\) is not a subspace if neither \(U \subseteq V\) nor \(V \subseteq U\).
\end{proof}
\begin{definition}[Sum of subspaces]
    \[U + V = \{u+v \mid u \in U,\, v \in V\}.\]
\end{definition}
\begin{proposition}
    If $U$ and $V$ are subspaces of $W$, then $U+V$ is a subspace of $W$.
\end{proposition}
\begin{proof}
    We apply the subspace criterion.
    \begin{enumerate}
        \item Since $0 \in U$ and $0 \in V$, we have $0 = 0+0 \in U+V$.
        \item Let $w_1,w_2 \in U+V$. Then
        \[w_1=u_1+v_1,\quad w_2=u_2+v_2,\]
        with $u_1,u_2 \in U$ and $v_1,v_2 \in V$. Hence
        \[w_1+w_2=(u_1+u_2)+(v_1+v_2) \in U+V.\]
        \item Let $w \in U+V$ and $\lambda \in K$. Then $w=u+v$ and
        \[\lambda w=\lambda u+\lambda v \in U+V.\]
    \end{enumerate}
\end{proof}
\begin{proposition}
    \[U + V = \Span(U \cup V).\]
    Equivalently, $U+V$ is the smallest subspace of $W$ containing both $U$ and $V$.
\end{proposition}
\begin{proof}
    ($\subseteq$): Let $w \in U+V$. Then $w=u+v$ with $u \in U$, $v \in V$. Since $u,v \in U \cup V$, we have $w \in \Span(U \cup V)$.

    ($\supseteq$): Let $w \in \Span(U \cup V)$. Then
    \[w = a_1u_1 + \cdots + a_nu_n + b_1v_1 + \cdots + b_mv_m\]
    with $u_i \in U$, $v_j \in V$. Grouping terms,
    \[w = (a_1u_1 + \cdots + a_nu_n) + (b_1v_1 + \cdots + b_mv_m),\]
    so $w \in U+V$.
\end{proof}

\subsection{Direct sums}
\marginpar{January 23, 2026}
Let $W$ be a vector space and $U,V \subseteq W$ be subspaces.
\begin{definition}[Direct sum]
    We say that $U$ and $V$ are in direct sum if
    \[U \cap V = \{0\}.\]
    In this case, their sum
    \[U+V=\{u+v \mid u \in U,\; v \in V\}\]
    is denoted by $U \oplus V$.
    We write
    \[W = U \oplus V \iff \begin{cases}W = U+V,\\U \cap V = \{0\}.\end{cases}\]
\end{definition}
\begin{proof}
    Assume \(W = U \oplus V\). Then \(W = U+V\) and \(U \cap V = \{0\}\). First, since \(W = U+V\), every \(w \in W\) can be written as
    \[w = u+v, \quad u \in U,\ v \in V.\]
    To show uniqueness, suppose
    \[w = u_1 + v_1 = u_2 + v_2\]
    with \(u_1,u_2 \in U\) and \(v_1,v_2 \in V\). Then
    \[u_1 - u_2 = v_2 - v_1.\]
    Since \(U\) is a subspace, \(u_1-u_2 \in U\). Since \(V\) is a subspace, \(v_2-v_1 \in V\). Hence
    \[u_1-u_2 \in U \cap V.\]
    But \(U \cap V = \{0\}\), so
    \[u_1-u_2 = 0.\]
    Thus \(u_1 = u_2\), and therefore \(v_1 = v_2\). Hence the decomposition is unique. Conversely, assume every \(w \in W\) can be written uniquely as \(w=u+v\) with \(u \in U\), \(v \in V\). Then clearly \(W = U+V\). To show \(U \cap V = \{0\}\), let \(w \in U \cap V\). Then
    \[w = w + 0 = 0 + w,\]
    where \(w \in U\), \(0 \in V\), and \(0 \in U\), \(w \in V\).
    By uniqueness of the decomposition, we must have
    \[w = 0.\]
    Thus \(U \cap V = \{0\}\), so \(W = U \oplus V\).
\end{proof}

\subsubsection{Direct sums in function spaces}
\begin{definition}[Subspace of functions supported on a subset]
    Let $D$ be a set and $A \subseteq D$. Consider the function space $F(D,\mathbb{R})$. Define
    \[U = \{f \in F(D,\mathbb{R}) : f(x)=0\,\forall x \notin A\}.\]
    Then $U$ is a subspace of $F(D,\mathbb{R})$ (it is closed under addition and scalar multiplication, and contains the zero function). Moreover, $U$ can be naturally identified with $F(A,\mathbb{R})$, since each function in $U$ is completely determined by its values on $A$.
\end{definition}
\begin{remark}
    Every function in $U$ is completely determined by its values on $A$ and vanishes outside $A$.
\end{remark}
\begin{example}[Functions supported on a subset]
    Let $D = \{1,2,3,4,5\}$ and $A = \{2,4\}$. Then
    \[U = \{ f \in F(D,\mathbb{R}) : f(1)=f(3)=f(5)=0 \}.\]

    Each function in $U$ is completely determined by its values on $A$, i.e.,
    \[f \iff (f(2), f(4)) \in F(A,\mathbb{R}) \cong \mathbb{R}^2.\]
    Hence $U$ is a 2-dimensional subspace of $F(D,\mathbb{R})$.
\end{example}

\subsection{Finite Dimensional Spaces}
Let V be a vector space over a field k. Then there are two possibilities:
\begin{itemize}
    \item The zero vector space: $V = \{0\}$
    \item The non-zero vector space: $V \neq \{0\}$
\end{itemize}
In the second case, there exists a non-zero vector $v_1 \in V$ such that $v_1 \neq 0$. Then $span(v_1)$ is a subspace of $V$, and there are two possibilities:
\begin{itemize}
    \item $V = span(v_1)$, i.e., $v_1$ is a generator of $V$.
    \item $V \neq span(v_1)$, i.e., there exists a $v_2 \in V$ such that $v_2 \notin span(v_1)$.
\end{itemize}
In the second case, $span(v_1,v_2) \subseteq V$. This process can be repeated to obtain a sequence of vectors $v_1,v_2,\ldots,v_n$ such that $span(v_1,\ldots,v_n) \subseteq V$. The maximum number of linearly independent vectors in $V$ is called the dimension of $V$, denoted by $\dim V$.

2 cases (2d space plane):
\begin{itemize}
    \item If $V = span(v_1, v_2)$, you have a 2D plane.
    \item If the space is larger than that plane, there must exist a $v_3$ that sticks out of it, meaning $span(v_1, v_2, v_3) \subseteq V$.
\end{itemize}

2 cases (3d space):
\begin{itemize}
    \item If $v_3$ covers the whole space ($v_3 = V$), you are in 3D.
    \item If not, there is a $v_4$, and so on.
\end{itemize}
\begin{definition}[Finite Dimensional Vector Space]
    A vector space \(V\) over a field $k$ is finite-dimensional if there exists a finite set of vectors \(v_1,\dots,v_n \in V\) such that
    \[V = \Span(v_1,\dots,v_n).\]
    The dimension of \(V\) is the minimal such \(n\):
    \[\dim_kV = \min \{n : V = \Span_k(v_1,\dots,v_n)\}.\]
\end{definition}
\begin{example}[Complex numbers as a real vector space]
    Consider $\mathbb{C}$ as a vector space over $\mathbb{R}$. A standard basis is
    \[\{1, i\},\]
    so $\dim_{\mathbb{R}} \mathbb{C} = 2$. Any $z \in \mathbb{C}$ can be written uniquely as
    \[z = x \cdot 1 + y \cdot i, \quad x,y \in \mathbb{R}.\]
    Now consider the space of $n \times n$ complex matrices $M_n(\mathbb{C})$.
    \[A = \begin{pmatrix} z_{11} & z_{12} & \cdots & z_{1n}\\ \vdots & \vdots & \ddots & \vdots\\ z_{n1} & z_{n2} & \cdots & z_{nn} \end{pmatrix} = \begin{pmatrix} x_{11} + y_{11}i & x_{12} + y_{12}i & \cdots & x_{1n} + y_{1n}i\\ \vdots & \vdots & \ddots & \vdots\\ x_{n1} + y_{n1}i & x_{n2} + y_{n2}i & \cdots & x_{nn} + y_{nn}i \end{pmatrix}\]
    Hence a standard basis over $\mathbb{C}$ consists of the matrices $E_{ij}$ with a 1 in the $(i,j)$-th entry and 0 elsewhere, giving
    \[\dim_{\mathbb{C}} M_n(\mathbb{C}) = n^2.\]
    Over $\mathbb{R}$, each $E_{ij}$ splits into two real matrices corresponding to the real and imaginary parts, so
    \[\dim_{\mathbb{R}} M_n(\mathbb{C}) = 2 n^2.\]
\end{example}
\begin{example}[Infinite-dimensional space of polynomials]
    Let
    \[P_\infty(\mathbb{R}) = \{ a_0 + a_1 x + a_2 x^2 + \cdots : a_i \in \mathbb{R} \}\]

    Thus $P_\infty(\mathbb{R})$ is infinite-dimensional, and a standard basis is
    \[\{ 1, x, x^2, x^3, \dots \}.\]
\end{example}

\subsection{Linear Independence}
\begin{definition}[Linear Independence]
    $v_1,\ldots,v_n$ are linearly independent if $\lambda_1v_1 + \ldots + \lambda_nv_n = 0 \implies \lambda_1 = \ldots = \lambda_n = 0$.
\end{definition}
\begin{example}
    Let $V = \mathbb{R}^2$. Define
    \[e_1 = \begin{pmatrix}1\\0\end{pmatrix} \quad e_2 = \begin{pmatrix}0\\1\end{pmatrix}\]
    Prove $e_1, e_2$ are linearly independent.\\
    Assume $xe_1 + ye_2 = 0$, then prove $x = y = 0$.
    \[xe_1 + ye_2 = x\begin{pmatrix}1\\0\end{pmatrix} + y\begin{pmatrix}0\\1\end{pmatrix} = \begin{pmatrix}x\\y\end{pmatrix} = 0 \implies x = y = 0.\]
\end{example}
\marginpar{January 28, 2026}
\begin{problem}[Linearly independent vectors]
    \[v_1 = \begin{pmatrix}1\\1\\0\end{pmatrix}, \quad v_2 = \begin{pmatrix}1\\3\\2\end{pmatrix}, \quad v_3 = \begin{pmatrix}4\\9\\5\end{pmatrix}.\]
    Are these vectors linearly independent? Are there $x,y,z \in \mathbb{R}$ such that 
    \[xv_1 + yv_2 + zv_3 = 0?\]
\end{problem}
\begin{definition}[Linear Dependence]
    Linearly dependent (LD) means the opposite of linearly independent(LI).
\end{definition}
\begin{proposition}
    $v_1,\ldots,v_n$ are linearly dependent if and only if one vector is in the span of the other vectors.
\end{proposition}
\begin{example}
    \[3v_1 + 5v_2 - 2v_3 = 0\]
    As we can see, $v_1$ is in the span of $v_2$ and $v_3$.
    \[v_1 = \frac{1}{3}(-5v_2+2v_3)\]
\end{example}
\begin{remark}
    If one vector $v_1 = 0$ then the family is LD.
\end{remark}
\begin{proof}
    $(\Rightarrow)$ Assume that the vectors $v_1,\ldots,v_n$ are linearly dependent. Then there exist scalars $\lambda_1,\ldots,\lambda_n$, not all zero, such that
    \[\lambda_1v_1+\cdots+\lambda_nv_n=0.\]
    At least one coefficient is nonzero. Suppose $\lambda_k\neq 0$. Solving for $v_k$ gives
    \[v_k=\frac{-1}{\lambda_k}\sum_{i\ne k}\lambda_i v_i.\]
    Thus $v_k$ can be written as a linear combination of the remaining vectors, so
    \[v_k\in \operatorname{span}(v_1,\ldots,v_{k-1},v_{k+1},\ldots,v_n).\]
    $(\Leftarrow)$ Conversely, assume that for some $k$ we have
    \[v_k\in \operatorname{span}(v_1,\ldots,v_{k-1},v_{k+1},\ldots,v_n).\]
    Then there exist scalars $\lambda_i$ such that
    \[v_k=\sum_{i\ne k}\lambda_i v_i.\]
    Rewriting gives
    \[0=\sum_{i\ne k}\lambda_i v_i - v_k.\] 
    This is a nontrivial linear relation among $v_1,\ldots,v_n$, so the vectors are linearly dependent.
\end{proof}
\begin{example}
    Let
    \[V=\mathbb{R}^3.\]
    Consider the vectors
    \[v_1=\begin{pmatrix}a_1\\0\\0\end{pmatrix},\quad v_2=\begin{pmatrix}b_1\\b_2\\0\end{pmatrix},\quad v_3=\begin{pmatrix}c_1\\c_2\\c_3\end{pmatrix}.\]
    To test linear independence, assume
    \[xv_1+yv_2+zv_3=0.\]
    Substituting the vectors gives
    \[x\begin{pmatrix}a_1\\0\\0\end{pmatrix}+y\begin{pmatrix}b_1\\b_2\\0\end{pmatrix}+z\begin{pmatrix}c_1\\c_2\\c_3\end{pmatrix}=\begin{pmatrix}0\\0\\0\end{pmatrix}.\]
    This yields the system
    \[\begin{cases}a_1x+b_1y+c_1z=0\\b_2y+c_2z=0\\c_3z=0\end{cases}\]
    From the third equation, if $c_3\neq0$ then $z=0$. Substituting into the second equation gives $b_2y=0$, so if $b_2\neq0$ then $y=0$. Substituting into the first equation gives $a_1x=0$, so if $a_1\neq0$ then $x=0$. Hence if
    \[a_1\neq0,\quad b_2\neq0,\quad c_3\neq0,\]
    the only solution is $x=y=z=0$, so the vectors are linearly independent.
    Example of linearly independent vectors:
    \[v_1=\begin{pmatrix}1\\0\\0\end{pmatrix},\quad v_2=\begin{pmatrix}1\\2\\0\end{pmatrix},\quad v_3=\begin{pmatrix}1\\2\\3\end{pmatrix}.\]
    Example of linearly dependent vectors:
    \[v_1=\begin{pmatrix}1\\0\\0\end{pmatrix},\quad v_2=\begin{pmatrix}1\\2\\0\end{pmatrix},\quad v_3=\begin{pmatrix}1\\3\\0\end{pmatrix}.
    \]
    In the second case, the third coordinate of every vector is zero, so the vectors lie in a $2$-dimensional subspace of $\mathbb{R}^3$, implying linear dependence.
\end{example}

\subsection{Main Basis Theorem}

\begin{definition}[Basis]
    Let $V$ be a vector space over a field $k$. A set of vectors
    \[(v_1,\ldots,v_n)\]
    is called a basis of $V$ if it spans $V$, and it is linearly independent.
\end{definition}
\begin{remark}
    If $V$ admits a basis with $n$ vectors, then the dimension of $V$ is
    \[\dim(V) = n = \#\text{(basis)}.\]
\end{remark}
\marginpar{January 30, 2026}
\begin{example}[Standard basis of $\mathbb{R}^3$]
    The standard basis of $\mathbb{R}^3$ is
    \[e_1=\begin{pmatrix}1\\0\\0\end{pmatrix},\quad e_2=\begin{pmatrix}0\\1\\0\end{pmatrix},\quad e_3=\begin{pmatrix}0\\0\\1\end{pmatrix}.\]
    Thus a basis of $\mathbb{R}^3$ is
    \[\{e_1,e_2,e_3\}=\left\{\begin{pmatrix}1\\0\\0\end{pmatrix},\begin{pmatrix}0\\1\\0\end{pmatrix},\begin{pmatrix}0\\0\\1\end{pmatrix}\right\}.\]
\end{example}
\begin{proof}
    Let
    \[v=\begin{pmatrix}x\\y\\z\end{pmatrix}\in\mathbb{R}^3.\]
    Then
    \[v=xe_1+ye_2+ze_3,\]
    so $\{e_1,e_2,e_3\}$ spans $\mathbb{R}^3$. Now assume
    \[xe_1+ye_2+ze_3=\begin{pmatrix}0\\0\\0\end{pmatrix}.\]
    Comparing coordinates gives $x=0$, $y=0$, and $z=0$. Hence the vectors are linearly independent. Therefore $\{e_1,e_2,e_3\}$ is a basis of $\mathbb{R}^3$.
\end{proof}
\begin{lemma}
    Let $(v_1,\ldots,v_n)\subset \mathbb{R}^k$.
    \begin{itemize}
        \item If $n<k$, the vectors cannot span $\mathbb{R}^k$.
        \item If $n>k$, the vectors cannot be linearly independent.
    \end{itemize}
\end{lemma}
\begin{lemma}
    Let $V$ be a vector space. If $u_1,\ldots,u_n$ span $V$ and $v_1,\ldots,v_m$ are linearly independent in $V$, then
    \[n \ge m.\]
    In particular, any spanning set of $V$ must contain at least as many vectors as any linearly independent set.
\end{lemma}
\begin{theorem}[Main Basis Theorem]
    Let $(v_1,\ldots,v_n)$ be vectors in a vector space $V$. The following statements are equivalent:
    \begin{enumerate}
        \item Basis: $(v_1,\ldots,v_n)$ span V and are LI.
        \item Minimal Span: $(v_1,\ldots,v_n)$ span $V$ and no set of $m<n$ vectors spans $V$.
        \item Maximal LI: $(v_1,\ldots,v_n)$ are LI and no set of $m>n$ vectors are LI.
        \item Unique coordinates: Every vectors $v \in V$ can be written as a sum: $v = \sum_{i=1}^{n}\lambda_iv_i$, in a unique way.
    \end{enumerate}
\end{theorem}
\begin{corollary}[Coordinate Vectors]
    Every vector $v\in V$ can be written uniquely as
    \[v = x_1v_1+\cdots+x_nv_n.\]
    For a vector $v\in V$, the vector
    \[[v]_e=\begin{pmatrix}x_1\\\vdots\\x_n\end{pmatrix}\]
    is called the coordinate vector of $v$ with respect to the basis $(v_1,\ldots,v_n)$.
\end{corollary}
\begin{problem}[Coordinate Vectors]
    Let $X = \{-2, -1, 0, 1, 2\}$ and let $U = \{f \in F(X) : f(-x) - f(x) = 0 \text{ for every } x \in X\}$. Consider the following functions defined on $X$:
    \[g_1(x) = 1, \quad g_2(x) = \frac{1}{4}x^2(1-x^2), \quad g_3(x) = \frac{1}{3}x^2(4-x^2)\]
    for every $x \in X$.Prove that $S = (g_1, g_2, g_3)$ is a basis of $U$.Let $g$ be the function defined on $X$ as follows:
    \[g(2) = g(-2) = 3; \quad g(1) = g(-1) = g(0) = 2.\]
    Find the coordinate vector $[g]_S$ in the basis $S$.
\end{problem}
\begin{problem}[Coordinate Vectors 2]
    Let $D = \{1, 2, 3\}$, and let $F(D)$ denote the vector space of real-valued functions defined on $D$. Find the coordinates of the function $f(x) = 2x^2 - 3$ in the standard basis.
\end{problem}
\begin{corollary}
    If $(e_1,\ldots,e_n)$ and $(f_1,\ldots,f_m)$ are two basis of $V$, then $n=m$.
\end{corollary}
\begin{proof}
    Let $(e_1,\ldots,e_n)$ and $(f_1,\ldots,f_m)$ be bases of $V$. Since $(e_1,\ldots,e_n)$ spans $V$ and $(f_1,\ldots,f_m)$ is linearly independent, the lemma implies $n\ge m$. Reversing the roles of the two bases gives $m\ge n$. Hence $n=m$.
\end{proof}

\begin{enumerate}

\subsection{Proving a set forms a basis}
Let $V$ be finite-dimensional.
\begin{method}[Dimension Test]
    A list $(v_1,\ldots,v_n)$ is a basis of $V$ if and only if
    \[\dim(\Span(v_1,\ldots,v_n)) = n.\]
    Compute the dimension using row reduction.
\end{method}
\begin{proposition}[Basis Criterion for $n$-dimensional Space]
    Let $\dim V = n$ and $v_1,\ldots,v_n \in V$. The following are equivalent:
    \begin{enumerate}
        \item $(v_1,\ldots,v_n)$ is a basis,
        \item $(v_1,\ldots,v_n)$ spans $V$,
        \item $(v_1,\ldots,v_n)$ is linearly independent.
    \end{enumerate}
\end{proposition}
\begin{proof}
    A spanning set of $n = \dim V$ vectors is minimal, hence linearly independent.  
    An independent set of $n$ vectors is maximal, hence spans $V$.
\end{proof}
\begin{principle}[Transfer / Exchange Principle]
    If 
    \[u_1,\ldots,u_n \text{ span } V \quad\text{and}\quad v_1,\ldots,v_n \text{ are linearly independent},\] 
    then both lists are bases of $V$.
\end{principle}
\begin{example}
    In $P_n$,
    \[(1,x,\ldots,x^n) \text{ spans}, \qquad ((x-a)^0,\ldots,(x-a)^n) \text{ is linearly independent},\]
    so both are bases of $P_n$.
\end{example}

\subsection{Unique sum decompositions}
Let $W$ be a vector space and $U,V \subseteq W$ subspaces.
\begin{definition}[Direct Sum]
    The sum of subspaces is
    \[U+V=\{u+v : u\in U,\, v\in V\}.\]
    We say the sum is direct, written $U \oplus V$, if
    \[U \cap V = \{0\}.\]
\end{definition}
\begin{proposition}
    The following are equivalent:
    \begin{enumerate}
        \item $U \oplus V$ (i.e.\ $U \cap V = \{0\}$),
        \item every $w \in U+V$ can be written uniquely as $w=u+v$ with $u\in U$, $v\in V$.
    \end{enumerate}
\end{proposition}
\begin{proof}
    ($\Rightarrow$) Suppose $w=u_1+v_1=u_2+v_2$. Then
    \[u_1-u_2 = v_2-v_1 \in U \cap V = \{0\},\]
    so $u_1=u_2$ and $v_1=v_2$.

    ($\Leftarrow$) If decomposition is unique and $w\in U\cap V$, then
    \[0 = 0+0 = w+(-w),\]
    so uniqueness forces $w=0$.
\end{proof}
\begin{theorem}[Basis Concatenation]
    Suppose $U \oplus U'$ in $V$. Let $(e_1,\ldots,e_n)$ be a basis of $U$ and $(f_1,\ldots,f_m)$ a basis of $U'$. Then 
    \[(e_1,\ldots,e_n,f_1,\ldots,f_m)\]
    is a basis of $U \oplus U'$, and 
    \[\dim(U \oplus U') = \dim U + \dim U'.\]
\end{theorem}
\begin{proof}
    LI: $a_1e_1 + \ldots + a_ne_n + b_1f_1 + \ldots + b_mf_m = 0$.  
    Set $w = a_1e_1 + \ldots + a_ne_n = - (b_1f_1 + \ldots + b_mf_m) \in U \cap U'$.  
    Since $U \cap U' = \{0\}$, all $a_i = b_i = 0$.  
    Span: $v \in U+U' \implies v = u + u'$.
\end{proof}
\begin{theorem}[Dimension of Sum of Subspaces]
    For $U, U' \subset V$,
    \[\dim(U+U') = \dim U + \dim U' - \dim(U \cap U').\]
\end{theorem}
\begin{corollary}
    If $U \subseteq V$, then $\dim U \leq \dim V$.
\end{corollary}
\begin{corollary}
    If $U \subseteq V$, then $U$ admits a supplement: $\exists U' \subseteq V$ s.t. $V = U \oplus U'$.  
    Moreover, $\dim V = \dim U + \dim U'$.
\end{corollary}

\subsection{More Basis Theorems}
\begin{theorem}[Basis Extension]
    Let $V$ be finite-dimensional, and $v_1,\ldots,v_m$ LI in $V$. Then $v_1,\ldots,v_m$ can be extended to a basis of $V$.
\end{theorem}
\begin{proof}
    Let $U = \Span(v_1,\ldots,v_m) \subseteq V$.
    
    Step 1:  
    Case 1: $U = V \implies$ done.  
    Case 2: $U \neq V \implies \exists v_{m+1} \in V \setminus U$. Then $(v_1,\ldots,v_{m+1})$ is LI. Let $U_{m+1} = \Span(v_1,\ldots,v_{m+1})$.

    Step 2:  
    Case 1: $U_{m+1} = V \implies$ done.  
    Case 2: $U_{m+1} \neq V \implies$ continue adding $v_{m+2}$.

    Step 3: Continue until $v_n$, $n = \dim V$, gives $n$ LI vectors in $V$.
\end{proof}
\begin{theorem}[Basis Extraction / Nose Basis Theorem]
    Let $v_1,\ldots,v_n$ span $V$. Then one can extract a basis $\{e_1,\ldots,e_m\}$ from $\{v_1,\ldots,v_n\}$.
\end{theorem}
\begin{example}
    $v_1 = \begin{pmatrix}1\\0\end{pmatrix}, v_2 = \begin{pmatrix}0\\1\end{pmatrix}, v_3 = \begin{pmatrix}1\\1\end{pmatrix}$ spans $\mathbb{R}^3$.
\end{example}

The vectors \(e_1,\ldots,e_n\) span \(\mathbb{R}^n\).

\end{proof}
\paragraph{Claim 2.}
The vectors \(e_1,\ldots,e_n\) are linearly independent.
\begin{proof}
    Suppose
    \[\lambda_1 e_1 + \cdots + \lambda_n e_n = 0.\]
    Then
    \[\begin{pmatrix}\lambda_1\\ \vdots \\ \lambda_n\end{pmatrix} = \begin{pmatrix}0\\ \vdots \\ 0\end{pmatrix},\]
    so \(\lambda_1=\cdots=\lambda_n=0\). Therefore \(e_1,\ldots,e_n\) are linearly independent.
\end{proof}

\subsection{Algebraic Structures: Rings, Fields, and Vector Spaces}
\paragraph{Rings vs Vector Spaces.} Although rings and vector spaces both have addition and a zero element, they differ fundamentally in how multiplication works:

\begin{itemize}
    \item \textbf{Ring:} A set $R$ with addition and multiplication between elements of $R$. Addition forms an abelian group, multiplication is associative, and distributes over addition. Scalars are elements \emph{inside the set}.
    \item \textbf{Vector Space:} A set $V$ with addition and multiplication by scalars from a \emph{field} $F$. Addition forms an abelian group, scalar multiplication distributes appropriately. There is no multiplication between vectors themselves; only scalar multiplication is defined.
\end{itemize}

\end{solution}
